15pub fn authority_keys(
16	aura_pub_key: &str,
17	grandpa_pub_key: &str,
18	sidechain_pub_key: &str,
19) -> AuthorityKeys {
20	let aura_pk = sr25519::Public::from_raw(from_hex(aura_pub_key).unwrap().try_into().unwrap());
21	let granda_pk =
22		ed25519::Public::from_raw(from_hex(grandpa_pub_key).unwrap().try_into().unwrap());
23	let sidechain_pk = sidechain_domain::SidechainPublicKey(from_hex(sidechain_pub_key).unwrap());
24
25	let session_keys = (aura_pk, granda_pk).into();
26	AuthorityKeys { session: session_keys, cross_chain: sidechain_pk.try_into().unwrap() }
27}

105			staging_endowed_accounts(),

107		)?)
108		.build())
109}

111/// Configure initial storage state for FRAME modules.
112pub fn staging_genesis(
113	initial_authorities: Vec<AuthorityKeys>,
114	root_key: Option<AccountId>,
115	endowed_accounts: Vec<AccountId>,
116	_enable_println: bool,
117) -> Result<serde_json::Value, envy::Error> {
118	let genesis_utxo = sp_sidechain::read_genesis_utxo_from_env_with_defaults()?;
119	let config = RuntimeGenesisConfig {
120		system: SystemConfig { ..Default::default() },
121		balances: BalancesConfig {
122			// Configure endowed accounts with initial balance of 1 << 60.
123			balances: endowed_accounts.iter().cloned().map(|k| (k, 1 << 60)).collect(),
124			dev_accounts: None,
125		},
126		aura: AuraConfig { authorities: vec![] },
127		grandpa: GrandpaConfig { authorities: vec![], ..Default::default() },
128		sudo: SudoConfig {
129			// Assign network admin rights.
130			key: root_key,
131		},
132		transaction_payment: Default::default(),
133		session: SessionConfig {
134			initial_validators: initial_authorities
135				.iter()
136				.map(|authority_keys| {
137					(authority_keys.cross_chain.clone().into(), authority_keys.session.clone())
138				})
139				.collect(),
140		},
141		sidechain: SidechainConfig { genesis_utxo, ..Default::default() },
142		pallet_session: Default::default(),
143		session_committee_management: SessionCommitteeManagementConfig {
144			initial_authorities: initial_authorities
145				.into_iter()
146				.map(|keys| CommitteeMember::permissioned(keys.cross_chain, keys.session))
147				.collect(),
148			main_chain_scripts: sp_session_validator_management::MainChainScripts::read_from_env()?,
149		},
150		native_token_management: NativeTokenManagementConfig {
151			main_chain_scripts: Some(sp_native_token_management::MainChainScripts::read_from_env()?),
152			..Default::default()
153		},
154		governed_map: GovernedMapConfig {
155			main_chain_scripts: Some(sp_governed_map::MainChainScriptsV1::read_from_env()?),
156			..Default::default()
157		},
158		test_helper_pallet: TestHelperPalletConfig {
159			participation_data_release_period: 30,
160			..Default::default()
161		},
162		bridge: BridgeConfig {
163			main_chain_scripts: Some(sp_partner_chains_bridge::MainChainScripts::read_from_env()?),
164			initial_checkpoint: Some(genesis_utxo),
165			..Default::default()
166		},
167	};
168
169	Ok(serde_json::to_value(config).expect("Genesis config must be serialized correctly"))
170}
