22#[derive(Clone)]
23pub struct DataSources {
24	pub mc_hash: Arc<dyn McHashDataSource + Send + Sync>,
25	pub authority_selection: Arc<dyn AuthoritySelectionDataSource + Send + Sync>,
26	pub native_token: Arc<dyn NativeTokenManagementDataSource + Send + Sync>,
27	pub sidechain_rpc: Arc<dyn SidechainRpcDataSource + Send + Sync>,
28	pub block_participation: Arc<dyn BlockParticipationDataSource + Send + Sync>,
29	pub governed_map: Arc<dyn GovernedMapDataSource + Send + Sync>,
30}
31
32pub(crate) async fn create_cached_data_sources(
33	metrics_opt: Option<McFollowerMetrics>,
34) -> std::result::Result<DataSources, ServiceError> {
35	if use_mock_follower() {
36		create_mock_data_sources().map_err(|err| {
37			ServiceError::Application(
38				format!("Failed to create mock data sources: {err}. Check configuration.").into(),
39			)
40		})
41	} else {
42		create_cached_db_sync_data_sources(metrics_opt).await.map_err(|err| {
43			ServiceError::Application(
44				format!("Failed to create db-sync data sources: {err}").into(),
45			)
46		})
47	}
48}
49
50fn use_mock_follower() -> bool {
51	std::env::var("USE_MOCK_DATA_SOURCES")
52		.ok()
53		.and_then(|v| v.parse::<bool>().ok())
54		.unwrap_or(false)
55}

57pub fn create_mock_data_sources(
58) -> std::result::Result<DataSources, Box<dyn Error + Send + Sync + 'static>> {
59	let block = Arc::new(BlockDataSourceMock::new_from_env()?);
60	Ok(DataSources {
61		sidechain_rpc: Arc::new(SidechainRpcDataSourceMock::new(block.clone())),
62		mc_hash: Arc::new(McHashDataSourceMock::new(block)),
63		authority_selection: Arc::new(AuthoritySelectionDataSourceMock::new_from_env()?),
64		native_token: Arc::new(NativeTokenDataSourceMock::new()),
65		block_participation: Arc::new(StakeDistributionDataSourceMock::new()),
66		governed_map: Arc::new(GovernedMapDataSourceMock::new([].into())),
67	})
68}

70pub const CANDIDATES_FOR_EPOCH_CACHE_SIZE: usize = 64;
71pub const STAKE_CACHE_SIZE: usize = 100;
72
73pub async fn create_cached_db_sync_data_sources(
74	metrics_opt: Option<McFollowerMetrics>,
75) -> Result<DataSources, Box<dyn Error + Send + Sync + 'static>> {
76	let pool = partner_chains_db_sync_data_sources::data_sources::get_connection_from_env().await?;
77	// block data source is reused between mc_hash and sidechain_rpc to share cache
78	let block = Arc::new(BlockDataSourceImpl::new_from_env(pool.clone()).await?);
79	Ok(DataSources {
80		sidechain_rpc: Arc::new(SidechainRpcDataSourceImpl::new(
81			block.clone(),
82			metrics_opt.clone(),
83		)),
84		mc_hash: Arc::new(McHashDataSourceImpl::new(block, metrics_opt.clone())),
85		authority_selection: Arc::new(
86			CandidatesDataSourceImpl::new(pool.clone(), metrics_opt.clone())
87				.await?
88				.cached(CANDIDATES_FOR_EPOCH_CACHE_SIZE)?,
89		),
90		native_token: Arc::new(NativeTokenManagementDataSourceImpl::new_from_env(
91			pool.clone(),
92			metrics_opt.clone(),
93		)?),
94		block_participation: Arc::new(StakeDistributionDataSourceImpl::new(
95			pool,
96			metrics_opt,
97			STAKE_CACHE_SIZE,
98		)),
99		governed_map: Arc::new(GovernedMapDataSourceMock::new(
100			[("key1".into(), vec![1, 2, 3].into())].into(),
101		)),
102	})
103}
