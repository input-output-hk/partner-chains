1use crate::PaymentFilePath;
2use partner_chains_cardano_offchain::await_tx::FixedDelayRetries;
3use partner_chains_cardano_offchain::governed_map::run_insert;
4use sidechain_domain::byte_string::ByteString;

7#[derive(Clone, Debug, clap::Subcommand)]
8#[allow(clippy::large_enum_variant)]
9pub enum GovernedMapCmd {
10	/// Inserts a key-value pair into the Governed Gap. If a value for the key already exists it won't be updated.
11	///
12	/// NOTE: In rare cases, race conditions may occur, and two inserts with the same key will both succeed.
13	/// In that case the second one in terms of block and transaction number is considered valid.
14	Insert(InsertCmd),
15}
16
17impl GovernedMapCmd {
18	pub async fn execute(self) -> crate::SubCmdResult {
19		match self {
20			Self::Insert(cmd) => cmd.execute().await,
21		}
22	}
23}
24
25#[derive(Clone, Debug, clap::Parser)]
26pub struct InsertCmd {
27	#[clap(flatten)]
28	common_arguments: crate::CommonArguments,
29	#[arg(long)]
30	key: String,
31	#[arg(long)]
32	value: ByteString,
33	#[clap(flatten)]
34	payment_key_file: PaymentFilePath,
35	#[arg(long, short('g'))]
36	genesis_utxo: UtxoId,
37}
38
39impl InsertCmd {
40	pub async fn execute(self) -> crate::SubCmdResult {
41		let payment_key = self.payment_key_file.read_key()?;
42
43		let client = self.common_arguments.get_ogmios_client().await?;
44
45		let result = run_insert(
46			self.genesis_utxo,
47			self.key,
48			self.value,
49			&payment_key,
50			&client,
51			&FixedDelayRetries::five_minutes(),
52		)
53		.await?;
54		Ok(serde_json::json!(result))
55	}
56}
