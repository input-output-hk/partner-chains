8#[derive(Clone, Debug, clap::Subcommand)]
9#[allow(clippy::large_enum_variant)]
10pub enum GovernedMapCmd {
11	/// Inserts a key-value pair into the Governed Gap. If a value for the key already exists it won't be updated.
12	///
13	/// NOTE: In rare cases, race conditions may occur, and two inserts with the same key will both succeed.
14	/// In that case the second one in terms of block and transaction number is considered valid.
15	Insert(InsertCmd),
16	/// Lists all key-value pairs currently stored in the Governed Map
17	List(ListCmd),
18	/// Removes a key-value pair from the Governed Map
19	Remove(RemoveCmd),
20	/// Retrieves the value stored in the Governed Map for the given key
21	Get(GetCmd),
22}
23
24impl GovernedMapCmd {
25	pub async fn execute(self) -> crate::SubCmdResult {
26		match self {
27			Self::Insert(cmd) => cmd.execute().await,
28			Self::List(cmd) => cmd.execute().await,
29			Self::Remove(cmd) => cmd.execute().await,
30			Self::Get(cmd) => cmd.execute().await,
31		}
32	}
33}

35#[derive(Clone, Debug, clap::Parser)]
36pub struct InsertCmd {
37	#[clap(flatten)]
38	common_arguments: crate::CommonArguments,
39	#[arg(long)]
40	key: String,
41	#[arg(long)]
42	value: ByteString,
43	#[clap(flatten)]
44	payment_key_file: PaymentFilePath,
45	#[arg(long, short('g'))]
46	genesis_utxo: UtxoId,
47}
48
49impl InsertCmd {
50	pub async fn execute(self) -> crate::SubCmdResult {
51		let payment_key = self.payment_key_file.read_key()?;
52
53		let client = self.common_arguments.get_ogmios_client().await?;
54
55		let result = run_insert(
56			self.genesis_utxo,
57			self.key,
58			self.value,
59			&payment_key,
60			&client,
61			&FixedDelayRetries::five_minutes(),
62		)
63		.await?;
64		Ok(serde_json::json!(result))
65	}
66}

88impl ListCmd {
89	pub async fn execute(self) -> crate::SubCmdResult {
90		let client = self.common_arguments.get_ogmios_client().await?;
91		let kv_pairs: Vec<_> = run_list(self.genesis_utxo, &client)
92			.await?
93			.map(|datum| json!({"key": datum.key, "value": datum.value.to_hex_string()}))
94			.collect();
95
96		Ok(json!(kv_pairs))
97	}
98}

131		let Some(value) = run_get(self.genesis_utxo, self.key.clone(), &client).await? else {
