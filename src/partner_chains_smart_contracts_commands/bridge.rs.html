1use crate::{GenesisUtxo, PaymentFilePath};
2use partner_chains_cardano_offchain::bridge::{deposit_with_ics_spend, deposit_without_ics_input};
3use sp_runtime::AccountId32;
4
5#[derive(Clone, Debug, clap::Parser)]
6/// Command for sending a native token bridge transfer. Token to be sent is defined in the Reserve Settings.
7pub struct BridgeCmd {
8	#[clap(flatten)]
9	common_arguments: crate::CommonArguments,
10	#[arg(long)]
11	/// Number of tokens to transfer
12	amount: u64,
13	#[arg(long)]
14	/// Address in the partner chain to transfer the tokens, in hex format.
15	pc_address: AccountId32,
16	#[arg(long)]
17	/// When true, the transaction won't spend an UTXO from ICS Validator. This costs more ada, but is simpler.
18	simple: bool,
19	#[clap(flatten)]
20	/// Path to the payment key file
21	payment_key_file: PaymentFilePath,
22	#[clap(flatten)]
23	/// Genesis UTXO
24	genesis_utxo: GenesisUtxo,
25}
26
27impl BridgeCmd {
28	/// Creates the D-parameter and upserts it on the main chain.
29	pub async fn execute(self) -> crate::SubCmdResult {
30		let payment_key = self.payment_key_file.read_key()?;
31		let client = self.common_arguments.get_ogmios_client().await?;
32		let tx_hash = if self.simple {
33			deposit_without_ics_input(
34				self.genesis_utxo.into(),
35				self.amount,
36				self.pc_address.as_ref(),
37				&payment_key,
38				&client,
39				&self.common_arguments.retries(),
40			)
41			.await?
42		} else {
43			deposit_with_ics_spend(
44				self.genesis_utxo.into(),
45				self.amount,
46				self.pc_address.as_ref(),
47				&payment_key,
48				&client,
49				&self.common_arguments.retries(),
50			)
51			.await?
52		};
53		Ok(serde_json::json!(tx_hash))
54	}
55}
