13#[derive(Clone, Debug, clap::Subcommand)]
14#[allow(clippy::large_enum_variant)]
15pub enum ReserveCmd {
16	/// Initialize the reserve management system for your chain
17	Init(InitReserveCmd),
18	/// Creates the reserve for your chain. `init` and `create` will be merged into a single command in a future release.
19	Create(CreateReserveCmd),
20	/// Deposits tokens from payment key wallet to the reserve
21	Deposit(DepositReserveCmd),
22	/// Update reserve management system settings for your chain
23	UpdateSettings(UpdateReserveSettingsCmd),
24	/// Releases all the remaining funds from the reserve to the illiquid supply
25	Handover(HandoverReserveCmd),
26	/// Releases funds from the reserve to the illiquid supply
27	Release(ReleaseReserveCmd),
28}
29
30impl ReserveCmd {
31	pub async fn execute(self) -> crate::SubCmdResult {
32		match self {
33			Self::Init(cmd) => cmd.execute().await,
34			Self::Create(cmd) => cmd.execute().await,
35			Self::Deposit(cmd) => cmd.execute().await,
36			Self::UpdateSettings(cmd) => cmd.execute().await,
37			Self::Handover(cmd) => cmd.execute().await,
38			Self::Release(cmd) => cmd.execute().await,
39		}
40	}
41}

52
53impl InitReserveCmd {
54	pub async fn execute(self) -> crate::SubCmdResult {
55		let payment_key = self.payment_key_file.read_key()?;
56		let client = self.common_arguments.get_ogmios_client().await?;
57		let result = init_reserve_management(
58			self.genesis_utxo.into(),
59			&payment_key,
60			&client,
61			&self.common_arguments.retries(),
62		)
63		.await?;
64		Ok(serde_json::json!(result))
65	}
66}

68#[derive(Clone, Debug, clap::Parser)]
69pub struct CreateReserveCmd {
70	#[clap(flatten)]
71	common_arguments: crate::CommonArguments,
72	#[clap(flatten)]
73	payment_key_file: PaymentFilePath,
74	#[clap(flatten)]
75	genesis_utxo: GenesisUtxo,
76	/// Script hash of the 'total accrued function', also called V-function, that computes how many tokens could be released from the reserve at given moment.
77	#[arg(long)]
78	total_accrued_function_script_hash: ScriptHash,
79	/// Initial amount of tokens to deposit. They must be present in the payment wallet.
80	#[arg(long)]
81	initial_deposit_amount: u64,
82	/// Reserve token asset id encoded in form <policy_id_hex>.<asset_name_hex>.
83	#[arg(long)]
84	token: AssetId,
85}
86
87impl CreateReserveCmd {
88	pub async fn execute(self) -> crate::SubCmdResult {
89		let payment_key = self.payment_key_file.read_key()?;
90		let client = self.common_arguments.get_ogmios_client().await?;
91		let result = create_reserve_utxo(
92			ReserveParameters {
93				total_accrued_function_script_hash: self.total_accrued_function_script_hash,
94				token: self.token,
95				initial_deposit: self.initial_deposit_amount,
96			},
97			self.genesis_utxo.into(),
98			&payment_key,
99			&client,
100			&self.common_arguments.retries(),
101		)
102		.await?;
103		Ok(serde_json::json!(result))
104	}
105}
