1use crate::GetGenesisUtxo;
2use sidechain_domain::UtxoId;
3use sp_api::ProvideRuntimeApi;
4use sp_blockchain::HeaderBackend;
5use sp_runtime::traits::Block as BlockT;
6use std::sync::Arc;
7
8pub async fn get_genesis_utxo<B, C>(client: Arc<C>) -> Result<String, String>
9where
10	B: BlockT,
11	C: ProvideRuntimeApi<B> + Send + Sync + 'static,
12	C::Api: GetGenesisUtxo<B>,
13	C: HeaderBackend<B>,
14{
15	let api = client.runtime_api();
16	let best_block = client.info().best_hash;
17	let genesis_utxo = api.genesis_utxo(best_block).map_err(|err| err.to_string())?;
18	let output =
19		serde_json::to_string_pretty(&Output { genesis_utxo }).map_err(|err| err.to_string())?;
20	Ok(output)
21}
22
23#[derive(serde::Serialize, serde::Deserialize, Clone, PartialEq, Debug)]
24pub struct Output {
25	pub genesis_utxo: UtxoId,
26}
