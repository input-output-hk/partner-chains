17#[derive(Debug, Clone, thiserror::Error)]
18pub enum OgmiosClientError {
19	#[error("Couldn't construct parameters: '{0}'")]
20	ParametersError(String),
21	#[error("JsonRPC request failed: '{0}'")]
22	RequestError(String),
23	#[error("Could not parse response: '{0}'")]
24	ResponseError(String),
25}
26
27pub trait OgmiosClient {
28	#[allow(async_fn_in_trait)]
29	async fn request<T: DeserializeOwned>(
30		&self,
31		method: &str,
32		params: OgmiosParams,
33	) -> Result<T, OgmiosClientError>;
34}
35
36#[derive(Clone)]
37pub enum OgmiosParams {
38	Positional(Vec<serde_json::Value>),
39	ByName(HashMap<&'static str, serde_json::Value>),
40}
41
42impl OgmiosParams {
43	pub fn empty_positional() -> Self {
44		OgmiosParams::Positional(Vec::new())
45	}
46
47	pub fn empty_by_name() -> Self {
48		OgmiosParams::ByName(HashMap::new())
49	}
50}
51
52impl ToRpcParams for OgmiosParams {
53	fn to_rpc_params(self) -> Result<Option<Box<serde_json::value::RawValue>>, serde_json::Error> {
54		match self {
55			OgmiosParams::ByName(map) => {
56				let mut object_params = ObjectParams::new();
57				map.into_iter().try_for_each(|(k, v)| object_params.insert(k, v))?;
58				object_params.to_rpc_params()
59			},
60			OgmiosParams::Positional(v) => {
61				let mut array_params = ArrayParams::new();
62				v.into_iter().try_for_each(|v| array_params.insert(v))?;
63				array_params.to_rpc_params()
64			},
65		}
66	}
67}

69pub struct ByNameParamsBuilder {
70	params: HashMap<&'static str, serde_json::Value>,
71}
72
73impl Default for ByNameParamsBuilder {
74	fn default() -> Self {
75		Self::new()
76	}
77}
78
79impl ByNameParamsBuilder {
80	pub fn new() -> Self {
81		ByNameParamsBuilder { params: HashMap::new() }
82	}
83
84	pub fn insert<T: serde::Serialize>(
85		self,
86		key: &'static str,
87		value: T,
88	) -> Result<Self, OgmiosClientError> {
89		let value = serde_json::to_value(value)
90			.map_err(|e| OgmiosClientError::ParametersError(e.to_string()))?;
91		let mut params = self.params;
92		params.insert(key, value);
93		Ok(Self { params })
94	}
95
96	pub fn build(self) -> OgmiosParams {
97		OgmiosParams::ByName(self.params)
98	}
