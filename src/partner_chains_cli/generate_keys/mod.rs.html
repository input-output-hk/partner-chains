51impl<T: PartnerChainRuntime> CmdRun for GenerateKeysCmd<T> {
52	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
53		context.eprint(
54			"This 🧙 wizard will generate the following keys and save them to your node's keystore:",
55		);
56		context.eprint(&format!("→ {} {} key", CROSS_CHAIN.scheme, CROSS_CHAIN.name));
57		for key_def in T::key_definitions() {
58			context.eprint(&format!("→ {} {} key", key_def.scheme, key_def.name));
59		}
60		context.eprint("It will also generate a network key for your node if needed.");
61		context.enewline();
62
63		let chain_spec_path = write_temp_chain_spec(
64			context,
65			T::create_chain_spec(&CreateChainSpecConfig::<T::Keys>::default()),
66		);
67
68		let config = GenerateKeysConfig::load(context);
69		context.enewline();
70
71		generate_spo_keys::<C, T>(&config, &chain_spec_path, context)?;
72
73		context.enewline();
74
75		generate_network_key(&config, &chain_spec_path, context)?;
76		context.enewline();
77
78		context.eprint("🚀 All done!");
79		context.delete_file(&chain_spec_path)?;
80		Ok(())
81	}
82}

84fn write_temp_chain_spec<C: IOContext>(context: &C, chain_spec: serde_json::Value) -> String {
85	let dir_path = context.new_tmp_dir();
86	let dir_path = dir_path.to_str().expect("temp dir path is correct utf-8");
87	let path = format!("{dir_path}/chain-spec.json");
88	let content = format!("{chain_spec}");
89	context.write_file(&path, &content);
90	path

93pub(crate) fn generate_spo_keys<C: IOContext, T: PartnerChainRuntime>(
94	config: &GenerateKeysConfig,
95	chain_spec_path: &str,
96	context: &C,
97) -> anyhow::Result<()> {
98	if prompt_can_write("keys file", KEYS_FILE_PATH, context) {
99		let partner_chains_key =
100			generate_or_load_key(config, context, chain_spec_path, &CROSS_CHAIN)?;
101		context.enewline();
102		let mut keys: BTreeMap<String, ByteString> = BTreeMap::new();
103		for key_definition in T::key_definitions() {
104			let generated_key =
105				generate_or_load_key(config, context, chain_spec_path, &key_definition)?;
106			context.enewline();
107			keys.insert(key_definition.key_type.to_owned(), generated_key);
108		}
109
110		let public_keys_json =
111			serde_json::to_string_pretty(&PermissionedCandidateKeys { partner_chains_key, keys })
112				.expect("PermissionedCandidateKeys have only UTF-8 encodable ids");
113		context.write_file(KEYS_FILE_PATH, &public_keys_json);
114
115		context.eprint(&format!(
116			"🔑 The following public keys were generated and saved to the {} file:",
117			KEYS_FILE_PATH,
118		));
119		context.print(&(public_keys_json).to_string());
120		context.eprint("You may share them with your chain governance authority");
121		context.eprint("if you wish to be included as a permissioned candidate.");
122	} else {
123		context.eprint("Refusing to overwrite keys file - skipping");
124	}
125	Ok(())
126}

128pub(crate) fn generate_network_key<C: IOContext>(
129	config: &GenerateKeysConfig,
130	chain_spec_path: &str,
131	context: &C,
132) -> anyhow::Result<()> {
133	let maybe_existing_key =
134		context.read_file(&config.network_key_path()).as_deref().map(decode_network_key);
135	match maybe_existing_key {
136		Some(Ok(_)) => {
137			context.eprint(
138				"🔑 A valid network key is already present in the keystore, skipping generation",
139			);
140		},
141		None => {
142			context.eprint("⚙️ Generating network key");
143			run_generate_network_key(config, chain_spec_path, context)?;
144		},
145		Some(Err(err)) => {
146			context.eprint(&format!(
147				"⚠️ Network key in keystore is invalid ({}), wizard will regenerate it",
148				err,
149			));
150			context.eprint("⚙️ Regenerating the network key");
151			context.delete_file(&config.network_key_path())?;
152			run_generate_network_key(config, chain_spec_path, context)?;
153		},
154	};
155	Ok(())
156}

158fn run_generate_network_key<C: IOContext>(
159	config: &GenerateKeysConfig,
160	chain_spec_path: &str,
161	context: &C,
162) -> anyhow::Result<()> {
163	let node_executable = context.current_executable()?;
164	let network_key_directory = network_key_directory(&config.substrate_node_base_path);
165	let network_key_path = config.network_key_path();
166	context.run_command(&format!("mkdir -p {network_key_directory}"))?;
167	context.run_command(&format!(
168		"{node_executable} key generate-node-key --chain {chain_spec_path} --file {network_key_path}"
169	))?;
170	Ok(())
171}

173fn decode_network_key(key_str: &str) -> anyhow::Result<ed25519::Pair> {
174	hex::decode(key_str)
175		.context("Invalid hex")
176		.and_then(|slice| ed25519::Pair::from_seed_slice(&slice).context("Invalid ed25519 bytes"))
177}

179#[derive(Debug, Deserialize)]
180#[serde(rename_all = "camelCase")]
181struct KeyGenerationOutput {
182	public_key: String,
183	secret_phrase: String,
184}
185
186fn generate_keys<C: IOContext>(
187	context: &C,
188	KeyDefinition { scheme, name, .. }: &KeyDefinition,
189) -> anyhow::Result<KeyGenerationOutput> {
190	let executable = context.current_executable()?;
191	context.eprint(&format!("⚙️ Generating {name} ({scheme}) key"));
192	let output = context
193		.run_command(&format!("{executable} key generate --scheme {scheme} --output-type json"))?;
194
195	serde_json::from_str(&output)
196		.map_err(|_| anyhow!("Failed to parse generated keys json: {output}"))
197}

199fn store_keys<C: IOContext>(
200	context: &C,
201	GenerateKeysConfig { substrate_node_base_path: base_path }: &GenerateKeysConfig,
202	key_def: &KeyDefinition,
203	KeyGenerationOutput { secret_phrase, public_key }: &KeyGenerationOutput,
204	chain_spec_file_path: &str,
205) -> anyhow::Result<()> {
206	let node_executable = context.current_executable()?;
207	let KeyDefinition { scheme, key_type, name } = key_def;
208	context.eprint(&format!("💾 Inserting {name} ({scheme}) key"));
209	let keystore_path = keystore_path(base_path);
210	let cmd = format!(
211		"{node_executable} key insert --chain {chain_spec_file_path} --keystore-path {keystore_path} --scheme {scheme} --key-type {key_type} --suri '{secret_phrase}'"
212	);
213	let _ = context.run_command(&cmd)?;
214	let store_path = format!("{}/{}{public_key}", keystore_path, key_def.key_type_hex(),);
215	context.eprint(&format!("💾 {name} key stored at {store_path}",));
216	Ok(())
217}

219fn generate_or_load_key<C: IOContext>(
220	config: &GenerateKeysConfig,
221	context: &C,
222	chain_spec_path: &str,
223	key_def: &KeyDefinition,
224) -> anyhow::Result<ByteString> {
225	let keystore_path = config.keystore_path();
226	let existing_keys = context.list_directory(&keystore_path)?.unwrap_or_default();
227
228	let key: anyhow::Result<String> = if let Some(key) = find_existing_key(&existing_keys, key_def)
229	{
230		if context.prompt_yes_no(
231			&format!("A {} key already exists in store: {key} - overwrite it?", key_def.name),
232			false,
233		) {
234			let new_key = generate_keys(context, key_def)?;
235			store_keys(context, config, key_def, &new_key, chain_spec_path)?;
236
237			let old_key_path = format!("{keystore_path}/{}{key}", key_def.key_type_hex());
238			context
239				.delete_file(&old_key_path)
240				.context(format!("Failed to overwrite {} key at {old_key_path}", key_def.name))?;
241
242			Ok(new_key.public_key)
243		} else {
244			Ok(format!("0x{key}"))
245		}
246	} else {
247		let new_key = generate_keys(context, key_def)?;
248		store_keys(context, config, key_def, &new_key, chain_spec_path)?;
249
250		Ok(new_key.public_key)
251	};
252	ByteString::decode_hex(&key?).map_err(|e| anyhow!(e))
