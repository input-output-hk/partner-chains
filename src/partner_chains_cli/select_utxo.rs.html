11fn utxo_id(utxo: &OgmiosUtxo) -> UtxoId {
12	UtxoId {
13		tx_hash: McTxHash(utxo.transaction.id),
14		index: sidechain_domain::UtxoIndex(utxo.index),
15	}
16}
17
18fn to_display_string(utxo: &OgmiosUtxo) -> String {
19	format!("{0} ({1} lovelace)", utxo_id(utxo), utxo.value.lovelace)
20}

22pub(crate) fn query_utxos<C: IOContext>(
23	context: &C,
24	ogmios_config: &ServiceConfig,
25	address: &str,
26) -> Result<Vec<OgmiosUtxo>, anyhow::Error> {
27	let ogmios_addr = ogmios_config.url();
28	context.print(&format!("⚙️ Querying UTXOs of {address} from Ogmios at {ogmios_addr}..."));
29	let response = context
30		.ogmios_rpc(&ogmios_config, OgmiosRequest::QueryUtxo { address: address.into() })
31		.map_err(|e| anyhow!(e))?;
32	match response {
33		OgmiosResponse::QueryUtxo(utxos) => Ok(utxos),
34		other => Err(anyhow::anyhow!(format!(
35			"Unexpected response from Ogmios when querying for utxos: {other:?}"
36		))),
37	}
38}

40pub(crate) fn select_from_utxos<C: IOContext>(
41	context: &C,
42	prompt: &str,
43	utxos: Vec<OgmiosUtxo>,
44) -> Result<UtxoId, anyhow::Error> {
45	let utxo_display_options: Vec<String> =
46		utxos.iter().map(|utxo| to_display_string(utxo)).collect();
47	let selected_utxo_display_string = context.prompt_multi_option(prompt, utxo_display_options);
48	let selected_utxo = utxos
49		.iter()
50		.find(|utxo| to_display_string(utxo) == selected_utxo_display_string)
51		.map(|utxo| utxo_id(utxo).to_string())
52		.ok_or_else(|| anyhow!("⚠️ Failed to find selected UTXO"))?;
53	UtxoId::from_str(&selected_utxo).map_err(|e| {
54		context.eprint(&format!("⚠️ Failed to parse selected UTXO: {e}"));
55		anyhow!(e)
56	})
57}
