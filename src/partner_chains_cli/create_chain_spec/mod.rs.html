32	"/genesis/runtimeGenesis/config/session/initialValidators";

36impl<T: CreateChainSpecRuntimeBindings> CmdRun for CreateChainSpecCmd<T> {
37	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
38		let config = CreateChainSpecConfig::load(context)?;
39		context.print("This wizard will create a chain spec JSON file according to the provided configuration, using WASM runtime code from the compiled node binary.");
40		Self::print_config(context, &config);
41		if context.prompt_yes_no("Do you want to continue?", true) {
42			Self::run_build_spec_command(context, &config)?;
43			Self::update_chain_spec_authorities(context, &config)?;
44			context.print("chain-spec.json file has been created.");
45			context.print(
46				"If you are the governance authority, you can distribute it to the validators.",
47			);
48			context.print("Run 'setup-main-chain-state' command to set D-parameter and permissioned candidates on Cardano.");
49			Ok(())
50		} else {
51			context.print("Aborted.");
52			Ok(())
53		}
54	}
55}

57impl<T: CreateChainSpecRuntimeBindings> CreateChainSpecCmd<T> {
58	fn print_config<C: IOContext>(context: &C, config: &CreateChainSpecConfig) {
59		context.print("Chain parameters:");
60		context.print(format!("- Genesis UTXO: {}", config.genesis_utxo).as_str());
61		context.print("SessionValidatorManagement Main Chain Configuration:");
62		context.print(
63			format!("- committee_candidate_address: {}", config.committee_candidate_address)
64				.as_str(),
65		);
66		context
67			.print(format!("- d_parameter_policy_id: {}", config.d_parameter_policy_id).as_str());
68		context.print(
69			format!(
70				"- permissioned_candidates_policy_id: {}",
71				config.permissioned_candidates_policy_id
72			)
73			.as_str(),
74		);
75		context.print("Native Token Management Configuration (unused if empty):");
76		context.print(&format!("- asset name: {}", config.native_token_asset_name));
77		context.print(&format!("- asset policy ID: {}", config.native_token_policy));
78		context.print(&format!("- illiquid supply address: {}", config.illiquid_supply_address));
79		use colored::Colorize;
80		if config.initial_permissioned_candidates_raw.is_empty() {
81			context.print("WARNING: The list of initial permissioned candidates is empty. Generated chain spec will not allow the chain to start.".red().to_string().as_str());
82			let update_msg = format!(
83				"Update 'initial_permissioned_candidates' field of {} file with keys of initial committee.",
84				config_fields::INITIAL_PERMISSIONED_CANDIDATES.config_file
85			);
86			context.print(update_msg.red().to_string().as_str());
87			context.print(INITIAL_PERMISSIONED_CANDIDATES_EXAMPLE.yellow().to_string().as_str());
88		} else {
89			context.print("Initial permissioned candidates:");
90			for candidate in config.initial_permissioned_candidates_raw.iter() {
91				context.print(format!("- {}", candidate).as_str());
92			}
93		}
94	}

96	fn run_build_spec_command<C: IOContext>(
97		context: &C,
98		config: &CreateChainSpecConfig,
99	) -> anyhow::Result<String> {
100		let node_executable = context.current_executable()?;
101		context.set_env_var("GENESIS_UTXO", &config.genesis_utxo.to_string());
102		context.set_env_var(
103			"COMMITTEE_CANDIDATE_ADDRESS",
104			&config.committee_candidate_address.to_string(),
105		);
106		context.set_env_var("D_PARAMETER_POLICY_ID", &config.d_parameter_policy_id.to_string());
107		context.set_env_var(
108			"PERMISSIONED_CANDIDATES_POLICY_ID",
109			&config.permissioned_candidates_policy_id.to_string(),
110		);
111		context.set_env_var("NATIVE_TOKEN_POLICY_ID", &config.native_token_policy);
112		context.set_env_var("NATIVE_TOKEN_ASSET_NAME", &config.native_token_asset_name);
113		context.set_env_var("ILLIQUID_SUPPLY_VALIDATOR_ADDRESS", &config.illiquid_supply_address);
114		context.run_command(
115			format!("{node_executable} build-spec --disable-default-bootnode > chain-spec.json")
116				.to_string()
117				.as_str(),
118		)
119	}

121	fn update_chain_spec_authorities<C: IOContext>(
122		context: &C,
123		config: &CreateChainSpecConfig,
124	) -> anyhow::Result<()> {
125		let json = context
126			.read_file("chain-spec.json")
127			.context("Could not read chain-spec.json file. File is expected to exists.")?;
128		let mut chain_spec: serde_json::Value = serde_json::from_str(&json)?;
129
130		let initial_validators = config
131			.initial_permissioned_candidates_parsed
132			.iter()
133			.map(|c| {
134				serde_json::to_value((c.account_id_32(), c.session_keys::<T::AuthorityKeys>()))
135			})
136			.collect::<Result<Vec<serde_json::Value>, _>>()?;
137		let initial_validators = serde_json::Value::Array(initial_validators);
138		Self::update_field(&mut chain_spec, SESSION_INITIAL_VALIDATORS_PATH, initial_validators)?;
139
140		let initial_authorities = config
141			.initial_permissioned_candidates_parsed
142			.iter()
143			.map(|c| -> anyhow::Result<serde_json::Value> {
144				let initial_member =
145					T::initial_member(c.sidechain.into(), c.session_keys::<T::AuthorityKeys>());
146				Ok(serde_json::to_value(initial_member)?)
147			})
148			.collect::<Result<Vec<serde_json::Value>, _>>()?;
149		let initial_authorities = serde_json::Value::Array(initial_authorities);
150		Self::update_field(
151			&mut chain_spec,
152			SESSION_VALIDATOR_MANAGEMENT_INITIAL_AUTHORITIES_PATH,
153			initial_authorities,
154		)?;
155		context.write_file("chain-spec.json", serde_json::to_string_pretty(&chain_spec)?.as_str());
156		Ok(())
157	}
158
159	fn update_field(
160		chain_spec: &mut JValue,
161		field_name: &str,
162		value: JValue,
163	) -> Result<(), anyhow::Error> {
164		if let Some(field) = chain_spec.pointer_mut(field_name) {
165			*field = value;
166			Ok(())
167		} else {
168			Err(anyhow!(
169				"Internal error: Could not find {field_name} in chain spec file! Possibly this wizard does not support the current chain spec version."
170			))
171		}
172	}
173}

175#[derive(Debug)]
176struct CreateChainSpecConfig {
177	genesis_utxo: UtxoId,
178	initial_permissioned_candidates_raw: Vec<PermissionedCandidateKeys>,
179	initial_permissioned_candidates_parsed: Vec<ParsedPermissionedCandidatesKeys>,
180	committee_candidate_address: String,
181	d_parameter_policy_id: String,
182	permissioned_candidates_policy_id: String,
183	native_token_policy: String,
184	native_token_asset_name: String,
185	illiquid_supply_address: String,
186}
187
188impl CreateChainSpecConfig {
189	pub fn load<C: IOContext>(c: &C) -> Result<Self, anyhow::Error> {
190		let initial_permissioned_candidates_raw =
191			load_config_field(c, &config_fields::INITIAL_PERMISSIONED_CANDIDATES)?;
192		let initial_permissioned_candidates_parsed: Vec<ParsedPermissionedCandidatesKeys> =
193			initial_permissioned_candidates_raw
194				.iter()
195				.map(TryFrom::try_from)
196				.collect::<Result<Vec<ParsedPermissionedCandidatesKeys>, anyhow::Error>>()?;
197		Ok(Self {
198			genesis_utxo: load_config_field(c, &config_fields::GENESIS_UTXO)?,
199			initial_permissioned_candidates_raw,
200			initial_permissioned_candidates_parsed,
201			committee_candidate_address: load_config_field(
202				c,
203				&config_fields::COMMITTEE_CANDIDATES_ADDRESS,
204			)?,
205			d_parameter_policy_id: load_config_field(c, &config_fields::D_PARAMETER_POLICY_ID)?,
206			permissioned_candidates_policy_id: load_config_field(
207				c,
208				&config_fields::PERMISSIONED_CANDIDATES_POLICY_ID,
209			)?,
210			native_token_policy: load_config_field(c, &config_fields::NATIVE_TOKEN_POLICY)?,
211			native_token_asset_name: load_config_field(c, &config_fields::NATIVE_TOKEN_ASSET_NAME)?,
212			illiquid_supply_address: load_config_field(c, &config_fields::ILLIQUID_SUPPLY_ADDRESS)?,
213		})
214	}
215}
216
217fn load_config_field<C: IOContext, T: DeserializeOwned>(
218	context: &C,
219	field: &ConfigFieldDefinition<T>,
220) -> Result<T, anyhow::Error> {
221	field.load_from_file(context).ok_or_else(|| {
222		context.eprint(format!("The '{}' configuration file is missing or invalid.\nIf you are the governance authority, please make sure you have run the `prepare-configuration` command to generate the chain configuration file.\nIf you are a validator, you can obtain the chain configuration file from the governance authority.", field.config_file).as_str());
223		anyhow!("failed to read '{}'", field.path.join("."))
224	})
225}
