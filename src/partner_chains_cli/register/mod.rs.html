25#[derive(Clone, Debug)]
26pub struct SidechainPublicKeyParam(pub SidechainPublicKey);
27
28impl Display for SidechainPublicKeyParam {
29	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
30		write!(f, "0x{}", hex::encode(&self.0.0))
31	}
32}
33
34impl FromStr for SidechainPublicKeyParam {
35	type Err = secp256k1::Error;
36
37	fn from_str(s: &str) -> Result<Self, Self::Err> {
38		let trimmed = s.trim_start_matches("0x");
39		let pk = PublicKey::from_str(trimmed)?;
40		Ok(SidechainPublicKeyParam(SidechainPublicKey(pk.serialize().to_vec())))
41	}
42}

44#[derive(Clone, Debug)]
45pub struct PlainPublicKeyParam(pub String);
46
47impl Display for PlainPublicKeyParam {

49		write!(f, "{}", self.0)
50	}
51}
52
53impl FromStr for PlainPublicKeyParam {
54	type Err = Infallible;
55
56	fn from_str(s: &str) -> Result<Self, Self::Err> {
57		Ok(PlainPublicKeyParam(s.to_string()))
58	}
59}
