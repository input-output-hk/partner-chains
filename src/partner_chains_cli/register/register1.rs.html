21impl CmdRun for Register1Cmd {
22	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
23		context.print("⚙️ Registering as a committee candidate (step 1/3)");
24		let genesis_utxo = load_chain_config_field(context, &config_fields::GENESIS_UTXO)?;
25
26		let node_data_base_path = config_fields::SUBSTRATE_NODE_DATA_BASE_PATH
27			.load_from_file(context)
28			.ok_or(anyhow::anyhow!(
29				"⚠️ Keystore not found. Please run the `generate-keys` command first"
30			))?;
31
32		let GeneratedKeysFileContent { sidechain_pub_key, aura_pub_key, grandpa_pub_key } =
33			read_generated_keys(context).map_err(|e| {
34			    context.eprint("⚠️ The keys file `partner-chains-cli-keys.json` is missing or invalid. Please run the `generate-keys` command first");
35				anyhow!(e)
36			})?;
37
38		context.print("This wizard will query your UTXOs using address derived from the payment verification key and Ogmios service");
39		let ogmios_configuration = prompt_ogmios_configuration(context)?;
40		let shelley_genesis_config =
41			get_shelley_config(&format!("{ogmios_configuration}"), context)?;
42		let address = derive_address(context, shelley_genesis_config.network)?;
43		let utxo_query_result = query_utxos(context, &ogmios_configuration, &address)?;
44
45		if utxo_query_result.is_empty() {
46			context.eprint("⚠️ No UTXOs found for the given address");
47			context.eprint(
48				"The registering transaction requires at least one UTXO to be present at the address.",
49			);
50			return Err(anyhow::anyhow!("No UTXOs found"));
51		};
52
53		let registration_utxo: UtxoId =
54			select_from_utxos(context, "Select UTXO to use for registration", utxo_query_result)?;
55
56		context.print("Please do not spend this UTXO, it needs to be consumed by the registration transaction.");
57		context.print("");
58
59		let sidechain_pub_key_typed: SidechainPublicKey =
60			SidechainPublicKey(from_hex(&sidechain_pub_key).map_err(|e| {
61				context.eprint(&format!("⚠️ Failed to decode sidechain public key: {e}"));
62				anyhow!(e)
63			})?);
64
65		let registration_message = RegisterValidatorMessage {
66			genesis_utxo,
67			sidechain_pub_key: sidechain_pub_key_typed,
68			registration_utxo,
69		};
70
71		let ecdsa_pair = get_ecdsa_pair_from_file(
72			context,
73			&keystore_path(&node_data_base_path),
74			&sidechain_pub_key,
75		)
76		.map_err(|e| {
77			context.eprint(&format!("⚠️ Failed to read sidechain key from the keystore: {e}"));
78			anyhow!(e)
79		})?;
80
81		let sidechain_signature =
82			sign_registration_message_with_sidechain_key(registration_message, ecdsa_pair)?;
83		let executable = context.current_executable()?;
84		context.print("Run the following command to generate signatures on the next step. It has to be executed on the machine with your SPO cold signing key.");
85		context.print("");
86		context.print(&format!("{executable} wizards register2 \\\n --genesis-utxo {genesis_utxo} \\\n --registration-utxo {registration_utxo} \\\n --aura-pub-key {aura_pub_key} \\\n --grandpa-pub-key {grandpa_pub_key} \\\n --sidechain-pub-key {sidechain_pub_key} \\\n --sidechain-signature {sidechain_signature}"));
87
88		Ok(())
89	}

92fn get_ecdsa_pair_from_file<C: IOContext>(
93	context: &C,
94	keystore_path: &str,
95	sidechain_pub_key: &str,
96) -> Result<ecdsa::Pair, anyhow::Error> {
97	let mut seed_phrase_file_name = CROSS_CHAIN.key_type_hex();
98	seed_phrase_file_name.push_str(sidechain_pub_key.replace("0x", "").as_str());
99	let seed_phrase_file_path = format!("{keystore_path}/{seed_phrase_file_name}");
100	let seed = context
101		.read_file(&seed_phrase_file_path)
102		.ok_or_else(|| anyhow::anyhow!("seed phrase file not found"))?;
103	let stripped_quotes = seed.trim_matches('\"');
104	Ok(ecdsa::Pair::from_string(stripped_quotes, None)?)
105}
106
107fn sign_registration_message_with_sidechain_key(
108	message: RegisterValidatorMessage,
109	ecdsa_pair: ecdsa::Pair,
110) -> Result<String, anyhow::Error> {
111	let seed = ecdsa_pair.seed();
112	let secret_key = secp256k1::SecretKey::from_slice(&seed).map_err(|e| anyhow!(e))?;
113	let (_, sig) = sc_public_key_and_signature_for_datum(secret_key, message.clone());
114	Ok(hex::encode(sig.serialize_compact()))
115}

131pub fn load_chain_config_field<C: IOContext, T>(
132	context: &C,
133	field: &config::ConfigFieldDefinition<T>,
134) -> Result<T, anyhow::Error>
135where
136	T: DeserializeOwned,
137{
138	field.load_from_file(context).ok_or_else(|| {
139		context.eprint("⚠️ The chain configuration file `pc-chain-config.json` is missing or invalid.\n If you are the governance authority, please make sure you have run the `prepare-configuration` command to generate the chain configuration file.\n If you are a validator, you can obtain the chain configuration file from the governance authority.");
140		anyhow::anyhow!("failed to read {}", field.path.join("."))
141	})
142}

144fn derive_address<C: IOContext>(
145	context: &C,
146	cardano_network: NetworkType,
147) -> Result<String, anyhow::Error> {
148	let cardano_payment_verification_key_file =
149		config_fields::CARDANO_PAYMENT_VERIFICATION_KEY_FILE
150			.prompt_with_default_from_file_and_save(context);
151	let key_bytes: [u8; 32] = cardano_key::get_payment_verification_key_bytes_from_file(
152		&cardano_payment_verification_key_file,
153		context,
154	)?;
155	let address =
156		partner_chains_cardano_offchain::csl::payment_address(&key_bytes, cardano_network.to_csl());
157	address.to_bech32(None).map_err(|e| anyhow!(e.to_string()))
158}
