21impl CmdRun for Register1Cmd {
22	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
23		context.print("⚙️ Registering as a committee candidate (step 1/3)");
24		let genesis_utxo = load_chain_config_field(context, &config_fields::GENESIS_UTXO)?;
25
26		let node_data_base_path = config_fields::SUBSTRATE_NODE_DATA_BASE_PATH
27			.load_from_file(context)
28			.ok_or(anyhow::anyhow!(
29				"⚠️ Keystore not found. Please run the `generate-keys` command first"
30			))?;
31
32		let GeneratedKeysFileContent { sidechain_pub_key, aura_pub_key, grandpa_pub_key } =
33			read_generated_keys(context).map_err(|e| {
34			    context.eprint("⚠️ The keys file `partner-chains-cli-keys.json` is missing or invalid. Please run the `generate-keys` command first");
35				anyhow!(e)
36			})?;
37
38		context.print("This wizard will query your UTXOs using address derived from the payment verification key and Ogmios service");
39		let ogmios_configuration = prompt_ogmios_configuration(context)?;
40		let shelley_genesis_config =
41			get_shelley_config(&format!("{ogmios_configuration}"), context)?;
42		let address = derive_address(context, shelley_genesis_config.network)?;
43		let utxo_query_result = query_utxos(context, &ogmios_configuration, &address)?;
44
45		if utxo_query_result.is_empty() {
46			context.eprint("⚠️ No UTXOs found for the given address");
47			context.eprint(
48				"The registering transaction requires at least one UTXO to be present at the address.",
49			);
50			return Err(anyhow::anyhow!("No UTXOs found"));
51		};
52
53		let registration_utxo: UtxoId =
54			select_from_utxos(context, "Select UTXO to use for registration", utxo_query_result)?;
55
56		context.print(
57			"Please do not spend this UTXO, it needs to be consumed by the registration transaction.",
58		);
59		context.print("");
60
61		let sidechain_pub_key_typed: SidechainPublicKey =
62			SidechainPublicKey(from_hex(&sidechain_pub_key).map_err(|e| {
63				context.eprint(&format!("⚠️ Failed to decode sidechain public key: {e}"));
64				anyhow!(e)
65			})?);
66
67		let registration_message = RegisterValidatorMessage {
68			genesis_utxo,
69			sidechain_pub_key: sidechain_pub_key_typed,
70			registration_utxo,
71		};
72
73		let ecdsa_pair = get_ecdsa_pair_from_file(
74			context,
75			&keystore_path(&node_data_base_path),
76			&sidechain_pub_key,
77		)
78		.map_err(|e| {
79			context.eprint(&format!("⚠️ Failed to read sidechain key from the keystore: {e}"));
80			anyhow!(e)
81		})?;
82
83		let sidechain_signature =
84			sign_registration_message_with_sidechain_key(registration_message, ecdsa_pair)?;
85		let executable = context.current_executable()?;
86		context.print("Run the following command to generate signatures on the next step. It has to be executed on the machine with your SPO cold signing key.");
87		context.print("");
88		context.print(&format!("{executable} wizards register2 \\\n --genesis-utxo {genesis_utxo} \\\n --registration-utxo {registration_utxo} \\\n --aura-pub-key {aura_pub_key} \\\n --grandpa-pub-key {grandpa_pub_key} \\\n --sidechain-pub-key {sidechain_pub_key} \\\n --sidechain-signature {sidechain_signature}"));
89
90		Ok(())
91	}
92}

94fn get_ecdsa_pair_from_file<C: IOContext>(
95	context: &C,
96	keystore_path: &str,
97	sidechain_pub_key: &str,
98) -> Result<ecdsa::Pair, anyhow::Error> {
99	let mut seed_phrase_file_name = CROSS_CHAIN.key_type_hex();
100	seed_phrase_file_name.push_str(sidechain_pub_key.replace("0x", "").as_str());
101	let seed_phrase_file_path = format!("{keystore_path}/{seed_phrase_file_name}");
102	let seed = context
103		.read_file(&seed_phrase_file_path)
104		.ok_or_else(|| anyhow::anyhow!("seed phrase file not found"))?;
105	let stripped_quotes = seed.trim_matches('\"');
106	Ok(ecdsa::Pair::from_string(stripped_quotes, None)?)
107}
108
109fn sign_registration_message_with_sidechain_key(
110	message: RegisterValidatorMessage,
111	ecdsa_pair: ecdsa::Pair,
112) -> Result<String, anyhow::Error> {
113	let seed = ecdsa_pair.seed();
114	let secret_key = secp256k1::SecretKey::from_slice(&seed).map_err(|e| anyhow!(e))?;
115	let (_, sig) = sc_public_key_and_signature_for_datum(secret_key, message.clone());
116	Ok(hex::encode(sig.serialize_compact()))
117}

126pub fn read_generated_keys<C: IOContext>(context: &C) -> anyhow::Result<GeneratedKeysFileContent> {

133pub fn load_chain_config_field<C: IOContext, T>(
134	context: &C,
135	field: &config::ConfigFieldDefinition<T>,
136) -> Result<T, anyhow::Error>
137where
138	T: DeserializeOwned,
139{
140	field.load_from_file(context).ok_or_else(|| {
141		context.eprint("⚠️ The chain configuration file `pc-chain-config.json` is missing or invalid.\n If you are the governance authority, please make sure you have run the `prepare-configuration` command to generate the chain configuration file.\n If you are a validator, you can obtain the chain configuration file from the governance authority.");
142		anyhow::anyhow!("failed to read {}", field.path.join("."))
143	})
144}

146fn derive_address<C: IOContext>(
147	context: &C,
148	cardano_network: NetworkType,
149) -> Result<String, anyhow::Error> {
150	let cardano_payment_verification_key_file =
151		config_fields::CARDANO_PAYMENT_VERIFICATION_KEY_FILE
152			.prompt_with_default_from_file_and_save(context);
153	let key_bytes: [u8; 32] = cardano_key::get_payment_verification_key_bytes_from_file(
154		&cardano_payment_verification_key_file,
155		context,
156	)?;
157	let address =
158		partner_chains_cardano_offchain::csl::payment_address(&key_bytes, cardano_network.to_csl());
159	address.to_bech32(None).map_err(|e| anyhow!(e.to_string()))
160}
