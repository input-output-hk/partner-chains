12pub fn prepare_main_chain_config<C: IOContext>(
13	context: &C,
14	ogmios_config: &ServiceConfig,
15	genesis_utxo: UtxoId,
16) -> anyhow::Result<()> {
17	let cardano_parameteres = prepare_cardano_params(ogmios_config, context)?;
18	cardano_parameteres.save(context);
19	set_up_cardano_addresses(context, genesis_utxo, ogmios_config)?;
20
21	if INITIAL_PERMISSIONED_CANDIDATES.load_from_file(context).is_none() {
22		INITIAL_PERMISSIONED_CANDIDATES.save_to_file(&vec![], context)
23	}
24	prepare_native_token(context)?;
25	context.eprint(OUTRO);
26	Ok(())
27}

29fn set_up_cardano_addresses<C: IOContext>(
30	context: &C,
31	genesis_utxo: UtxoId,
32	ogmios_config: &ServiceConfig,
33) -> anyhow::Result<()> {
34	let offchain_impl = context.offchain_impl(ogmios_config)?;
35	let runtime = tokio::runtime::Runtime::new().map_err(|e| anyhow::anyhow!(e))?;
36	let scripts_data = runtime
37		.block_on(offchain_impl.get_scripts_data(genesis_utxo))
38		.map_err(|e| anyhow::anyhow!("Offchain call failed: {e:?}!"))?;
39
40	let committee_candidate_validator_addr = scripts_data.addresses.committee_candidate_validator;
41	let d_parameter_policy_id = hex::encode(scripts_data.policy_ids.d_parameter.0);
42	let permissioned_candidates_policy_id =
43		hex::encode(scripts_data.policy_ids.permissioned_candidates.0);
44	let illiquid_supply_addr = scripts_data.addresses.illiquid_circulation_supply_validator;
45	COMMITTEE_CANDIDATES_ADDRESS.save_to_file(&committee_candidate_validator_addr, context);
46	D_PARAMETER_POLICY_ID.save_to_file(&d_parameter_policy_id, context);
47	PERMISSIONED_CANDIDATES_POLICY_ID.save_to_file(&permissioned_candidates_policy_id, context);
48	ILLIQUID_SUPPLY_ADDRESS.save_to_file(&illiquid_supply_addr, context);
49	context.print(&format!(
50		"Cardano addresses have been set up:
51- Committee Candidates Address: {committee_candidate_validator_addr}
52- D Parameter Policy ID: {d_parameter_policy_id}
53- Permissioned Candidates Policy ID: {permissioned_candidates_policy_id}
54- Illiquid Supply Address: {illiquid_supply_addr}"
55	));
56	Ok(())
57}

59fn prepare_native_token<C: IOContext>(context: &C) -> anyhow::Result<()> {
60	context.print(
61		"Partner Chains can store their initial token supply on Cardano as Cardano native tokens.",
62	);
63	context.print("Creation of the native token is not supported by this wizard and must be performed manually before this step.");
64	if context.prompt_yes_no("Do you want to configure a native token for you Partner Chain?", true)
65	{
66		NATIVE_TOKEN_POLICY.prompt_with_default_from_file_and_save(context);
67		NATIVE_TOKEN_ASSET_NAME.prompt_with_default_from_file_and_save(context);
68	} else {
69		NATIVE_TOKEN_POLICY.save_to_file(&PolicyId::default().to_hex_string(), context);
70		NATIVE_TOKEN_ASSET_NAME.save_to_file(&"0x".into(), context);
71	}
72
73	Ok(())
74}
