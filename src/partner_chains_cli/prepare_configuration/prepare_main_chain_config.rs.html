14pub fn prepare_main_chain_config<C: IOContext>(
15	context: &C,
16	ogmios_config: &ServiceConfig,
17	genesis_utxo: UtxoId,
18) -> anyhow::Result<()> {
19	let cardano_params = prepare_cardano_params(ogmios_config, context)?;
20	cardano_params.save(context);
21	set_up_cardano_addresses(context, genesis_utxo, ogmios_config)?;
22
23	if INITIAL_PERMISSIONED_CANDIDATES.load_from_file(context).is_none() {
24		INITIAL_PERMISSIONED_CANDIDATES.save_to_file(&vec![], context)
25	}
26	prepare_native_token(context)?;
27	context.eprint(OUTRO);
28	Ok(())
29}

31fn set_up_cardano_addresses<C: IOContext>(
32	context: &C,
33	genesis_utxo: UtxoId,
34	ogmios_config: &ServiceConfig,
35) -> anyhow::Result<()> {
36	let offchain_impl = context.offchain_impl(ogmios_config)?;
37	let runtime = tokio::runtime::Runtime::new().map_err(|e| anyhow::anyhow!(e))?;
38	let scripts_data = runtime
39		.block_on(offchain_impl.get_scripts_data(genesis_utxo))
40		.map_err(|e| anyhow::anyhow!("Offchain call failed: {e:?}!"))?;
41
42	let committee_candidate_validator_addr =
43		MainchainAddress::from_str(&scripts_data.addresses.committee_candidate_validator)
44			.map_err(|e| anyhow!("Internal error: invalid address '{e}' in scripts data"))?;
45	let d_parameter_policy_id = scripts_data.policy_ids.d_parameter;
46	let permissioned_candidates_policy_id = scripts_data.policy_ids.permissioned_candidates;
47	let illiquid_supply_addr =
48		MainchainAddress::from_str(&scripts_data.addresses.illiquid_circulation_supply_validator)
49			.map_err(|e| anyhow!("Internal error: invalid address '{e}' in scripts data"))?;
50	let governed_map_validator_addr =
51		MainchainAddress::from_str(&scripts_data.addresses.governed_map_validator)
52			.map_err(|e| anyhow!("Internal error: invalid address '{e}' in scripts data"))?;
53	let governed_map_policy_id = scripts_data.policy_ids.governed_map;
54	COMMITTEE_CANDIDATES_ADDRESS.save_to_file(&committee_candidate_validator_addr, context);
55	D_PARAMETER_POLICY_ID.save_to_file(&d_parameter_policy_id, context);
56	PERMISSIONED_CANDIDATES_POLICY_ID.save_to_file(&permissioned_candidates_policy_id, context);
57	ILLIQUID_SUPPLY_ADDRESS.save_to_file(&illiquid_supply_addr, context);
58	GOVERNED_MAP_VALIDATOR_ADDRESS.save_to_file(&governed_map_validator_addr, context);
59	GOVERNED_MAP_POLICY_ID.save_to_file(&governed_map_policy_id, context);
60	context.print(&format!(
61		"Cardano addresses have been set up:
62- Committee Candidates Address: {committee_candidate_validator_addr}
63- D Parameter Policy ID: {d_parameter_policy_id}
64- Permissioned Candidates Policy ID: {permissioned_candidates_policy_id}
65- Illiquid Supply Address: {illiquid_supply_addr}
66- Governed Map Validator Address: {governed_map_validator_addr}
67- Governed Map Policy Id: {governed_map_policy_id}"
68	));
69	Ok(())
70}

72fn prepare_native_token<C: IOContext>(context: &C) -> anyhow::Result<()> {
73	context.print(
74		"Partner Chains can store their initial token supply on Cardano as Cardano native tokens.",
75	);
76	context.print("Creation of the native token is not supported by this wizard and must be performed manually before this step.");
77	if context.prompt_yes_no("Do you want to configure a native token for you Partner Chain?", true)
78	{
79		NATIVE_TOKEN_POLICY
80			.prompt_with_default_from_file_parse_and_save(context)
81			.map_err(|e| {
82				anyhow!(
83					"Could not parse PolicyId, expected hex string representing 28 bytes. Error: '{e}'."
84				)
85			})?;
86		NATIVE_TOKEN_ASSET_NAME
87			.prompt_with_default_from_file_parse_and_save(context)
88			.map_err(|e| {
89				anyhow!("Could not parse AssetName, expected valid hex string. Error: '{e}'")
90			})?;
91	} else {
92		NATIVE_TOKEN_POLICY.save_to_file(&PolicyId::default(), context);
93		NATIVE_TOKEN_ASSET_NAME.save_to_file(&AssetName::empty(), context);
94	}
95
96	Ok(())
97}
