36impl CmdRun for PrepareConfigurationCmd {
37	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
38		establish_bootnodes(context)?;
39		let (genesis_utxo, private_key, ogmios_config) = select_genesis_utxo(context)?;
40		if let Some(_tx_id) = init_governance::run_init_governance(
41			self.common_arguments.retries(),
42			genesis_utxo,
43			&private_key,
44			&ogmios_config,
45			context,
46		)? {
47			prepare_main_chain_config(context, &ogmios_config, genesis_utxo)?;
48			context.eprint("üöÄ All done!");
49			Ok(())
50		} else {
51			context
52				.eprint("Chain governance has not been initialized. Please run the wizard again.");
53			Ok(())
54		}
55	}
56}

58fn establish_bootnodes(context: &impl IOContext) -> anyhow::Result<()> {
59	context.eprint(INTRO);
60
61	let peer_id = peer_id_from_config(context)?;
62
63	let (default_protocol, default_hostname_or_ip, default_port) = read_bootnode_defaults(context);
64
65	let protocol = choose_protocol(context, default_protocol);
66
67	let port: u16 = context
68		.prompt(CHOOSE_PORT_PROMPT, Some(&default_port.to_string()))
69		.parse()
70		.context("‚ö†Ô∏è Port must be a valid number")?;
71
72	let default_address = if default_protocol == protocol {
73		&default_hostname_or_ip
74	} else {
75		protocol.default_address()
76	};
77
78	let bootnode = match protocol {
79		Protocol::Dns => {
80			let hostname = context.prompt(CHOOSE_HOSTNAME_PROMPT, Some(default_address));
81			dns_bootnode(&hostname, port, &peer_id)
82		},
83		Protocol::Ipv4 => {
84			let ipv4_address = context.prompt(CHOOSE_IP_ADDRESS_PROMPT, Some(default_address));
85			let _: Ipv4Addr = ipv4_address.parse().context("‚ö†Ô∏è Invalid IP address")?;
86
87			ipv4_bootnode(&ipv4_address, port, &peer_id)
88		},
89	};
90
91	BOOTNODES.save_to_file(&vec![bootnode], context);
92	context.eprint(&outro());
93	Ok(())
94}

96fn read_bootnode_defaults(context: &impl IOContext) -> (Protocol, String, u16) {
97	let (default_protocol, default_hostname_or_ip, default_port) = deconstruct_bootnode(
98		BOOTNODES
99			.load_from_file(context)
100			.and_then(|bootnodes| bootnodes.into_iter().next()),
101	)
102	.unwrap_or((Protocol::Dns, Protocol::Dns.default_address().to_string(), DEFAULT_PORT));
103	(default_protocol, default_hostname_or_ip, default_port)
104}
105
106fn choose_protocol(context: &impl IOContext, default_protocol: Protocol) -> Protocol {
107	let mut protocols: Vec<String> = vec![Protocol::Dns.into(), Protocol::Ipv4.into()];
108	// default protocol should be the first in the list
109	protocols
110		.sort_by_key(|protocol| if *protocol != String::from(default_protocol) { 1 } else { 0 });
111
112	Protocol::from_str(&context.prompt_multi_option(CHOOSE_PROTOCOL_PROMPT, protocols))
113		.expect("Invalid protocol cannot be chosen from valid options only")
114}
115
116fn deconstruct_bootnode(bootnode_opt: Option<String>) -> Option<(Protocol, String, u16)> {
117	let bootnode = bootnode_opt?;
118	let mut parts = bootnode.split('/').skip(1);
119	let protocol = parts.next()?.parse().ok()?;
120	let hostname = parts.next()?.to_string();
121	parts.next()?;
122	let port = parts.next()?.parse().ok()?;
123	Some((protocol, hostname, port))
124}

126fn peer_id_from_config(context: &impl IOContext) -> anyhow::Result<String> {
127	let substrate_node_base_path =
128		SUBSTRATE_NODE_DATA_BASE_PATH.prompt_with_default_from_file_and_save(context);
129
130	let network_key_path = network_key_path(&substrate_node_base_path);
131
132	peer_id_from_network_key(
133		context
134			.read_file(&network_key_path)
135			.as_deref()
136			.ok_or(NetworkKeyNotFoundError(network_key_path))?,
137	)
138}

140#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
141enum Protocol {
142	Dns,
143	Ipv4,
144}
145
146impl Protocol {
147	fn default_address(&self) -> &str {
148		match self {
149			Protocol::Dns => "localhost",
150			Protocol::Ipv4 => "127.0.0.1",
151		}
152	}
153}
154
155impl From<Protocol> for String {
156	fn from(value: Protocol) -> Self {
157		match value {
158			Protocol::Dns => "hostname".to_string(),
159			Protocol::Ipv4 => "IP address".to_string(),
160		}
161	}
162}
163
164impl FromStr for Protocol {
165	type Err = String;
166
167	fn from_str(s: &str) -> Result<Self, Self::Err> {
168		match s {
169			"hostname" | "dns" => Ok(Protocol::Dns),
170			"IP address" | "ip4" => Ok(Protocol::Ipv4),
171			_ => Err("Invalid protocol".to_string()),
172		}
173	}
174}

207fn peer_id_from_network_key(key_str: &str) -> anyhow::Result<String> {
208	let hex_vec = hex::decode(key_str).context("Invalid hex")?;
209	let keypair = Keypair::ed25519_from_bytes(hex_vec).context("Invalid Ed25519 bytes")?;
210	Ok(keypair.public().to_peer_id().to_base58())
211}

213impl<T> InitGovernance for T
214where
215	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,
216{
217	async fn init_governance(
218		&self,
219		await_tx: FixedDelayRetries,
220		governance_parameters: &MultiSigParameters,
221		payment_key: &CardanoPaymentSigningKey,
222		genesis_utxo_id: UtxoId,
223	) -> Result<McTxHash, String> {
224		partner_chains_cardano_offchain::init_governance::run_init_governance(
225			governance_parameters,
226			payment_key,
227			Some(genesis_utxo_id),
228			self,
229			await_tx,
230		)
231		.await
232		.map(|result| result.tx_hash)
233		.map_err(|e| e.to_string())
234	}
235}

237impl<T: QueryNetwork> GetScriptsData for T {
238	async fn get_scripts_data(&self, genesis_utxo: UtxoId) -> Result<ScriptsData, String> {
239		let network = self
240			.shelley_genesis_configuration()

242			.map_err(|e| format!("Ogmios error: {e}"))?
243			.network
244			.to_csl();
245		get_scripts_data(genesis_utxo, network).map_err(|e| (e.to_string()))
246	}
