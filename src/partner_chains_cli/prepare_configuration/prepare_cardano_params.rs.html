7pub fn prepare_cardano_params<C: IOContext>(
8	ogmios_config: &ServiceConfig,
9	context: &C,
10) -> anyhow::Result<CardanoParameters> {
11	let addr = format!("{}", ogmios_config);
12	let eras_summaries = get_eras_summaries(&addr, context)?;
13	let shelley_config = get_shelley_config(&addr, context)?;
14	caradano_parameters(eras_summaries, shelley_config)
15}
16
17fn get_eras_summaries<C: IOContext>(addr: &str, context: &C) -> anyhow::Result<Vec<EraSummary>> {
18	let eras_summaries = context.ogmios_rpc(addr, OgmiosRequest::QueryLedgerStateEraSummaries)?;
19	match eras_summaries {
20		OgmiosResponse::QueryLedgerStateEraSummaries(eras_summaries) => Ok(eras_summaries),
21		other => Err(anyhow::anyhow!(format!(
22			"Unexpected response from Ogmios when quering for era summaries: {other:?}"
23		))),
24	}
25}
26
27fn caradano_parameters(
28	eras_summaries: Vec<EraSummary>,
29	shelley_config: ShelleyGenesisConfiguration,
30) -> anyhow::Result<CardanoParameters> {
31	let first_epoch_era = get_first_epoch_era(eras_summaries)?;
32	Ok(CardanoParameters {
33		security_parameter: shelley_config.security_parameter.into(),
34		active_slots_coeff: shelley_config.active_slots_coefficient,
35		epoch_duration_millis: (shelley_config.epoch_length as u64)
36			.checked_mul(shelley_config.slot_length_millis)
37			.ok_or_else(|| anyhow::anyhow!("Epoch duration overflow"))?,
38		first_epoch_number: first_epoch_era.start.epoch,
39		first_slot_number: first_epoch_era.start.slot,
40		first_epoch_timestamp_millis: shelley_config
41			.start_time
42			.checked_add(first_epoch_era.start.time_seconds)
43			.and_then(|seconds| seconds.checked_mul(1000))
44			.ok_or_else(|| anyhow::anyhow!("First epoch timestamp overflow"))?,
45		slot_duration_millis: shelley_config.slot_length_millis,
46	})
47}
48
49// This functions gets the first era
50// such that all following eras have the same slot length and epoch length.
51fn get_first_epoch_era(eras_summaries: Vec<EraSummary>) -> Result<EraSummary, anyhow::Error> {
52	let latest_era = eras_summaries.last().ok_or_else(|| anyhow::anyhow!("No eras found"))?;
53	let first_epoch_era = eras_summaries
54		.iter()
55		.find(|era| era.parameters == latest_era.parameters)
56		.ok_or_else(|| anyhow::anyhow!("No eras found"))?;
57	Ok(first_epoch_era.clone())
58}
