16pub(crate) fn run_init_governance<C: IOContext>(
17	await_tx: FixedDelayRetries,
18	genesis_utxo: UtxoId,
19	payment_key: &CardanoPaymentSigningKey,
20	ogmios_config: &ServiceConfig,
21	context: &C,
22) -> anyhow::Result<Option<McTxHash>> {
23	let multisig_parameters = prompt_initial_governance(payment_key, context)?;
24	let should_continue = context.prompt_yes_no(
25		&format!(
26			"Governance will be initialized with:\n{}\nDo you want to continue?",
27			multisig_parameters
28		),
29		false,
30	);
31	if should_continue {
32		let offchain = context.offchain_impl(ogmios_config)?;
33		let runtime = tokio::runtime::Runtime::new().map_err(|e| anyhow::anyhow!(e))?;
34		let tx_id = runtime
35			.block_on(offchain.init_governance(
36				await_tx,
37				&multisig_parameters,
38				&payment_key,
39				genesis_utxo,
40			))
41			.map_err(|e| anyhow::anyhow!("Governance initialization failed: {e:?}!"))?;
42		context.eprint(&format!("Governance initialized successfully for UTXO: {}", genesis_utxo));
43		Ok(Some(tx_id))
44	} else {
45		Ok(None)
46	}
47}

49fn prompt_initial_governance<C: IOContext>(
50	payment_key: &CardanoPaymentSigningKey,
51	context: &C,
52) -> Result<MultiSigParameters, anyhow::Error> {
53	context.eprint("Please provide the initial chain governance key hashes:");
54	INITIAL_GOVERNANCE_AUTHORITIES.save_if_empty(
55		GovernanceAuthoritiesKeyHashes(vec![payment_key.to_pub_key_hash()]),
56		context,
57	);
58	let authorities = INITIAL_GOVERNANCE_AUTHORITIES
59		.prompt_with_default_from_file_parse_and_save(context)
60		.map_err(|e| anyhow!("Failed to parse governance authorities: {}", e))?;
61
62	INITIAL_GOVERNANCE_THRESHOLD.save_if_empty(1, context);
63	let threshold = INITIAL_GOVERNANCE_THRESHOLD
64		.prompt_with_default_from_file_parse_and_save(context)
65		.map_err(|e| anyhow!("Failed do parse threshold: {}", e))?;
66
67	MultiSigParameters::new(authorities.0.as_ref(), threshold).map_err(
68		|err| anyhow!(
69			"Initial Governance data is invalid: '{}'. Please run the wizard again and provide correct value or edit values in '{}' and the run the wizard again. Example: '{}'",
70			err,
71			INITIAL_GOVERNANCE_AUTHORITIES.config_file,
72			&example_governance_auth()
73		)
74	)
75}
