13pub fn select_genesis_utxo<C: IOContext>(
14	context: &C,
15) -> anyhow::Result<(UtxoId, CardanoPaymentSigningKey, ServiceConfig)> {
16	context.eprint(INTRO);
17	let ogmios_configuration = prompt_ogmios_configuration(context)?;
18	let shelley_config = get_shelley_config(&ogmios_configuration.to_string(), context)?;
19	let private_key = get_private_key(context)?;
20	let address = derive_address(&private_key, shelley_config.network)?;
21	let utxo_query_result = query_utxos(context, &ogmios_configuration, &address)?;
22
23	if utxo_query_result.is_empty() {
24		context.eprint("⚠️ No UTXOs found for the given address");
25		context.eprint("There has to be at least one UTXO in the governance authority wallet.");
26		return Err(anyhow::anyhow!("No UTXOs found"));
27	};
28	let genesis_utxo =
29		select_from_utxos(context, "Select an UTXO to use as the genesis UTXO", utxo_query_result)?;
30
31	GENESIS_UTXO.save_to_file(&genesis_utxo, context);
32	Ok((genesis_utxo, private_key, ogmios_configuration))
33}

35fn get_private_key<C: IOContext>(context: &C) -> Result<CardanoPaymentSigningKey, anyhow::Error> {
36	let cardano_signing_key_file = config_fields::CARDANO_PAYMENT_SIGNING_KEY_FILE
37		.prompt_with_default_from_file_and_save(context);
38	let pkey =
39		cardano_key::get_mc_payment_signing_key_from_file(&cardano_signing_key_file, context)?;
40	Ok(pkey)
41}
42
43fn derive_address(
44	private_key: &CardanoPaymentSigningKey,
45	cardano_network: NetworkType,
46) -> Result<String, anyhow::Error> {
47	let address = partner_chains_cardano_offchain::csl::key_hash_address(
48		&private_key.to_csl_pub_key().hash(),
49		cardano_network.to_csl(),
50	);
51	address.to_bech32(None).map_err(|e| anyhow!(e.to_string()))
52}
