26	common_arguments: crate::CommonArguments,

29impl CmdRun for DeregisterCmd {
30	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
31		let chain_config = read_chain_config_file(context)?;
32		context.print(
33		  &format!("This wizard will remove the specified candidate from the committee candidates based on the following chain parameters:\n{}.\nCommittee Candidate Validator Address is '{}'\n",
34          &chain_config.chain_parameters,
35          &chain_config.cardano_addresses.committee_candidates_address)
36    	);
37		context.print("Payment signing key and cold verification key used for registration are required to deregister.");
38		let payment_signing_key_path =
39			CARDANO_PAYMENT_SIGNING_KEY_FILE.prompt_with_default_from_file_and_save(context);
40		let payment_signing_key =
41			get_mc_payment_signing_key_from_file(&payment_signing_key_path, context)?;
42		let cold_vkey_path =
43			CARDANO_COLD_VERIFICATION_KEY_FILE.prompt_with_default_from_file_and_save(context);
44		let stake_ownership_pub_key =
45			get_stake_pool_verification_key_from_file(&cold_vkey_path, context)?;
46		let ogmios_config = establish_ogmios_configuration(context)?;
47		let offchain = context.offchain_impl(&ogmios_config)?;
48
49		let runtime = tokio::runtime::Runtime::new().map_err(|e| anyhow::anyhow!(e))?;
50		runtime
51			.block_on(offchain.deregister(
52				self.common_arguments.retries(),
53				chain_config.chain_parameters.genesis_utxo,
54				&payment_signing_key,
55				stake_ownership_pub_key,
56			))
57			.map_err(|e| anyhow::anyhow!("Candidate deregistration failed: {e:?}!"))?;
58
59		Ok(())
60	}
61}
62
63fn read_chain_config_file<C: IOContext>(
64	context: &C,
65) -> Result<crate::config::ChainConfig, anyhow::Error> {
66	let chain_config = crate::config::load_chain_config(context);
67	chain_config
68		.map_err(|_|anyhow!(
69			"Couldn't parse chain configuration file {}. The chain configuration file that was used for registration is required in the working directory.",
70			context.chain_config_file_path()))
71}

73impl<T> Deregister for T
74where
75	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,
76{
77	async fn deregister(
78		&self,
79		await_tx: FixedDelayRetries,
80		genesis_utxo: UtxoId,
81		payment_signing_key: &CardanoPaymentSigningKey,
82		stake_ownership_pub_key: StakePoolPublicKey,
83	) -> Result<Option<McTxHash>, String> {
84		run_deregister(genesis_utxo, payment_signing_key, stake_ownership_pub_key, self, await_tx)
85			.await
86			.map_err(|e| e.to_string())
87	}
