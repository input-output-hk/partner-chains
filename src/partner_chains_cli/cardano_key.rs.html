8fn parse_json_key_file(
9	path: &str,
10	context: &impl IOContext,
11) -> anyhow::Result<CardanoKeyFileContent> {
12	let Some(file_content) = context.read_file(path) else {
13		return Err(anyhow!("Failed to read Cardano key file {path}"));
14	};
15	serde_json::from_str::<CardanoKeyFileContent>(&file_content)
16		.map_err(|err| anyhow!("Failed to parse Cardano key file {path}: '{err}'"))
17}

19pub(crate) fn get_mc_payment_signing_key_from_file(

28pub(crate) fn get_payment_verification_key_bytes_from_file(
29	path: &str,
30	context: &impl IOContext,
31) -> anyhow::Result<[u8; 32]> {
32	let key_file = parse_json_key_file(path, context)?;
33	let key_type = key_file.r#type.clone();
34	if key_type == "PaymentExtendedVerificationKeyShelley_ed25519_bip32" {
35		let key_bytes = key_file
36			.raw_key_bytes::<64>()
37			.map_err(|e| anyhow!("Failed to parse key bytes in {path}. {e}"))?;
38		let prefix: [u8; 32] =
39			key_bytes[0..32].try_into().expect("[0..32] of [u8; 64] has to fit [u8; 32]");
40		Ok(prefix)
41	} else if key_type == "PaymentVerificationKeyShelley_ed25519" {
42		let key_bytes = key_file
43			.raw_key_bytes()
44			.map_err(|e| anyhow!("Failed to parse key bytes in {path}. {e}"))?;
45		Ok(key_bytes)
46	} else {
47		Err(anyhow!(
48			"Unexpected key type '{key_type}' in {path}. Expected a Payment Verification Key."
49		))
50	}
51}
