27impl StartNodeConfig {

45impl<T: PartnerChainRuntime> CmdRun for StartNodeCmd<T> {
46	fn run<C: IOContext>(&self, context: &C) -> anyhow::Result<()> {
47		let config = StartNodeConfig::load(context);
48
49		if !check_keystore::<C, T>(&config, context)? || !check_chain_spec(context) {
50			return Ok(());
51		}
52
53		let data_source = CARDANO_DATA_SOURCE.default.unwrap();
54		CARDANO_DATA_SOURCE.save_to_file(&data_source.to_string(), context);
55
56		let db_connection_string = POSTGRES_CONNECTION_STRING.load_or_prompt_and_save(context);
57
58		let Some(chain_config) = load_chain_config(context)? else { return Ok(()) };
59
60		if !self.silent
61			&& !prompt_values_fine(
62				&config,
63				&chain_config,
64				&db_connection_string,
65				&data_source,
66				context,
67			) {
68			context.eprint("Aborting. Edit configuration files and rerun the command.");
69			return Ok(());
70		}
71
72		start_node::<C, T>(config, chain_config, &db_connection_string, &data_source, context)?;
73
74		Ok(())
75	}
76}

78fn load_chain_config<C: IOContext>(context: &C) -> anyhow::Result<Option<StartNodeChainConfig>> {
79	let chain_config_file_path = context.chain_config_file_path();
80	let Some(chain_config_file) = context.read_file(&chain_config_file_path) else {
81		context.eprint(&format!("⚠️ Chain config file {chain_config_file_path} does not exists. Run prepare-configuration wizard first."));
82		return Ok(None);
83	};
84	let chain_config = match serde_json::from_str::<StartNodeChainConfig>(&chain_config_file) {
85		Ok(chain_config) => chain_config,
86		Err(err) => {
87			context.eprint(&format!("⚠️ Chain config file {chain_config_file_path} is invalid: {err}. Run prepare-configuration wizard or fix errors manually."));
88			return Ok(None);
89		},
90	};
91	Ok(Some(chain_config))
92}

94#[rustfmt::skip]
95fn prompt_values_fine<C: IOContext>(
96	StartNodeConfig { substrate_node_base_path }: &StartNodeConfig,
97	StartNodeChainConfig {
98		cardano,
99		bootnodes,
100	}: &StartNodeChainConfig,
101	db_connection_string: &str,
102	data_source: &str,
103	context: &C,
104) -> bool
105{
106	context.eprint("The following values will be used to run the node:");
107	context.eprint(&format!("    base path  = {}", substrate_node_base_path));
108	context.eprint(&format!("    chain spec = {}", CHAIN_SPEC_PATH));
109	context.eprint(&format!("    bootnodes  = [{}]", bootnodes.join(", ")));
110	context.eprint("    environment:");
111	context.eprint(&format!("        BLOCK_STABILITY_MARGIN             = {}", 0));
112	context.eprint(&format!("        CARDANO_SECURITY_PARAMETER         = {}", cardano.security_parameter));
113	context.eprint(&format!("        CARDANO_ACTIVE_SLOTS_COEFF         = {}", cardano.active_slots_coeff));
114	context.eprint(&format!("        FIRST_EPOCH_TIMESTAMP_MILLIS       = {}", cardano.first_epoch_timestamp_millis));
115	context.eprint(&format!("        EPOCH_DURATION_MILLIS              = {}", cardano.epoch_duration_millis));
116	context.eprint(&format!("        FIRST_EPOCH_NUMBER                 = {}", cardano.first_epoch_number));
117	context.eprint(&format!("        FIRST_SLOT_NUMBER                  = {}", cardano.first_slot_number));
118	context.eprint(&format!("        CARDANO_DATA_SOURCE                = {}", data_source));
119	context.eprint(&format!("        DB_SYNC_POSTGRES_CONNECTION_STRING = {}", db_connection_string));
120	context.prompt_yes_no("Proceed?", true)
121}

123fn check_chain_spec<C: IOContext>(context: &C) -> bool {
124	if context.file_exists(CHAIN_SPEC_PATH) {
125		true
126	} else {
127		context.eprint(&format!("Chain spec file {} missing.", CHAIN_SPEC_PATH));
128		context.eprint(
129			"Please run the create-chain-spec wizard first or you can get it from your chain governance.",
130		);
131		false
132	}
133}

135fn check_keystore<C: IOContext, T: PartnerChainRuntime>(
136	config: &StartNodeConfig,
137	context: &C,
138) -> anyhow::Result<bool> {
139	let existing_keys = context.list_directory(&config.keystore_path())?.unwrap_or_default();
140	Ok(key_present(&CROSS_CHAIN, &existing_keys, context)
141		&& T::key_definitions()
142			.iter()
143			.all(|key_def| key_present(key_def, &existing_keys, context)))
144}
145
146fn key_present<C: IOContext>(key: &KeyDefinition, existing_keys: &[String], context: &C) -> bool {
147	if find_existing_key(existing_keys, key).is_none() {
148		context.eprint(&format!(
149			"⚠️ {} key is missing from the keystore. Please run generate-keys wizard first.",
150			key.name
151		));
152		false
153	} else {
154		true
155	}
156}

158pub fn start_node<C: IOContext, T: PartnerChainRuntime>(
159	StartNodeConfig { substrate_node_base_path }: StartNodeConfig,
160	StartNodeChainConfig {
161		cardano:
162			CardanoParameters {
163				security_parameter,
164				active_slots_coeff,
165				first_epoch_number,
166				first_slot_number,
167				epoch_duration_millis,
168				first_epoch_timestamp_millis,
169				slot_duration_millis,
170			},
171		bootnodes,
172	}: StartNodeChainConfig,
173	db_connection_string: &str,
174	data_source: &str,
175	context: &C,
176) -> anyhow::Result<()> {
177	let executable = context.current_executable()?;
178	let environment_prefix = format!(
179		"CARDANO_SECURITY_PARAMETER='{security_parameter}' \\
180         CARDANO_ACTIVE_SLOTS_COEFF='{active_slots_coeff}' \\
181         DB_SYNC_POSTGRES_CONNECTION_STRING='{db_connection_string}' \\
182         CARDANO_DATA_SOURCE='{data_source}' \\
183         MC__FIRST_EPOCH_TIMESTAMP_MILLIS='{first_epoch_timestamp_millis}' \\
184         MC__EPOCH_DURATION_MILLIS='{epoch_duration_millis}' \\
185         MC__SLOT_DURATION_MILLIS='{slot_duration_millis}' \\
186         MC__FIRST_EPOCH_NUMBER='{first_epoch_number}' \\
187         MC__FIRST_SLOT_NUMBER='{first_slot_number}' \\
188         BLOCK_STABILITY_MARGIN='0' \\
189"
190	);
191
192	let bootnodes = bootnodes
193		.iter()
194		.map(|bootnode| format!("--bootnodes {}", bootnode))
195		.collect::<Vec<String>>()
196		.join(" ");
197
198	let ws_port = NODE_P2P_PORT.save_if_empty(
199		NODE_P2P_PORT
200			.default
201			.expect("Default NODE_WS_PORT should always be set")
202			.parse()
203			.expect("Default NODE_WS_PORT should be valid u16"),
204		context,
205	);
206	let keystore_path = keystore_path(&substrate_node_base_path);
207	let network_key_path = network_key_path(&substrate_node_base_path);
208	context.run_command(&format!(
209		"{environment_prefix} {executable} --validator --chain {CHAIN_SPEC_PATH} --base-path {substrate_node_base_path} --keystore-path {keystore_path} --node-key-file {network_key_path} --port {ws_port} {bootnodes}",
210	))?;
211
212	Ok(())
