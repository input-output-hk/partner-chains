16/// SessionManager, which takes committee from pallet_session_validator_management.
17impl<T: crate::Config + pallet_session::Config>
18	pallet_partner_chains_session::SessionManager<T::AccountId, T::AuthorityKeys>
19	for ValidatorManagementSessionManager<T>
20{
21	fn new_session_genesis(
22		_new_index: SessionIndex,
23	) -> Option<Vec<(T::AccountId, T::AuthorityKeys)>> {
24		Some(
25			crate::Pallet::<T>::current_committee_storage()
26				.committee
27				.into_iter()
28				.map(|member| (member.authority_id().into(), member.authority_keys()))
29				.collect::<Vec<_>>(),
30		)
31	}
32
33	// Instead of Some((*).expect) we could just use (*). However, we rather panic in presence of
34	// important programming errors.
35	fn new_session(new_index: SessionIndex) -> Option<Vec<(T::AccountId, T::AuthorityKeys)>> {
36		info!("New session {new_index}");
37		pallet_session::pallet::CurrentIndex::<T>::put(new_index);
38		Some(
39			crate::Pallet::<T>::rotate_committee_to_next_epoch()
40				.expect(
41					"Session should never end without current epoch validators defined. \
42				Check ShouldEndSession implementation or if it is used before starting new session",
43				)
44				.into_iter()
45				.map(|member| (member.authority_id().into(), member.authority_keys()))
46				.collect(),
47		)
48	}
49
50	fn end_session(end_index: SessionIndex) {
51		info!("End session {end_index}");
52	}
53
54	// Session is expected to be at least 1 block behind sidechain epoch.
55	fn start_session(start_index: SessionIndex) {
56		let epoch_number = T::current_epoch_number();
57		info!("Start session {start_index}, epoch {epoch_number}");
58	}
59}

61/// This implementation tries to end each session in the first block of each sidechain epoch in which
62/// the committee for the epoch is defined.
63impl<T, ScEpochNumber> pallet_partner_chains_session::ShouldEndSession<BlockNumberFor<T>>
64	for ValidatorManagementSessionManager<T>
65where
66	T: crate::Config<ScEpochNumber = ScEpochNumber>,
67	ScEpochNumber: Clone + PartialOrd,
68{
69	fn should_end_session(_n: BlockNumberFor<T>) -> bool {
70		let current_epoch_number = T::current_epoch_number();
71
72		current_epoch_number > crate::Pallet::<T>::current_committee_storage().epoch
73			&& crate::Pallet::<T>::next_committee().is_some()
74	}
75}
