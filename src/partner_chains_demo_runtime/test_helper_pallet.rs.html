89	impl<T: Config> sp_governed_map::OnGovernedMappingChange<MaxKeyLength, MaxValueLength>
90		for Pallet<T>
91	{
92		fn on_governed_mapping_change(
93			key: BoundedString<MaxKeyLength>,
94			new_value: Option<BoundedVec<u8, MaxValueLength>>,
95			old_value: Option<BoundedVec<u8, MaxValueLength>>,
96		) {
97			match (new_value, old_value) {
98				(Some(new_value), Some(old_value)) => log::info!(
99					"Governed Map value for key '{key}' has changed: {} → {}",
100					to_hex(&old_value, false),
101					to_hex(&new_value, false)
102				),
103				(Some(new_value), None) => {
104					log::info!(
105						"New Governed Map value for key '{key}': {}",
106						to_hex(&new_value, false)
107					)
108				},
109
110				(None, Some(old_value)) => {
111					log::info!(
112						"Governed Map value for key '{key}' deleted, old value: {}",
113						to_hex(&old_value, false)
114					)
115				},
116				_ => { /* technically unreachable */ },
117			}
118		}
119	}

170				.map(|data| Self::Call::handle_participation_data { data });

173		}
