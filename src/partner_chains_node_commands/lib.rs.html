79#[allow(clippy::large_enum_variant)]
80pub enum PartnerChainsSubcommand<
81	RuntimeBindings: PartnerChainRuntime + PartnerChainRuntimeBindings,
82	PartnerchainAddress: Clone + Sync + Send + FromStr + 'static,
83> {
84	/// Returns sidechain parameters
85	SidechainParams(SidechainParamsCmd),
86
87	/// Returns registration status for a given mainchain public key and epoch number.
88	/// If registration has been included in Cardano block in epoch N, then it should be returned by this command if epoch greater than N+1 is provided.
89	/// If this command won't show your registration after a few minutes after it has been included in a cardano block, you can start debugging for unsuccessful registration.
90	#[clap(
91		after_help = "Example: partner-chains-node -- registration-status --stake-pool-pub-key 0x702b81ab2e86cf73a87062af1eb0da666d451976d9d91c63a119ed94e6a33dc0 --mc-epoch-number 586"
92	)]
93	RegistrationStatus(RegistrationStatusCmd),
94
95	/// Returns ariadne parameters effective at given mainchain epoch number.
96	/// Parameters are effective two epochs after the block their change is included in.
97	AriadneParameters(AriadneParametersCmd),
98
99	/// Generates registration signatures for partner chains committee candidates
100	RegistrationSignatures(RegistrationSignaturesCmd),
101
102	/// Signs address association
103	SignAddressAssociation(AddressAssociationSignaturesCmd<PartnerchainAddress>),
104
105	/// Signs block producer metadata for submitting to the runtime
106	SignBlockProducerMetadata(BlockProducerMetadataSignatureCmd),
107
108	/// Commands for interacting with Partner Chain smart contracts on Cardano
109	#[command(subcommand)]
110	SmartContracts(SmartContractsCmd),
111
112	/// Partner Chains text "wizards" for setting up chain
113	#[command(subcommand)]
114	Wizards(partner_chains_cli::Command<RuntimeBindings>),
115}

117pub fn run<
118	Cli,
119	Block,
120	CommitteeMember,
121	Client,
122	BlockProducerMetadata,
123	RuntimeBindings: PartnerChainRuntime + PartnerChainRuntimeBindings,
124	PartnerchainAddress,
125>(
126	cli: &Cli,
127	get_deps: impl FnOnce(
128		sc_service::Configuration,
129	) -> Result<
130		(Arc<Client>, TaskManager, Arc<dyn AuthoritySelectionDataSource + Send + Sync>),
131		sc_service::error::Error,
132	>,
133	cmd: PartnerChainsSubcommand<RuntimeBindings, PartnerchainAddress>,
134) -> sc_cli::Result<()>
135where
136	Cli: SubstrateCli,
137	Client: ProvideRuntimeApi<Block> + HeaderBackend<Block> + 'static,
138	Client::Api: GetGenesisUtxo<Block>
139		+ GetSidechainStatus<Block>
140		+ SessionValidatorManagementApi<
141			Block,
142			CommitteeMember,
143			AuthoritySelectionInputs,
144			ScEpochNumber,
145		> + CandidateValidationApi<Block>,
146	Block: BlockT,
147	NumberFor<Block>: From<u32> + Into<u32>,
148	CommitteeMember: CommitteeMemberT + Encode + Decode + Send + Sync + 'static,
149	CommitteeMember::AuthorityId: Decode + Encode + AsRef<[u8]> + Send + Sync + 'static,
150	CommitteeMember::AuthorityKeys: Decode + Encode,
151	BlockProducerMetadata: DeserializeOwned + Encode + Send + Sync,
152	PartnerchainAddress: Serialize + Clone + Sync + Send + FromStr + Encode + 'static,
153{
154	match cmd {
155		PartnerChainsSubcommand::SidechainParams(cmd) => {
156			let runner = cli.create_runner(&cmd)?;
157			runner.async_run(|config| {
158				let (client, task_manager, _) = get_deps(config)?;
159				Ok((print_result(sp_sidechain::query::get_genesis_utxo(client)), task_manager))
160			})
161		},
162		PartnerChainsSubcommand::RegistrationStatus(cmd) => {
163			let runner = cli.create_runner(&cmd)?;
164			runner.async_run(move |config| {
165				let (client, task_manager, ds) = get_deps(config)?;
166				let query = SessionValidatorManagementQuery::new(client.clone(), ds.clone());
167				Ok((
168					print_result(cli_get_registration_status(
169						query,
170						cmd.mc_epoch_number,
171						cmd.stake_pool_pub_key.clone(),
172					)),
173					task_manager,
174				))
175			})
176		},
177		PartnerChainsSubcommand::AriadneParameters(cmd) => {
178			let runner = cli.create_runner(&cmd)?;
179			runner.async_run(move |config| {
180				let (client, task_manager, ds) = get_deps(config)?;
181				let query = SessionValidatorManagementQuery::new(client.clone(), ds.clone());
182				Ok((
183					print_result(cli_get_ariadne_parameters(query, cmd.mc_epoch_number)),
184					task_manager,
185				))
186			})
187		},
188		PartnerChainsSubcommand::RegistrationSignatures(cmd) => Ok(println!("{}", cmd.execute())),
189		PartnerChainsSubcommand::SignAddressAssociation(cmd) => {
190			cmd.execute().map_err(|e| sc_service::Error::Application(e.into()))?;
191			Ok(())
192		},
193		PartnerChainsSubcommand::SignBlockProducerMetadata(cmd) => {
194			cmd.execute::<BlockProducerMetadata>()
195				.map_err(|e| sc_service::Error::Application(e.into()))?;
196			Ok(())
197		},
198		PartnerChainsSubcommand::SmartContracts(cmd) => {
199			setup_log4rs()?;
200			Ok(cmd.execute_blocking()?)
201		},
202		PartnerChainsSubcommand::Wizards(cmd) => {
203			setup_log4rs()?;
204			Ok(cmd
205				.run(&DefaultCmdRunContext)
206				.map_err(|e| sc_cli::Error::Application(e.into()))?)
207		},
208	}
209}

211/// This sets logging to stderr, leaving stdout for smart-contracts JSON outputs.
212/// Ogmios interactions are logged to a file.
213fn setup_log4rs() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
214	let stderr = log4rs::append::console::ConsoleAppender::builder()
215		.target(log4rs::append::console::Target::Stderr)
216		.build();
217	let ogmios_log = log4rs::append::file::FileAppender::builder().build("ogmios_client.log")?;
218
219	let log_config = log4rs::config::Config::builder()
220		.appender(log4rs::config::Appender::builder().build("stderr", Box::new(stderr)))
221		.appender(log4rs::config::Appender::builder().build("ogmios-log", Box::new(ogmios_log)))
222		.logger(
223			log4rs::config::Logger::builder()
224				.appender("ogmios-log")
225				.additive(false)
226				.build("ogmios_client::jsonrpsee", log::LevelFilter::Debug),
227		)
228		.build(log4rs::config::Root::builder().appender("stderr").build(log::LevelFilter::Info))?;
229
230	log4rs::init_config(log_config)?;
231
232	Ok(())
233}

235async fn print_result<FIn>(command_future: FIn) -> Result<(), sc_cli::Error>
236where
237	FIn: Future<Output = Result<String, String>>,

239	let result = command_future.await.unwrap_or_else(|e| e);
240	println!("{}", result);
241	Ok(())
242}
