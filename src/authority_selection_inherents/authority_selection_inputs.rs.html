6/// The part of data for selection of authorities that comes from the main chain.
7/// It is unfiltered, so the selection algorithm should filter out invalid candidates.
8#[derive(Clone, Debug, Encode, Decode, DecodeWithMemTracking, TypeInfo, PartialEq, Eq)]
9pub struct AuthoritySelectionInputs {
10	pub d_parameter: DParameter,
11	pub permissioned_candidates: Vec<PermissionedCandidateData>,
12	pub registered_candidates: Vec<CandidateRegistrations>,
13	pub epoch_nonce: EpochNonce,
14}

16#[cfg(feature = "std")]
17#[derive(Debug, thiserror::Error)]
18pub enum AuthoritySelectionInputsCreationError {
19	#[cfg_attr(feature = "std", error("Failed to get Ariadne parameters for epoch: {0}, D-parameter: {1:?}, permissioned candidates: {2:?}: {3}"))]
20	AriadneParametersQuery(
21		McEpochNumber,
22		PolicyId,
23		PolicyId,
24		Box<dyn std::error::Error + Send + Sync>,
25	),
26	#[cfg_attr(feature = "std", error("Failed to get registered candidates for epoch: {0}, committee candidate address: {1}: {2}."))]
27	GetCandidatesQuery(McEpochNumber, String, Box<dyn std::error::Error + Send + Sync>),
28	#[cfg_attr(feature = "std", error("Failed to get epoch nonce for epoch: {0}: {1}."))]
29	GetEpochNonceQuery(McEpochNumber, Box<dyn std::error::Error + Send + Sync>),
30}

45/// Queries about the Authority Candidates
46#[cfg(feature = "std")]
47#[async_trait::async_trait]
48pub trait AuthoritySelectionDataSource {
49	/// Returns D-parameter and list of permissioned candidates that is effective for the given epoch.
50	/// The data from the latest block of `data_epoch(epoch)` will be used if available, otherwise returns data at the latest block of the chain.
51	async fn get_ariadne_parameters(
52		&self,
53		epoch_number: McEpochNumber,
54		d_parameter: PolicyId,
55		permissioned_candidates: PolicyId,
56	) -> Result<AriadneParameters, Box<dyn std::error::Error + Send + Sync>>;
57
58	/// Returns the list of registrations that is effective for the given epoch.
59	/// The data from the latest block of `data_epoch(epoch)` will be used if available, otherwise returns data at the latest block of the chain.
60	/// Each item is a list of one candidate registrations.
61	async fn get_candidates(
62		&self,
63		epoch: McEpochNumber,
64		committee_candidate_address: MainchainAddress,
65	) -> Result<Vec<CandidateRegistrations>, Box<dyn std::error::Error + Send + Sync>>;
66
67	/// Returns Cardano Epoch Nonce. None, if the nonce for given epoch is not known yet.
68	async fn get_epoch_nonce(
69		&self,
70		epoch: McEpochNumber,
71	) -> Result<Option<EpochNonce>, Box<dyn std::error::Error + Send + Sync>>;
72
73	///
74	/// # Arguments
75	///
76	/// * `for_epoch`: main chain epoch number during which candidate data is meant to be used
77	///
78	/// returns: Result<McEpochNumber, Box<dyn std::error::Error + Send + Sync>> - data source methods called with `for_epoch` will query only for data which was stored on main chain in the returned epoch or earlier
79	///
80	///
81	async fn data_epoch(
82		&self,
83		for_epoch: McEpochNumber,
84	) -> Result<McEpochNumber, Box<dyn std::error::Error + Send + Sync>>;
85}
86
87impl AuthoritySelectionInputs {
88	#[cfg(feature = "std")]
89	pub async fn from_mc_data(
90		candidate_data_source: &(dyn AuthoritySelectionDataSource + Send + Sync),
91		for_epoch: McEpochNumber,
92		scripts: sp_session_validator_management::MainChainScripts,
93	) -> Result<Self, AuthoritySelectionInputsCreationError> {
94		let ariadne_parameters_response = candidate_data_source
95			.get_ariadne_parameters(
96				for_epoch,
97				scripts.d_parameter_policy_id.clone(),
98				scripts.permissioned_candidates_policy_id.clone(),
99			)
100			.await
101			.map_err(|err| {
102				AuthoritySelectionInputsCreationError::AriadneParametersQuery(
103					for_epoch,
104					scripts.d_parameter_policy_id.clone(),
105					scripts.permissioned_candidates_policy_id.clone(),
106					err,
107				)
108			})?;
109
110		let d_parameter = ariadne_parameters_response.d_parameter;
111		let no_permissioned_candidates_expected = d_parameter.num_permissioned_candidates == 0;
112		let permissioned_candidates = match ariadne_parameters_response.permissioned_candidates {
113			None if no_permissioned_candidates_expected => Vec::new(),
114			None => {
115				return Err(AuthoritySelectionInputsCreationError::AriadneParametersQuery(
116					for_epoch,
117					scripts.d_parameter_policy_id,
118					scripts.permissioned_candidates_policy_id,
119					("Expected Data Not Found: Permissioned Candidates List".to_string()).into(),
120				))
121			},
122			Some(permissioned_candidates) => permissioned_candidates
123				.into_iter()
124				.map(|candidate| PermissionedCandidateData {
125					sidechain_public_key: candidate.sidechain_public_key,
126					aura_public_key: candidate.aura_public_key,
127					grandpa_public_key: candidate.grandpa_public_key,
128				})
129				.collect::<Vec<PermissionedCandidateData>>(),
130		};
131
132		let registered_candidates: Vec<CandidateRegistrations> = candidate_data_source
133			.get_candidates(for_epoch, scripts.committee_candidate_address.clone())
134			.await
135			.map_err(|err| {
136				AuthoritySelectionInputsCreationError::GetCandidatesQuery(
137					for_epoch,
138					scripts.committee_candidate_address.to_string(),
139					err,
140				)
141			})?;
142		let epoch_nonce_response =
143			candidate_data_source.get_epoch_nonce(for_epoch).await.map_err(|err| {
144				AuthoritySelectionInputsCreationError::GetEpochNonceQuery(for_epoch, err)
145			})?;
146		let epoch_nonce = epoch_nonce_response.unwrap_or(EpochNonce(vec![]));
147
148		Ok(Self { d_parameter, permissioned_candidates, registered_candidates, epoch_nonce })
149	}
