40#[derive(Clone, Debug, Encode, Decode, PartialEq, Eq, PartialOrd, Ord)]
41pub enum Candidate<TAccountId, TAccountKeys> {
42	Permissioned(PermissionedCandidate<TAccountId, TAccountKeys>),
43	Registered(CandidateWithStake<TAccountId, TAccountKeys>),
44}
45
46impl<AuthorityId, AuthorityKeys> From<Candidate<AuthorityId, AuthorityKeys>>
47	for CommitteeMember<AuthorityId, AuthorityKeys>
48{
49	fn from(candidate: Candidate<AuthorityId, AuthorityKeys>) -> Self {
50		match candidate {
51			Candidate::Permissioned(member) => {
52				Self::Permissioned { id: member.account_id, keys: member.account_keys }
53			},
54			Candidate::Registered(member) => Self::Registered {
55				id: member.account_id,
56				keys: member.account_keys,
57				stake_pool_pub_key: member.stake_pool_pub_key,
58			},
59		}
60	}
61}
62
63impl<TAccountId, TAccountKeys> Candidate<TAccountId, TAccountKeys> {
64	pub fn account_id(&self) -> &TAccountId {
65		match self {
66			Candidate::Permissioned(c) => &c.account_id,
67			Candidate::Registered(c) => &c.account_id,
68		}
69	}
70
71	pub fn account_keys(&self) -> &TAccountKeys {
72		match self {
73			Candidate::Permissioned(c) => &c.account_keys,
74			Candidate::Registered(c) => &c.account_keys,
75		}
76	}
77
78	pub fn stake_delegation(&self) -> Option<StakeDelegation> {
79		match self {
80			Candidate::Permissioned(_) => None,
81			Candidate::Registered(c) => Some(c.stake_delegation),
82		}
83	}
84}

86/// Get the valid trustless candidates from the registrations from inherent data
87pub fn filter_trustless_candidates_registrations<TAccountId, TAccountKeys>(
88	candidate_registrations: Vec<CandidateRegistrations>,
89	genesis_utxo: UtxoId,
90) -> Vec<(Candidate<TAccountId, TAccountKeys>, selection::Weight)>
91where
92	TAccountKeys: From<(sr25519::Public, ed25519::Public)>,
93	TAccountId: From<ecdsa::Public>,
94{
95	candidate_registrations
96		.into_iter()
97		.flat_map(|candidate_registrations| {
98			select_latest_valid_candidate(candidate_registrations, genesis_utxo)
99		})
100		.map(|c| {
101			let weight = c.stake_delegation.0.into();
102			(Candidate::Registered(c), weight)
103		})
104		.collect()
105}

107pub fn filter_invalid_permissioned_candidates<TAccountId, TAccountKeys>(
108	permissioned_candidates: Vec<PermissionedCandidateData>,
109) -> Vec<Candidate<TAccountId, TAccountKeys>>
110where
111	TAccountKeys: From<(sr25519::Public, ed25519::Public)>,
112	TAccountId: TryFrom<sidechain_domain::SidechainPublicKey>,
113{
114	permissioned_candidates
115		.into_iter()
116		.filter_map(|candidate| {
117			let (account_id, aura_key, grandpa_key) =
118				validate_permissioned_candidate_data(candidate).ok()?;
119			let account_keys = (aura_key, grandpa_key).into();
120			Some(Candidate::Permissioned(PermissionedCandidate { account_id, account_keys }))
121		})
122		.collect()
123}

125fn select_latest_valid_candidate<TAccountId, TAccountKeys>(
126	candidate_registrations: CandidateRegistrations,
127	genesis_utxo: UtxoId,
128) -> Option<CandidateWithStake<TAccountId, TAccountKeys>>
129where
130	TAccountId: From<ecdsa::Public>,
131	TAccountKeys: From<(sr25519::Public, ed25519::Public)>,
132{
133	let stake_delegation = validate_stake(candidate_registrations.stake_delegation).ok()?;
134	let stake_pool_pub_key = candidate_registrations.stake_pool_public_key;
135
136	let ((account_id, account_keys), _) = candidate_registrations
137		.registrations
138		.into_iter()
139		.filter_map(|registration_data| {
140			match validate_registration_data(&stake_pool_pub_key, &registration_data, genesis_utxo)
141			{
142				Ok(candidate) => Some((candidate, registration_data.utxo_info)),
143				Err(_) => None,
144			}
145		})
146		// Get the latest valid registration of the authority candidate
147		.max_by_key(|(_, utxo_info)| utxo_info.ordering_key())?;
148
149	Some(CandidateWithStake {
150		account_id: account_id.into(),
151		account_keys: account_keys.into(),
152		stake_delegation,
153		stake_pool_pub_key,
154	})
155}

204pub fn validate_permissioned_candidate_data<AccountId: TryFrom<SidechainPublicKey>>(
205	candidate: PermissionedCandidateData,
206) -> Result<(AccountId, sr25519::Public, ed25519::Public), PermissionedCandidateDataError> {
207	Ok((
208		candidate
209			.sidechain_public_key
210			.try_into()
211			.map_err(|_| PermissionedCandidateDataError::InvalidSidechainPubKey)?,
212		candidate
213			.aura_public_key
214			.try_into_sr25519()
215			.ok_or(PermissionedCandidateDataError::InvalidAuraKey)?,
216		candidate
217			.grandpa_public_key
218			.try_into_ed25519()
219			.ok_or(PermissionedCandidateDataError::InvalidGrandpaKey)?,
220	))
221}

223/// Is the registration data provided by the authority candidate valid?
224pub fn validate_registration_data(
225	stake_pool_pub_key: &StakePoolPublicKey,
226	registration_data: &RegistrationData,
227	genesis_utxo: UtxoId,
228) -> Result<(ecdsa::Public, (sr25519::Public, ed25519::Public)), RegistrationDataError> {
229	let aura_pub_key = registration_data
230		.aura_pub_key
231		.try_into_sr25519()
232		.ok_or(RegistrationDataError::InvalidAuraKey)?;
233	let grandpa_pub_key = registration_data
234		.grandpa_pub_key
235		.try_into_ed25519()
236		.ok_or(RegistrationDataError::InvalidGrandpaKey)?;
237	let sidechain_pub_key = ecdsa::Public::from(
238		<[u8; 33]>::try_from(registration_data.sidechain_pub_key.0.clone())
239			.map_err(|_| RegistrationDataError::InvalidSidechainPubKey)?,
240	);
241
242	let signed_message = RegisterValidatorSignedMessage {
243		genesis_utxo,
244		sidechain_pub_key: registration_data.sidechain_pub_key.0.clone(),
245		registration_utxo: registration_data.registration_utxo,
246	};
247
248	let signed_message_encoded = minicbor::to_vec(signed_message.to_datum())
249		.expect("`RegisterValidatorSignedMessage` should always be encodable");
250
251	verify_stake_pool_signature(stake_pool_pub_key, registration_data, &signed_message_encoded)?;
252	verify_sidechain_signature(
253		sidechain_pub_key,
254		&registration_data.sidechain_signature,
255		&signed_message_encoded,
256	)?;
257	verify_tx_inputs(registration_data)?;
258
259	// TODO - Stake Validation: https://input-output.atlassian.net/browse/ETCM-4082
260
261	Ok((sidechain_pub_key, (aura_pub_key, grandpa_pub_key)))
262}

264pub fn validate_stake(stake: Option<StakeDelegation>) -> Result<StakeDelegation, StakeError> {
265	match stake {
266		None => Err(StakeError::UnknownStake),
267		Some(stake) => {
268			if stake.is_zero() {
269				Err(StakeError::InvalidStake)
270			} else {
271				Ok(stake)
272			}
273		},
274	}
275}

277fn verify_stake_pool_signature(
278	stake_pool_pub_key: &StakePoolPublicKey,
279	registration_data: &RegistrationData,
280	signed_message_encoded: &[u8],
281) -> Result<(), RegistrationDataError> {
282	let spo_signature: [u8; 64] = registration_data
283		.mainchain_signature
284		.0
285		.try_into()
286		.map_err(|_| RegistrationDataError::InvalidMainchainSignature)?;
287	let spo_signature = ed25519::Signature::from(spo_signature);
288	if spo_signature.verify(signed_message_encoded, &ed25519::Public::from(stake_pool_pub_key.0)) {
289		Ok(())
290	} else {
291		Err(RegistrationDataError::InvalidMainchainSignature)
292	}
293}

295// Pre EIP155 signature.v convention
296const NEGATIVE_POINT_SIGN: u8 = 27;
297const POSITIVE_POINT_SIGN: u8 = 28;
298
299fn verify_sidechain_signature(
300	pub_key: ecdsa::Public,
301	signature: &SidechainSignature,
302	signed_message_encoded: &[u8],
303) -> Result<(), RegistrationDataError> {
304	let sidechain_signature = <[u8; 64]>::try_from(signature.0.clone())
305		.map_err(|_| RegistrationDataError::InvalidSidechainSignature)?;
306	let mut sidechain_signature_with_v = [0u8; 65];
307	sidechain_signature_with_v[..64].copy_from_slice(&sidechain_signature);
308	// TODO: Extract to crypto util. See https://github.com/input-output-hk/partner-chains/pull/61#discussion_r1293205895
309	let is_valid = [NEGATIVE_POINT_SIGN, POSITIVE_POINT_SIGN].iter().any(|v| {
310		sidechain_signature_with_v[64] = *v;
311		ecdsa::Signature::from(sidechain_signature_with_v).verify(signed_message_encoded, &pub_key)
312	});
313	if is_valid { Ok(()) } else { Err(RegistrationDataError::InvalidSidechainSignature) }
314}
315
316fn verify_tx_inputs(registration_data: &RegistrationData) -> Result<(), RegistrationDataError> {
317	if registration_data.tx_inputs.contains(&registration_data.registration_utxo) {
318		Ok(())
319	} else {
320		Err(RegistrationDataError::InvalidTxInput)
321	}
322}
