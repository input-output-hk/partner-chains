14/// Signed Message of the Authority Candidate to register
15/// It's ToDatum implementation has to produce datum that has to match main chain structure,
16/// because it will be hashed and signed for signature verification.
17#[derive(Debug, ToDatum)]
18pub struct RegisterValidatorSignedMessage {
19	/// Genesis UTxO identifying the Partner Chain
20	pub genesis_utxo: UtxoId,
21	/// Partner Chain public key of the registered candidate
22	pub sidechain_pub_key: Vec<u8>,
23	/// UTxO that is an input parameter to the registration transaction.
24	/// It is spent during the registration process. Prevents replay attacks.
25	pub registration_utxo: UtxoId,
26}

60impl<AuthorityId, AuthorityKeys> From<Candidate<AuthorityId, AuthorityKeys>>
61	for CommitteeMember<AuthorityId, AuthorityKeys>
62{
63	fn from(candidate: Candidate<AuthorityId, AuthorityKeys>) -> Self {
64		match candidate {
65			Candidate::Permissioned(member) => {
66				Self::Permissioned { id: member.account_id, keys: member.account_keys }
67			},
68			Candidate::Registered(member) => Self::Registered {
69				id: member.account_id,
70				keys: member.account_keys,
71				stake_pool_pub_key: member.stake_pool_pub_key,
72			},
73		}
74	}
75}

77impl<TAccountId, TAccountKeys> Candidate<TAccountId, TAccountKeys> {
78	/// Returns the account id of the candidate
79	pub fn account_id(&self) -> &TAccountId {
80		match self {
81			Candidate::Permissioned(c) => &c.account_id,
82			Candidate::Registered(c) => &c.account_id,
83		}
84	}
85
86	/// Returns the account keys of the candidate
87	pub fn account_keys(&self) -> &TAccountKeys {
88		match self {
89			Candidate::Permissioned(c) => &c.account_keys,
90			Candidate::Registered(c) => &c.account_keys,
91		}
92	}
93
94	/// Returns the stake delegation of the candidate
95	pub fn stake_delegation(&self) -> Option<StakeDelegation> {
96		match self {
97			Candidate::Permissioned(_) => None,
98			Candidate::Registered(c) => Some(c.stake_delegation),
99		}
100	}

103/// Get the valid trustless candidates from the registrations from inherent data
104pub fn filter_trustless_candidates_registrations<TAccountId, TAccountKeys: MaybeFromCandidateKeys>(
105	candidate_registrations: Vec<CandidateRegistrations>,
106	genesis_utxo: UtxoId,
107) -> Vec<(Candidate<TAccountId, TAccountKeys>, selection::Weight)>
108where
109	TAccountId: From<ecdsa::Public>,
110{
111	candidate_registrations
112		.into_iter()
113		.flat_map(|candidate_registrations| {
114			select_latest_valid_candidate::<TAccountId, TAccountKeys>(
115				candidate_registrations,
116				genesis_utxo,
117			)
118		})
119		.map(|c| {
120			let weight = c.stake_delegation.0.into();
121			(Candidate::Registered(c), weight)
122		})
123		.collect()
124}
125/// Filters invalid candidates from a list of [PermissionedCandidateData].
126pub fn filter_invalid_permissioned_candidates<TAccountId, TAccountKeys: MaybeFromCandidateKeys>(
127	permissioned_candidates: Vec<PermissionedCandidateData>,
128) -> Vec<Candidate<TAccountId, TAccountKeys>>
129where
130	TAccountId: From<ecdsa::Public>,
131{
132	permissioned_candidates
133		.into_iter()
134		.filter_map(|candidate| {
135			let (partner_chain_key, account_keys) =
136				validate_permissioned_candidate_data::<TAccountKeys>(candidate).ok()?;
137			let account_id = partner_chain_key.into();
138			Some(Candidate::Permissioned(PermissionedCandidate { account_id, account_keys }))
139		})
140		.collect()
141}

143fn select_latest_valid_candidate<TAccountId, TAccountKeys: MaybeFromCandidateKeys>(
144	candidate_registrations: CandidateRegistrations,
145	genesis_utxo: UtxoId,
146) -> Option<CandidateWithStake<TAccountId, TAccountKeys>>
147where
148	TAccountId: From<ecdsa::Public>,
149{
150	let stake_delegation = validate_stake(candidate_registrations.stake_delegation).ok()?;
151	let stake_pool_pub_key = candidate_registrations.stake_pool_public_key;
152
153	let ((account_id, account_keys), _) = candidate_registrations
154		.registrations
155		.into_iter()
156		.filter_map(|registration_data| {
157			match validate_registration_data::<TAccountKeys>(
158				&stake_pool_pub_key,
159				&registration_data,
160				genesis_utxo,
161			) {
162				Ok(candidate) => Some((candidate, registration_data.utxo_info)),
163				Err(_) => None,
164			}
165		})
166		// Get the latest valid registration of the authority candidate
167		.max_by_key(|(_, utxo_info)| utxo_info.ordering_key())?;
168
169	Some(CandidateWithStake {
170		account_id: account_id.into(),
171		account_keys,
172		stake_delegation,
173		stake_pool_pub_key,
174	})
175}

230/// Validates Account keys and Partner Chain public keys of [PermissionedCandidateData].
231pub fn validate_permissioned_candidate_data<TAccountKeys: MaybeFromCandidateKeys>(
232	candidate: PermissionedCandidateData,
233) -> Result<(ecdsa::Public, TAccountKeys), PermissionedCandidateDataError> {
234	let ecdsa_bytes: [u8; 33] = candidate
235		.sidechain_public_key
236		.0
237		.try_into()
238		.map_err(|_| PermissionedCandidateDataError::InvalidSidechainPubKey)?;
239	Ok((
240		ecdsa_bytes.into(),
241		MaybeFromCandidateKeys::maybe_from(&candidate.keys)
242			.ok_or_else(|| PermissionedCandidateDataError::InvalidAccountKeys)?,
243	))
244}

246/// Validates registration data provided by the authority candidate.
247///
248/// Validates:
249/// * Account keys and Partner Chain public keys of the candidate
250/// * stake pool signature
251/// * sidechain signature
252/// * transaction inputs contain correct registration utxo
253pub fn validate_registration_data<TAccountKeys: MaybeFromCandidateKeys>(
254	stake_pool_pub_key: &StakePoolPublicKey,
255	registration_data: &RegistrationData,
256	genesis_utxo: UtxoId,
257) -> Result<(ecdsa::Public, TAccountKeys), RegistrationDataError> {
258	let account_keys = MaybeFromCandidateKeys::maybe_from(&registration_data.keys)
259		.ok_or(RegistrationDataError::InvalidAccountKeys)?;
260	let sidechain_pub_key = ecdsa::Public::from(
261		<[u8; 33]>::try_from(registration_data.sidechain_pub_key.0.clone())
262			.map_err(|_| RegistrationDataError::InvalidSidechainPubKey)?,
263	);
264
265	let signed_message = RegisterValidatorSignedMessage {
266		genesis_utxo,
267		sidechain_pub_key: registration_data.sidechain_pub_key.0.clone(),
268		registration_utxo: registration_data.registration_utxo,
269	};
270
271	let signed_message_encoded = minicbor::to_vec(signed_message.to_datum())
272		.expect("`RegisterValidatorSignedMessage` should always be encodable");
273
274	verify_stake_pool_signature(stake_pool_pub_key, registration_data, &signed_message_encoded)?;
275	verify_sidechain_signature(
276		sidechain_pub_key,
277		&registration_data.sidechain_signature,
278		&signed_message_encoded,
279	)?;
280	verify_tx_inputs(registration_data)?;
281
282	Ok((sidechain_pub_key, account_keys))
283}

285/// Validates stake delegation. Stake must be known and positive.
286pub fn validate_stake(stake: Option<StakeDelegation>) -> Result<StakeDelegation, StakeError> {
287	match stake {
288		None => Err(StakeError::UnknownStake),
289		Some(stake) => {
290			if stake.is_zero() {
291				Err(StakeError::InvalidStake)
292			} else {
293				Ok(stake)
294			}
295		},
296	}
297}
298
299fn verify_stake_pool_signature(
300	stake_pool_pub_key: &StakePoolPublicKey,
301	registration_data: &RegistrationData,
302	signed_message_encoded: &[u8],
303) -> Result<(), RegistrationDataError> {
304	let spo_signature: [u8; 64] = registration_data
305		.mainchain_signature
306		.0
307		.try_into()
308		.map_err(|_| RegistrationDataError::InvalidMainchainSignature)?;
309	let spo_signature = ed25519::Signature::from(spo_signature);
310	if spo_signature.verify(signed_message_encoded, &ed25519::Public::from(stake_pool_pub_key.0)) {
311		Ok(())
312	} else {
313		Err(RegistrationDataError::InvalidMainchainSignature)
314	}
315}
316
317fn verify_sidechain_signature(
318	pub_key: ecdsa::Public,
319	signature: &SidechainSignature,
320	signed_message_encoded: &[u8],
321) -> Result<(), RegistrationDataError> {
322	let sidechain_signature = <[u8; 64]>::try_from(signature.0.clone())
323		.map_err(|_| RegistrationDataError::InvalidSidechainSignature)?;
324	let mut sidechain_signature_with_v = [0u8; 65];
325	sidechain_signature_with_v[..64].copy_from_slice(&sidechain_signature);
326	// TODO: Extract to crypto util. See https://github.com/input-output-hk/partner-chains/pull/61#discussion_r1293205895
327	// Pre EIP155 signature.v convention
328	const NEGATIVE_POINT_SIGN: u8 = 27;
329	const POSITIVE_POINT_SIGN: u8 = 28;
330	let is_valid = [NEGATIVE_POINT_SIGN, POSITIVE_POINT_SIGN].iter().any(|v| {
331		sidechain_signature_with_v[64] = *v;
332		ecdsa::Signature::from(sidechain_signature_with_v).verify(signed_message_encoded, &pub_key)
333	});
334	if is_valid { Ok(()) } else { Err(RegistrationDataError::InvalidSidechainSignature) }
335}
336
337fn verify_tx_inputs(registration_data: &RegistrationData) -> Result<(), RegistrationDataError> {
338	if registration_data.tx_inputs.contains(&registration_data.registration_utxo) {
339		Ok(())
340	} else {
341		Err(RegistrationDataError::InvalidTxInput)
342	}
343}

346	/// Runtime API trait for candidate validation
347	///
348	/// When implementing, make sure that the same validation is used here and in the committee selection logic!
349	pub trait CandidateValidationApi {
350		/// Should validate data provided by registered candidate,
351		/// and return [RegistrationDataError] in case of validation failure.
352		///
353		/// Should validate:
354		/// * Account keys and Partner Chain public keys of the candidate
355		/// * stake pool signature
356		/// * sidechain signature

362		/// Should validate data provided by permissioned candidate,
363		/// and return [PermissionedCandidateDataError] in case of validation failure.
364		///
365		/// Should validate:
366		/// * Account keys and Partner Chain public keys of the candidate
367		fn validate_permissioned_candidate_data(candidate: PermissionedCandidateData) -> Option<PermissionedCandidateDataError>;
368	}

648	#[test]
649	fn should_filter_out_permissioned_candidates_with_invalid_keys() {
650		let valid_sidechain_pub_key = SidechainPublicKey(
651			ecdsa::Pair::from_seed_slice(&[123u8; 32]).unwrap().public().0.to_vec(),
652		);
653		let aura_key_bytes =
654			hex!("d43593c715fdd31c61141abd04a99fd6822c8558854ccde39a5684e7a56da27d");
655		let valid_aura_key = CandidateKey { id: *b"sr25", bytes: aura_key_bytes.to_vec() };
656		let grandpa_key_bytes =
657			hex!("88dc3417d5058ec4b4503e0c12ea1a0a89be200fe98922423d4334014fa6b0ee");
658		let valid_grandpa_key = CandidateKey { id: *b"ed25", bytes: grandpa_key_bytes.to_vec() };
659		let valid_candidate = PermissionedCandidateData {
660			sidechain_public_key: valid_sidechain_pub_key.clone(),
661			keys: CandidateKeys(vec![
662				valid_aura_key.clone().into(),
663				valid_grandpa_key.clone().into(),
664			]),
665		};
666		let permissioned_candidates = vec![
667			PermissionedCandidateData {
668				keys: CandidateKeys(vec![
669					AuraPublicKey(vec![1, 2]).into(),
670					valid_grandpa_key.clone().into(),
671				]),
672				..valid_candidate.clone()
673			},
674			PermissionedCandidateData {
675				sidechain_public_key: SidechainPublicKey(vec![2; 2]),
676				..valid_candidate.clone()
677			},
678			PermissionedCandidateData {
679				keys: CandidateKeys(vec![
680					valid_aura_key.clone().into(),
681					GrandpaPublicKey(vec![3; 4]).into(),
682				]),
683				..valid_candidate.clone()
684			},
685			valid_candidate.clone(),
686		];
687		let valid_candidates = filter_invalid_permissioned_candidates::<AccountId, AccountKeys>(
688			permissioned_candidates,
689		);
690		assert_eq!(valid_candidates.len(), 1);
691		assert_eq!(
692			valid_candidates.first().unwrap().account_id(),
693			&valid_sidechain_pub_key.try_into().unwrap()
694		);
695		assert_eq!(
696			valid_candidates.first().unwrap().account_keys(),
697			&AccountKeys {
698				aura: sp_core::sr25519::Public::from(aura_key_bytes).into(),
699				grandpa: sp_core::ed25519::Public::from(grandpa_key_bytes).into(),
700			}
701		);
702	}
