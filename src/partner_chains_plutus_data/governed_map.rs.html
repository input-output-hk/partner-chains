1use crate::{DataDecodingError, DecodingResult};
2use cardano_serialization_lib::{PlutusData, PlutusList};
3use sidechain_domain::byte_string::ByteString;
4
5#[derive(Clone, Debug, PartialEq)]
6pub struct GovernedMapDatum {
7	pub key: String,
8	pub value: ByteString,
9}
10
11impl GovernedMapDatum {
12	pub fn new(key: String, value: ByteString) -> Self {
13		Self { key, value }
14	}
15}
16
17impl TryFrom<PlutusData> for GovernedMapDatum {
18	type Error = DataDecodingError;
19	fn try_from(datum: PlutusData) -> DecodingResult<Self> {
20		let error = || DataDecodingError {
21			datum: datum.clone(),
22			to: "GovernedMapDatum".into(),
23			msg: "Expected List([UTF8String, Bytes])".into(),
24		};
25
26		datum
27			.as_list()
28			.and_then(|list| {
29				list.get(0)
30					.as_bytes()
31					.and_then(|key| String::from_utf8(key).ok().map(|key| (list, key)))
32			})
33			.and_then(|(list, key)| list.get(1).as_bytes().map(|value| (key, value)))
34			.ok_or_else(error)
35			.map(|(key, value)| Self { key, value: value.into() })
36	}
37}
38
39pub fn governed_map_datum_to_plutus_data(governed_map_datum: &GovernedMapDatum) -> PlutusData {
40	let mut list = PlutusList::new();
41	list.add(&PlutusData::new_bytes(governed_map_datum.key.as_bytes().to_vec()));
42	list.add(&PlutusData::new_bytes(governed_map_datum.value.0.clone()));
43	PlutusData::new_list(&list)
44}
45
46impl From<GovernedMapDatum> for PlutusData {
47	fn from(governed_map_datum: GovernedMapDatum) -> Self {
48		governed_map_datum_to_plutus_data(&governed_map_datum)
49	}
50}
