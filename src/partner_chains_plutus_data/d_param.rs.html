7#[derive(Clone, Debug, PartialEq)]
8pub enum DParamDatum {
9	/// Initial/legacy datum schema. If a datum doesn't contain a version, it is assumed to be V0
10	V0 { num_permissioned_candidates: u16, num_registered_candidates: u16 },
11}
12
13impl TryFrom<PlutusData> for DParamDatum {
14	type Error = DataDecodingError;
15	fn try_from(datum: PlutusData) -> DecodingResult<Self> {
16		Self::decode(&datum)
17	}
18}
19
20impl From<DParamDatum> for sidechain_domain::DParameter {
21	fn from(datum: DParamDatum) -> Self {
22		match datum {
23			DParamDatum::V0 { num_permissioned_candidates, num_registered_candidates } => {
24				Self { num_permissioned_candidates, num_registered_candidates }
25			},
26		}
27	}
28}
29
30pub fn d_parameter_to_plutus_data(d_param: &sidechain_domain::DParameter) -> PlutusData {
31	let mut list = PlutusList::new();
32	list.add(&PlutusData::new_integer(&d_param.num_permissioned_candidates.into()));
33	list.add(&PlutusData::new_integer(&d_param.num_registered_candidates.into()));
34	let appendix = PlutusData::new_list(&list);
35	VersionedGenericDatum {
36		datum: PlutusData::new_empty_constr_plutus_data(&0u64.into()),
37		appendix,
38		version: 0,
39	}
40	.into()
41}
42
43impl VersionedDatumWithLegacy for DParamDatum {
44	const NAME: &str = "DParamDatum";
45
46	fn decode_legacy(data: &PlutusData) -> Result<Self, String> {
47		let d_parameter = data
48			.as_list()
49			.filter(|datum| datum.len() == 2)
50			.and_then(|items| {
51				Some(DParamDatum::V0 {
52					num_permissioned_candidates: items.get(0).as_u16()?,
53					num_registered_candidates: items.get(1).as_u16()?,
54				})
55			})
56			.ok_or("Expected [u16, u16]")?;
57
58		Ok(d_parameter)
59	}
60
61	fn decode_versioned(
62		version: u64,
63		_datum: &PlutusData,
64		appendix: &PlutusData,
65	) -> Result<Self, String> {
66		match version {
67			0 => DParamDatum::decode_legacy(appendix)
68				.map_err(|msg| format!("Cannot parse appendix: {msg}")),
69			_ => Err(format!("Unknown version: {version}")),
70		}
71	}
72}
