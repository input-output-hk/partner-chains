10/** Representation of the plutus type in the mainchain contract (rev 4ed2cc66c554ec8c5bec7b90ad9273e9069a1fb4)
11 *
12 * Note that the ECDSA secp256k1 public key is serialized in compressed format and the
13 * sidechain signature does not contain the recovery bytes (it's just r an s concatenated).
14 *
15 * data BlockProducerRegistration = BlockProducerRegistration
16 * { -- | Verification keys required by the stake ownership model
17 *   -- | @since v4.0.0
18 *  stakeOwnership :: StakeOwnership
19 * , -- | public key in the sidechain's desired format
20 *  sidechainPubKey :: LedgerBytes
21 * , -- | Signature of the sidechain
22 *   -- | @since v4.0.0
23 *  sidechainSignature :: Signature
24 * , -- | A UTxO that must be spent by the transaction
25 *   -- | @since v4.0.0
26 *  inputUtxo :: TxOutRef
27 * , -- | Owner public key hash
28 *   -- | @since v4.0.0
29 *  ownPkh :: PubKeyHash
30 * , -- | Sidechain authority discovery key
31 *   -- | @since Unreleased
32 *   auraKey :: LedgerBytes
33 * , -- | Sidechain grandpa key
34 *   -- | @since Unreleased
35 *   grandpaKey :: LedgerBytes
36 * }
37 */
38#[derive(Clone, Debug, PartialEq, Eq)]
39pub enum RegisterValidatorDatum {
40	/// Initial/legacy datum schema. If a datum doesn't contain a version, it is assumed to be V0
41	V0 {
42		/// Stake ownership information of registered candidate.
43		stake_ownership: AdaBasedStaking,
44		/// Sidechain public key of the candidate. See [SidechainPublicKey] for more details.
45		sidechain_pub_key: SidechainPublicKey,
46		/// Sidechain key signature of the registration message.
47		sidechain_signature: SidechainSignature,
48		/// UTxO id that is a part of the signed registration message.
49		/// It is spent during the registration process. Prevents replay attacks.
50		registration_utxo: UtxoId,
51		/// Hash of the registering SPO's Cardano public key.
52		/// Used by offchain code to find the registration UTXO when re-registering or de-registering.
53		own_pkh: MainchainKeyHash,
54		/// Registering SPO's Aura public key
55		aura_pub_key: AuraPublicKey,
56		/// Registering SPO's GRANDPA public key
57		grandpa_pub_key: GrandpaPublicKey,
58	},
59}

61impl TryFrom<PlutusData> for RegisterValidatorDatum {
62	type Error = DataDecodingError;
63
64	fn try_from(datum: PlutusData) -> DecodingResult<Self> {
65		Self::decode(&datum)
66	}
67}

69impl VersionedDatumWithLegacy for RegisterValidatorDatum {
70	const NAME: &str = "RegisterValidatorDatum";
71
72	fn decode_legacy(data: &PlutusData) -> Result<Self, String> {
73		decode_legacy_register_validator_datum(data).ok_or_else(|| "Invalid data".into())
74	}
75
76	fn decode_versioned(
77		version: u64,
78		datum: &PlutusData,
79		appendix: &PlutusData,
80	) -> Result<Self, String> {
81		match version {
82			0 => decode_v0_register_validator_datum(datum, appendix)
83				.ok_or("Can not parse appendix".to_string()),
84			_ => Err(format!("Unknown version: {version}")),
85		}
86	}
87}

89/// Converts [CandidateRegistration] domain type to [RegisterValidatorDatum::V0] encoded as [PlutusData].
90pub fn candidate_registration_to_plutus_data(
91	candidate_registration: &CandidateRegistration,
92) -> PlutusData {
93	RegisterValidatorDatum::V0 {
94		stake_ownership: candidate_registration.stake_ownership.clone(),
95		sidechain_pub_key: candidate_registration.partner_chain_pub_key.clone(),
96		sidechain_signature: candidate_registration.partner_chain_signature.clone(),
97		registration_utxo: candidate_registration.registration_utxo,
98		own_pkh: candidate_registration.own_pkh,
99		aura_pub_key: candidate_registration.aura_pub_key.clone(),
100		grandpa_pub_key: candidate_registration.grandpa_pub_key.clone(),
101	}
102	.into()
103}

129fn decode_v0_register_validator_datum(
130	datum: &PlutusData,
131	appendix: &PlutusData,
132) -> Option<RegisterValidatorDatum> {
133	let fields = appendix
134		.as_constr_plutus_data()
135		.filter(|datum| datum.alternative().is_zero())
136		.filter(|datum| datum.data().len() >= 6)?
137		.data();
138	let stake_ownership = decode_ada_based_staking_datum(fields.get(0))?;
139	let sidechain_pub_key = fields.get(1).as_bytes().map(SidechainPublicKey)?;
140	let sidechain_signature = fields.get(2).as_bytes().map(SidechainSignature)?;
141	let registration_utxo = decode_utxo_id_datum(fields.get(3))?;
142	let aura_pub_key = fields.get(4).as_bytes().map(AuraPublicKey)?;
143	let grandpa_pub_key = fields.get(5).as_bytes().map(GrandpaPublicKey)?;
144
145	let own_pkh = MainchainKeyHash(datum.as_bytes()?.try_into().ok()?);
146	Some(RegisterValidatorDatum::V0 {
147		stake_ownership,
148		sidechain_pub_key,
149		sidechain_signature,
150		registration_utxo,
151		own_pkh,
152		aura_pub_key,
153		grandpa_pub_key,
154	})
155}

157/// Parses plutus data schema that was used before datum versioning was added. Kept for backwards compatibility.
158fn decode_legacy_register_validator_datum(datum: &PlutusData) -> Option<RegisterValidatorDatum> {
159	let fields = datum
160		.as_constr_plutus_data()
161		.filter(|datum| datum.alternative().is_zero())
162		.filter(|datum| datum.data().len() >= 7)?
163		.data();
164	let stake_ownership = decode_ada_based_staking_datum(fields.get(0))?;
165	let sidechain_pub_key = fields.get(1).as_bytes().map(SidechainPublicKey)?;
166	let sidechain_signature = fields.get(2).as_bytes().map(SidechainSignature)?;
167	let registration_utxo = decode_utxo_id_datum(fields.get(3))?;
168	let own_pkh = MainchainKeyHash(fields.get(4).as_bytes()?.try_into().ok()?);
169	let aura_pub_key = fields.get(5).as_bytes().map(AuraPublicKey)?;
170	let grandpa_pub_key = fields.get(6).as_bytes().map(GrandpaPublicKey)?;
171	Some(RegisterValidatorDatum::V0 {
172		stake_ownership,
173		sidechain_pub_key,
174		sidechain_signature,
175		registration_utxo,
176		own_pkh,
177		aura_pub_key,
178		grandpa_pub_key,
179	})
180}

182fn decode_ada_based_staking_datum(datum: PlutusData) -> Option<AdaBasedStaking> {
183	let fields = datum
184		.as_constr_plutus_data()
185		.filter(|datum| datum.alternative().is_zero())
186		.filter(|datum| datum.data().len() >= 2)?
187		.data();
188	let pub_key = TryFrom::try_from(fields.get(0).as_bytes()?).ok()?;
189	let signature = MainchainSignature(fields.get(1).as_bytes()?.try_into().ok()?);
190	Some(AdaBasedStaking { pub_key, signature })
191}
192fn decode_utxo_id_datum(datum: PlutusData) -> Option<UtxoId> {
193	let fields = datum
194		.as_constr_plutus_data()
195		.filter(|datum| datum.alternative().is_zero())
196		.filter(|datum| datum.data().len() >= 2)?
197		.data();
198	let tx_hash = decode_tx_hash_datum(fields.get(0))?;
199	let index = UtxoIndex(fields.get(1).as_u16()?);
200	Some(UtxoId { tx_hash, index })
201}
202/// Plutus type for TxHash is a sum type, we can parse only variant with constructor 0.
203fn decode_tx_hash_datum(datum: PlutusData) -> Option<McTxHash> {
204	let constructor_datum = datum
205		.as_constr_plutus_data()
206		.filter(|datum| datum.alternative().is_zero())
207		.filter(|datum| datum.data().len() >= 1)?;
208	let bytes = constructor_datum.data().get(0).as_bytes()?;
209
210	Some(McTxHash(TryFrom::try_from(bytes).ok()?))
211}

213impl From<RegisterValidatorDatum> for PlutusData {
214	fn from(value: RegisterValidatorDatum) -> Self {
215		match value {
216			RegisterValidatorDatum::V0 {
217				stake_ownership,
218				sidechain_pub_key,
219				sidechain_signature,
220				registration_utxo,
221				own_pkh,
222				aura_pub_key,
223				grandpa_pub_key,
224			} => {
225				let mut appendix_fields = PlutusList::new();
226				appendix_fields.add(&stake_ownership_to_plutus_data(stake_ownership));
227				appendix_fields.add(&PlutusData::new_bytes(sidechain_pub_key.0));
228				appendix_fields.add(&PlutusData::new_bytes(sidechain_signature.0));
229				appendix_fields.add(&utxo_id_to_plutus_data(registration_utxo));
230				appendix_fields.add(&PlutusData::new_bytes(aura_pub_key.0));
231				appendix_fields.add(&PlutusData::new_bytes(grandpa_pub_key.0));
232				let appendix = ConstrPlutusData::new(&BigNum::zero(), &appendix_fields);
233				VersionedGenericDatum {
234					datum: PlutusData::new_bytes(own_pkh.0.to_vec()),
235					appendix: PlutusData::new_constr_plutus_data(&appendix),
236					version: 0,
237				}
238				.into()
239			},
240		}
241	}
242}

244fn stake_ownership_to_plutus_data(v: AdaBasedStaking) -> PlutusData {
245	let mut fields = PlutusList::new();
246	fields.add(&PlutusData::new_bytes(v.pub_key.0.to_vec()));
247	fields.add(&PlutusData::new_bytes(v.signature.0.to_vec()));
248	PlutusData::new_constr_plutus_data(&ConstrPlutusData::new(&BigNum::zero(), &fields))
249}
250
251fn utxo_id_to_plutus_data(v: UtxoId) -> PlutusData {
252	let mut fields = PlutusList::new();
253	fields.add(&tx_hash_to_plutus_data(v.tx_hash));
254	fields.add(&PlutusData::new_integer(&v.index.0.into()));
255	PlutusData::new_constr_plutus_data(&ConstrPlutusData::new(&BigNum::zero(), &fields))
256}
257
258fn tx_hash_to_plutus_data(v: McTxHash) -> PlutusData {
259	let mut fields = PlutusList::new();
260	fields.add(&PlutusData::new_bytes(v.0.to_vec()));
261	PlutusData::new_constr_plutus_data(&ConstrPlutusData::new(&BigNum::zero(), &fields))
262}
