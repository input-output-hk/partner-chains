1//! Pallet allowing Partner Chains to support movement of their native token from Cardano.
2//!
3//! Partner Chains Smart Contracts establish a notion of liquid and illiquid supply of the native
4//! token on Cardano, represented as native tokens being either freely available in user accounts
5//! or locked under a designated illiquid supply address. Movement of native tokens into the illiquid
6//! supply on Cardano signals that an equivalent amount of tokens should be made available on the
7//! Partner Chain.
8//!
9//! This pallet consumes inherent data containing information on the amount of native tokens newly
10//! locked on Cardano and produces an inherent extrinsic to handle their movement. The specific
11//! logic releasing the tokens is left for the Partner Chain developer to implement and is configured
12//! in the pallet via the `TokenTransferHandler`.
13//!
14//! IMPORTANT: The mechanism implemented via the pallet in this crate is only concerned with the
15//!            amount of tokens moved and does not attach any metadata to the transfers. In particular
16//!            it is not a fully-featured token bridge and needs to be combined with a separate
17//!            sender-receiver metadata channel to implement one.
18//!
19#![cfg_attr(not(feature = "std"), no_std)]
20

38/// Interface for user-provided logic to handle native token transfers into the illiquid supply on the main chain.
39///
40/// The handler will be called with **the total sum** of transfers since the previous partner chain block.
41pub trait TokenTransferHandler {
42	fn handle_token_transfer(token_amount: NativeTokenAmount) -> DispatchResult;
43}

86	#[pallet::inherent]
87	impl<T: Config> ProvideInherent for Pallet<T> {
88		type Call = Call<T>;
89		type Error = InherentError;
90		const INHERENT_IDENTIFIER: InherentIdentifier = INHERENT_IDENTIFIER;
91
92		fn create_inherent(data: &InherentData) -> Option<Self::Call> {
93			Self::get_transfered_tokens_from_inherent_data(data)
94				.filter(|data| data.token_amount.0 > 0)
95				.map(|data| Call::transfer_tokens { token_amount: data.token_amount })
96		}
97
98		fn check_inherent(call: &Self::Call, data: &InherentData) -> Result<(), Self::Error> {
99			let actual_transfer = match call {
100				Call::transfer_tokens { token_amount } => *token_amount,
101				_ => return Ok(()),
102			};
103
104			let expected_transfer = match Self::get_transfered_tokens_from_inherent_data(data) {
105				Some(data) => data.token_amount,
106				None => {
107					return Err(InherentError::UnexpectedTokenTransferInherent(actual_transfer))
108				},
109			};
110
111			if expected_transfer != actual_transfer {
112				return Err(InherentError::IncorrectTokenNumberTransfered(
113					expected_transfer,
114					actual_transfer,
115				));
116			}
117
118			Ok(())
119		}
120
121		fn is_inherent(call: &Self::Call) -> bool {
122			matches!(call, Call::transfer_tokens { .. })
123		}
124
125		fn is_inherent_required(data: &InherentData) -> Result<Option<Self::Error>, Self::Error> {
126			Ok(Self::get_transfered_tokens_from_inherent_data(data)
127				.filter(|data| data.token_amount.0 > 0)
128				.map(|data| InherentError::TokenTransferNotHandled(data.token_amount)))
129		}
130	}

132	impl<T: Config> Pallet<T> {
133		fn get_transfered_tokens_from_inherent_data(
134			data: &InherentData,
135		) -> Option<TokenTransferData> {
136			data.get_data::<TokenTransferData>(&INHERENT_IDENTIFIER)
137				.expect("Token transfer data is not encoded correctly")
138		}
139	}

141	#[pallet::call]
142	impl<T: Config> Pallet<T> {
143		#[pallet::call_index(0)]
144		#[pallet::weight((T::WeightInfo::transfer_tokens(), DispatchClass::Mandatory))]
145		pub fn transfer_tokens(
146			origin: OriginFor<T>,
147			token_amount: NativeTokenAmount,
148		) -> DispatchResult {
149			ensure_none(origin)?;
150			assert!(
151				MainChainScriptsConfiguration::<T>::exists(),
152				"BUG: Inherent should not be run unless the main chain scripts are set."
153			);
154			Initialized::<T>::mutate(|initialized| {
155				if !*initialized {
156					*initialized = true
157				}
158				true
159			});
160			T::TokenTransferHandler::handle_token_transfer(token_amount)
161		}
162
163		/// Changes the main chain scripts used for observing native token transfers.
164		///
165		/// This extrinsic must be run either using `sudo` or some other chain governance mechanism.
166		#[pallet::call_index(1)]
167		#[pallet::weight((T::WeightInfo::set_main_chain_scripts(), DispatchClass::Normal))]
168		pub fn set_main_chain_scripts(
169			origin: OriginFor<T>,
170			native_token_policy_id: PolicyId,
171			native_token_asset_name: AssetName,
172			illiquid_supply_validator_address: MainchainAddress,
173		) -> DispatchResult {
174			ensure_root(origin)?;
175			let new_scripts = sp_native_token_management::MainChainScripts {
176				native_token_policy_id,
177				native_token_asset_name,
178				illiquid_supply_validator_address,
179			};
180			MainChainScriptsConfiguration::<T>::put(new_scripts);
181			Ok(())
182		}
183	}
