13/// Mainchain block id, either a block hash or a block number
14pub enum McBlockId {
15	/// Domain type Mainchain block hash
16	McBlockHash(McBlockHash),
17	/// Domain type Mainchain block number
18	McBlockNumber(McBlockNumber),
19	/// Mainchain block hash returned as string by Blockfrost API
20	Raw(String),
21}
22
23impl From<McBlockHash> for McBlockId {
24	fn from(value: McBlockHash) -> Self {
25		McBlockId::McBlockHash(value)
26	}
27}
28
29impl From<McBlockNumber> for McBlockId {
30	fn from(value: McBlockNumber) -> Self {
31		McBlockId::McBlockNumber(value)
32	}
33}
34
35impl From<String> for McBlockId {
36	fn from(value: String) -> Self {
37		McBlockId::Raw(value)
38	}
39}
40
41impl std::fmt::Display for McBlockId {
42	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43		match self {
44			McBlockId::McBlockHash(mc_block_hash) => mc_block_hash.fmt(f),
45			McBlockId::McBlockNumber(mc_block_number) => mc_block_number.fmt(f),
46			McBlockId::Raw(str) => str.fmt(f),
47		}
48	}
49}

51#[async_trait]
52/// Mini Blockfrost API interface
53pub trait MiniBFApi {
54	/// UTXOs of the address.
55	async fn addresses_utxos(
56		&self,
57		address: MainchainAddress,
58	) -> Result<Vec<AddressUtxoContentInner>, String>;
59	/// Transactions on the address.
60	async fn addresses_transactions(
61		&self,
62		address: MainchainAddress,
63	) -> Result<Vec<AddressTransactionsContentInner>, String>;
64
65	/// List of a specific asset transactions.
66	async fn assets_transactions(
67		&self,
68		asset_id: AssetId,
69	) -> Result<Vec<AssetTransactionsInner>, String>;
70	/// List of a addresses containing a specific asset.
71	async fn assets_addresses(&self, asset_id: AssetId)
72	-> Result<Vec<AssetAddressesInner>, String>;
73
74	/// Return the latest block available to the backends, also known as the tip of the blockchain.
75	async fn blocks_latest(&self) -> Result<BlockContent, String>;
76	/// Return the content of a requested block.
77	async fn blocks_by_id(&self, id: impl Into<McBlockId> + Send) -> Result<BlockContent, String>;
78	/// Return the content of a requested block for a specific slot.
79	async fn blocks_slot(&self, slot_number: McSlotNumber) -> Result<BlockContent, String>;
80	/// Return the list of blocks following a specific block.
81	async fn blocks_next(
82		&self,
83		hash: impl Into<McBlockId> + Send,
84	) -> Result<Vec<BlockContent>, String>;
85	/// Return the transactions within the block.
86	async fn blocks_txs(&self, id: impl Into<McBlockId> + Send) -> Result<Vec<String>, String>;
87
88	/// Return the blocks minted for the epoch specified.
89	async fn epochs_blocks(&self, epoch_number: McEpochNumber) -> Result<Vec<String>, String>;
90	/// Return the protocol parameters for the epoch specified.
91	async fn epochs_parameters(
92		&self,
93		epoch_number: McEpochNumber,
94	) -> Result<EpochParamContent, String>;
95	/// Return the active stake distribution for the epoch specified by stake pool.
96	async fn epochs_stakes_by_pool(
97		&self,
98		epoch_number: McEpochNumber,
99		pool_id: &str,
100	) -> Result<Vec<EpochStakePoolContentInner>, String>;
101
102	/// History of stake pool parameters over epochs.
103	async fn pools_history(&self, pool_id: &str) -> Result<Vec<PoolHistoryInner>, String>;
104	/// List of registered stake pools with additional information.
105	async fn pools_extended(&self) -> Result<Vec<PoolListExtendedInner>, String>;
106
107	/// Query JSON value of a datum by its hash.
108	async fn scripts_datum_hash(&self, datum_hash: &str) -> Result<Vec<serde_json::Value>, String>;
109
110	/// Return content of the requested transaction.
111	async fn transaction_by_hash(&self, tx_hash: McTxHash) -> Result<TxContent, String>;
112	/// Return the inputs and UTXOs of the specific transaction.
113	async fn transactions_utxos(&self, tx_hash: McTxHash) -> Result<TxContentUtxo, String>;
114}
