19/// Values identifying on-chain entities involved in the native token management system on Cardano.
20#[derive(Default, Debug, Clone, PartialEq, Eq, TypeInfo, Encode, Decode, MaxEncodedLen)]
21#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
22pub struct MainChainScripts {
23	/// Minting policy ID of the native token
24	pub native_token_policy_id: PolicyId,
25	/// Asset name of the native token
26	pub native_token_asset_name: AssetName,
27	/// Address of the illiquid supply validator. All tokens sent to that address are effectively locked
28	/// and considered "sent" to the Partner Chain.
29	pub illiquid_supply_validator_address: MainchainAddress,
30}
31
32#[cfg(feature = "std")]
33impl MainChainScripts {
34	pub fn read_from_env() -> Result<Self, envy::Error> {
35		#[derive(serde::Serialize, serde::Deserialize)]
36		pub struct MainChainScriptsEnvConfig {
37			pub native_token_policy_id: PolicyId,
38			pub native_token_asset_name: AssetName,
39			pub illiquid_supply_validator_address: String,
40		}
41
42		let MainChainScriptsEnvConfig {
43			native_token_policy_id,
44			native_token_asset_name,
45			illiquid_supply_validator_address,
46		} = envy::from_env()?;
47		let illiquid_supply_validator_address =
48			FromStr::from_str(&illiquid_supply_validator_address).map_err(|err| {
49				envy::Error::Custom(format!("Incorrect main chain address: {}", err))
50			})?;
51		Ok(Self {
52			native_token_policy_id,
53			native_token_asset_name,
54			illiquid_supply_validator_address,
55		})
56	}
57}

133	impl NativeTokenManagementInherentDataProvider {
134		/// Creates a [NativeTokenManagementInherentDataProvider] that will provide as inherent data the aggregate number of
135		/// native token transfers into the illiquid supply on Cardano after the block with hash `parent_mc_hash` up to and
136		/// including the block with hash `mc_hash`.
137		///
138		/// This function is runtime-aware and will only create an active [NativeTokenManagementInherentDataProvider] instance
139		/// if the pallet is present and configured. Otherwise the returned inherent data provider will be inactive.
140		///
141		/// # Arguments
142		/// - `client`: runtime client exposing the [NativeTokenManagementApi] runtime API
143		/// - `data-source`: data source implementing [NativeTokenManagementDataSource]
144		/// - `mc_hash`: main chain block hash referenced by the currently produced or validated block
145		/// - `parent_mc_hash`: main chain block hash referenced by the parent of the currently producer or validated block.
146		///                     This argument should be [None] if the parent block was the genesis block or didn't reference
147		///                     any main chain block.
148		/// - `parent_hash`: block hash of the parent block of the currently produced or validated block
149		pub async fn new<Block, C>(
150			client: Arc<C>,
151			data_source: &(dyn NativeTokenManagementDataSource + Send + Sync),
152			mc_hash: McBlockHash,
153			parent_mc_hash: Option<McBlockHash>,
154			parent_hash: <Block as BlockT>::Hash,
155		) -> Result<Self, IDPCreationError>
156		where
157			Block: BlockT,
158			C: ProvideRuntimeApi<Block> + Send + Sync,
159			C::Api: NativeTokenManagementApi<Block>,
160		{
161			if client
162				.runtime_api()
163				.has_api::<dyn NativeTokenManagementApi<Block>>(parent_hash)?
164			{
165				let api = client.runtime_api();
166				let Some(scripts) = api.get_main_chain_scripts(parent_hash)? else {
167					return Ok(Self { token_amount: None });
168				};
169				let token_amount = data_source
170					.get_total_native_token_transfer(parent_mc_hash, mc_hash, scripts)
171					.await
172					.map_err(IDPCreationError::DataSourceError)?;
173
174				let token_amount = if token_amount.0 > 0 { Some(token_amount) } else { None };
175
176				Ok(Self { token_amount })
177			} else {
178				Ok(Self { token_amount: None })
179			}
180		}
181	}

183	#[async_trait::async_trait]
184	impl InherentDataProvider for NativeTokenManagementInherentDataProvider {
185		async fn provide_inherent_data(
186			&self,
187			inherent_data: &mut InherentData,
188		) -> Result<(), sp_inherents::Error> {
189			if let Some(token_amount) = self.token_amount {
190				inherent_data.put_data(INHERENT_IDENTIFIER, &TokenTransferData { token_amount })
191			} else {
192				Ok(())
193			}
194		}
195
196		async fn try_handle_error(
197			&self,
198			identifier: &InherentIdentifier,
199			mut error: &[u8],
200		) -> Option<Result<(), sp_inherents::Error>> {
201			if *identifier != INHERENT_IDENTIFIER {
202				return None;
203			}
204
205			let error = InherentError::decode(&mut error).ok()?;
206
207			Some(Err(sp_inherents::Error::Application(Box::from(error))))
208		}
209	}
