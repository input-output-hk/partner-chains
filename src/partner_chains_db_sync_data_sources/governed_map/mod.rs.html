46	async fn get_mapping_changes(
47		&self,
48		since_mc_block: Option<McBlockHash>,
49		up_to_mc_block: McBlockHash,
50		scripts: MainChainScriptsV1,
51	) -> std::result::Result<
52		Vec<(String, Option<ByteString>)>,
53		Box<dyn std::error::Error + Send + Sync>,
54	> {
55		let current_mappings = self.get_state_at_block(up_to_mc_block, scripts.clone()).await?;
56		let Some(since_mc_block) = since_mc_block else {
57			let changes =
58				current_mappings.into_iter().map(|(key, value)| (key, Some(value))).collect();
59			return Ok(changes);
60		};
61		let previous_mappings = self.get_state_at_block(since_mc_block, scripts.clone()).await?;
62		let mut changes = vec![];
63		for (key, value) in current_mappings.iter() {
64			if previous_mappings.get(key) != Some(value) {
65				changes.push((key.clone(), Some(value.clone())));
66			}
67		}
68		for key in previous_mappings.keys() {
69			if !current_mappings.contains_key(key) {
70				changes.push((key.clone(), None));
71			}
72		}
73
74		Ok(changes)
75	}
76}

79async fn get_mappings_entries(
80	pool: &PgPool,
81	hash: McBlockHash,
82	scripts: MainChainScriptsV1,
83) -> Result<BTreeMap<String, ByteString>> {
84	let Some(block) = crate::db_model::get_block_by_hash(pool, hash.clone()).await? else {
85		return Err(ExpectedDataNotFound(format!("Block hash: {hash}")));
86	};
87	let entries = crate::db_model::get_datums_at_address_with_token(
88		pool,
89		&scripts.validator_address.into(),
90		block.block_no,
91		Asset::new(scripts.asset_policy_id),
92	)
93	.await?;
94
95	let mut mappings = BTreeMap::new();
96	for entry in entries {
97		match GovernedMapDatum::try_from(entry.datum.0) {
98			Ok(GovernedMapDatum { key, value }) => {
99				mappings.insert(key, value);
100			},
101			Err(err) => warn!("Failed decoding map entry: {err}"),
102		}
103	}
104
105	Ok(mappings)
106}

129observed_async_trait!(
130impl GovernedMapDataSource for GovernedMapDataSourceCachedImpl {
131	async fn get_state_at_block(
132		&self,
133		mc_block: McBlockHash,
134		scripts: MainChainScriptsV1,
135	) -> std::result::Result<BTreeMap<String, ByteString>, Box<dyn std::error::Error + Send + Sync>>
136	{
137		Ok(get_mappings_entries(&self.pool, mc_block, scripts).await?.into())
138	}
139
140	async fn get_mapping_changes(
141		&self,
142		since_mc_block: Option<McBlockHash>,
143		up_to_mc_block: McBlockHash,
144		scripts: MainChainScriptsV1,
145	) -> std::result::Result<
146		Vec<(String, Option<ByteString>)>,
147		Box<dyn std::error::Error + Send + Sync>,
148	> {
149		if let Ok(mut cache) = self.cache.lock() {
150			cache.set_main_chain_scripts(scripts.clone());
151		};
152
153		let since_block_number = match since_mc_block {
154			Some(hash) => Some(
155				crate::db_model::get_block_by_hash(&self.pool, hash.clone())
156					.await?
157					.ok_or_else(|| Box::new(ExpectedDataNotFound(format!("Block hash: {hash}"))))?
158					.block_no,
159			),
160			None => None,
161		};
162
163		let Some(up_to_block) = self.blocks.get_block_by_hash(up_to_mc_block.clone()).await? else {
164			return Err(Box::new(ExpectedDataNotFound(format!("Block hash: {up_to_mc_block}"))));
165		};
166
167		let up_to_block_number = BlockNumber(up_to_block.number.0);
168
169		if let Some(cached_changes) =
170			self.get_changes_from_cache(since_block_number, up_to_block_number).await?
171		{
172			return Ok(cached_changes);
173		}
174
175		let latest_block_timestamp = self.blocks.get_latest_block_info().await?.timestamp;
176		let latest_stable_block =
177			match self.blocks.get_latest_stable_block_for(latest_block_timestamp.into()).await? {
178				Some(block) => BlockNumber(block.number.0),
179				None => up_to_block_number,
180			};
181		let since_block_plus =
182			BlockNumber(since_block_number.unwrap_or(BlockNumber(0)).0 + self.cache_size as u32);
183		let max_search_block = min(latest_stable_block, max(up_to_block_number, since_block_plus));
184
185		let changes = self
186			.get_changes_in_range_to_cache(since_block_number, max_search_block, scripts)
187			.await?;
188
189		if let Ok(mut cache) = self.cache.lock() {
190			cache.update(changes.clone());
191		}
192
193		Ok(filter_changes_in_range(changes, since_block_number, up_to_block_number))
194	}
195}
196);

198impl GovernedMapDataSourceCachedImpl {
199	async fn get_changes_from_cache(
200		&self,
201		since_block: Option<BlockNumber>,
202		up_to_block: BlockNumber,
203	) -> Result<Option<Vec<(String, Option<ByteString>)>>> {
204		if let Ok(cache) = self.cache.lock() {
205			Ok(cache.get_changes_in_range(since_block, up_to_block))
206		} else {
207			Ok(None)
208		}
209	}
210
211	async fn get_changes_in_range_to_cache(
212		&self,
213		since_block: Option<BlockNumber>,
214		up_to_block: BlockNumber,
215		scripts: MainChainScriptsV1,
216	) -> Result<Vec<Change>> {
217		let changes = crate::db_model::get_changes(
218			&self.pool,
219			&scripts.validator_address.into(),
220			since_block,
221			up_to_block,
222			Asset::new(scripts.asset_policy_id),
223		)
224		.await?;
225
226		let mut result = Vec::new();
227
228		for change in changes {
229			let GovernedMapDatum { key, value } = match GovernedMapDatum::try_from(change.datum.0) {
230				Ok(datum) => datum,
231				Err(err) => {
232					warn!("Failed decoding map entry: {err}");
233					continue;
234				},
235			};
236			match change.action {
237				GovernedMapAction::Spend => result.push(Change::new(change.block_no, key, None)),
238				GovernedMapAction::Create => {
239					result.push(Change::new(change.block_no, key, Some(value)))
240				},
241			};
242		}
243		Ok(result)
244	}
245}

247#[derive(derive_new::new, Clone)]
248struct Change {
249	block_no: BlockNumber,
250	key: String,
251	value: Option<ByteString>,
252}
253
254#[derive(Default)]
255pub(crate) struct Cache {
256	highest_block_number: Option<BlockNumber>,
257	lowest_block_number: Option<BlockNumber>,
258	changes: Vec<Change>,
259	address: Option<MainchainAddress>,
260	policy_id: Option<PolicyId>,
261}
262
263fn filter_changes_in_range(
264	changes: Vec<Change>,
265	since_block: Option<BlockNumber>,
266	up_to_block: BlockNumber,
267) -> Vec<(String, Option<ByteString>)> {
268	changes
269		.into_iter()
270		.filter(|change| {
271			change.block_no.0 <= up_to_block.0
272				&& since_block.map(|b| change.block_no.0 > b.0).unwrap_or(true)
273		})
274		.map(|change| (change.key, change.value))
275		.collect()
276}
277
278impl Cache {
279	fn get_changes_in_range(
280		&self,
281		since_block: Option<BlockNumber>,
282		up_to_block: BlockNumber,
283	) -> Option<Vec<(String, Option<ByteString>)>> {
284		let Some(highest_block_number) = self.highest_block_number else {
285			return None;
286		};
287		let Some(lowest_block_number) = self.lowest_block_number else {
288			return None;
289		};
290
291		if highest_block_number.0 < up_to_block.0
292			|| since_block.map(|b| b.0 < lowest_block_number.0).unwrap_or(false)
293		{
294			return None;
295		}
296
297		Some(filter_changes_in_range(self.changes.clone(), since_block, up_to_block))
298	}
299
300	fn update(&mut self, changes: Vec<Change>) {
301		self.changes = changes;
302		let (lowest_block_number, highest_block_number) = self
303			.changes
304			.iter()
305			.minmax_by_key(|change| change.block_no.0)
306			.into_option()
307			.map(|(min, max)| (min.block_no, max.block_no))
308			.unwrap_or((BlockNumber(0), BlockNumber(0)));
309		self.lowest_block_number = Some(lowest_block_number);
310		self.highest_block_number = Some(highest_block_number);
311	}
312
313	fn set_main_chain_scripts(&mut self, scripts: MainChainScriptsV1) {
314		if self.address != Some(scripts.validator_address.clone())
315			|| self.policy_id != Some(scripts.asset_policy_id.clone())
316		{
317			self.changes.clear();
318			self.highest_block_number = None;
319			self.lowest_block_number = None;
320			self.address = Some(scripts.validator_address);
321			self.policy_id = Some(scripts.asset_policy_id);
322		}
323	}
