18/// Db-Sync `tx_in.value` configuration field
19#[derive(Debug, PartialEq, Copy, Clone)]
20pub(crate) enum TxInConfiguration {
21	/// Transaction inputs are linked using `tx_in` table
22	Enabled,
23	/// Transaction inputs are linked using `consumed_by_tx_id` column in `tx_out` table
24	Consumed,
25}
26
27impl TxInConfiguration {
28	pub(crate) async fn from_connection(pool: &Pool<Postgres>) -> Result<Self, SqlxError> {
29		let tx_in_exists = sqlx::query_scalar::<_, i64>(
30			"select count(*) from information_schema.tables where table_name = 'tx_in';",
31		)
32		.fetch_one(pool)
33		.await? == 1;
34
35		if !tx_in_exists {
36			return Ok(Self::Consumed);
37		}
38
39		let tx_in_populated = sqlx::query_scalar::<_, bool>("SELECT EXISTS (SELECT 1 FROM tx_in);")
40			.fetch_one(pool)
41			.await?;
42
43		if tx_in_populated {
44			return Ok(Self::Enabled);
45		}
46
47		Ok(Self::Consumed)
48	}
49}

51/// Structure that queries, caches and provides Db-Sync configuration
52pub struct DbSyncConfigurationProvider {
53	/// Postgres connection pool
54	pub(crate) pool: PgPool,
55	/// Transaction input configuration used by Db-Sync
56	pub(crate) tx_in_config: Arc<Mutex<OnceCell<TxInConfiguration>>>,
57}
58
59impl DbSyncConfigurationProvider {
60	pub(crate) fn new(pool: PgPool) -> Self {
61		Self { tx_in_config: Arc::new(Mutex::new(OnceCell::new())), pool }
62	}
63
64	pub(crate) async fn get_tx_in_config(
65		&self,
66	) -> std::result::Result<TxInConfiguration, DataSourceError> {
67		let lock = self.tx_in_config.lock().await;
68		if let Some(tx_in_config) = lock.get() {
69			return Ok(*tx_in_config);
70		} else {
71			let tx_in_config = TxInConfiguration::from_connection(&self.pool).await?;
72			lock.set(tx_in_config).map_err(|_| {
73				DataSourceError::InternalDataSourceError(
74					"Failed to set tx_in_config in DbSyncConfigurationProvider".into(),
75				)
76			})?;
77			log::info!("Using configuration: {tx_in_config:?}");
78			return Ok(tx_in_config);
79		}
80	}
81}

306/// Query to get the highest block in the given time range, not higher than a given block
307/// # Arguments
308/// * `max_block_number` - block number upper boundary
309/// * `min_time` - time lower boundary
310/// * `min_slot` - slot lower boundary, for efficiency, because time is not indexed
311/// * `max_time` - time upper boundary
312/// * `max_slot` - slot upper boundary, for efficiency, because time is not indexed
313#[cfg(feature = "block-source")]
314pub(crate) async fn get_highest_block(
315	pool: &Pool<Postgres>,
316	max_block_number: BlockNumber,
317	min_time: NaiveDateTime,
318	min_slot: SlotNumber,
319	max_time: NaiveDateTime,
320	max_slot: SlotNumber,
321) -> Result<Option<Block>, SqlxError> {
322	Ok(sqlx::query_as::<_, Block>(
323		"
324SELECT block_no, hash, epoch_no, slot_no, time
325FROM block
326WHERE block_no <= $1 AND $2 <= slot_no AND slot_no <= $3 AND $4 <= time AND time <= $5
327ORDER BY block_no DESC
328LIMIT 1",
329	)
330	.bind(max_block_number)
331	.bind(min_slot)
332	.bind(max_slot)
333	.bind(min_time)
334	.bind(max_time)
335	.fetch_optional(pool)
336	.await?)

369/// Returns number of the latest epoch that is stable - no block in such an epoch can be rolled back.
370/// The latest stable epoch is one less than epoch of the highest stable block (HSB),
371/// because unstable part could be replaced with blocks sequence starting from the block that has
372/// the same epoch as HSB.
373pub(crate) async fn get_latest_stable_epoch(
374	pool: &Pool<Postgres>,
375	security_parameter: u32,
376) -> Result<Option<EpochNumber>, SqlxError> {
377	let sql = "SELECT stable_block.epoch_no - 1 as epoch_no
378FROM block INNER JOIN block as stable_block ON block.block_no - $1 = stable_block.block_no
379WHERE block.block_no IS NOT NULL
380ORDER BY block.block_no DESC
381LIMIT 1";
382	#[allow(deprecated)]
383	Ok(sqlx::query_as::<_, EpochNumberRow>(sql)
384		.bind(BlockNumber(security_parameter))
385		.fetch_optional(pool)
386		.await?
387		.map(EpochNumber::from))
388}

404/// Returns the token data of the given policy at the given slot.
405#[cfg(feature = "candidate-source")]
406pub(crate) async fn get_token_utxo_for_epoch(
407	pool: &Pool<Postgres>,
408	asset: &Asset,
409	epoch: EpochNumber,
410) -> Result<Option<TokenTxOutput>, SqlxError> {
411	// In practice queried assets always have empty name.
412	// However, it's important to keep multi_asset.name condition, to enable use of compound index on multi_asset policy and name.
413	let sql = "SELECT
414			origin_tx.hash        AS origin_tx_hash,
415        	tx_out.index          AS utxo_index,
416        	origin_block.epoch_no AS tx_epoch_no,
417        	origin_block.block_no AS tx_block_no,
418        	origin_block.slot_no  AS tx_slot_no,
419        	origin_tx.block_index AS tx_block_index,
420        	datum.value           AS datum
421        FROM ma_tx_out
422        INNER JOIN multi_asset          ON ma_tx_out.ident = multi_asset.id
423        INNER JOIN tx_out               ON ma_tx_out.tx_out_id = tx_out.id
424        INNER JOIN tx origin_tx         ON tx_out.tx_id = origin_tx.id
425        INNER JOIN block origin_block   ON origin_tx.block_id = origin_block.id
426        LEFT JOIN datum                 ON tx_out.data_hash = datum.hash
427        WHERE multi_asset.policy = $1
428		AND multi_asset.name = $2
429        AND origin_block.epoch_no <= $3
430        ORDER BY tx_block_no DESC, origin_tx.block_index DESC
431        LIMIT 1";
432	Ok(sqlx::query_as::<_, TokenTxOutput>(sql)
433		.bind(&asset.policy_id.0)
434		.bind(&asset.asset_name.0)
435		.bind(epoch)
436		.fetch_optional(pool)
437		.await?)
438}

462pub(crate) async fn get_governed_map_changes_tx_in_enabled(
463	pool: &Pool<Postgres>,
464	address: &Address,
465	after_block: Option<BlockNumber>,
466	to_block: BlockNumber,
467	asset: Asset,
468) -> Result<Vec<DatumChangeOutput>, SqlxError> {
469	let query = "
470		((SELECT
471			datum.value as datum, origin_block.block_no as block_no, origin_tx.block_index as block_index, $6 as action, 1 as action_order
472		FROM tx_out
473		INNER JOIN tx origin_tx			ON tx_out.tx_id = origin_tx.id
474		INNER JOIN block origin_block	ON origin_tx.block_id = origin_block.id
475		INNER JOIN datum				ON tx_out.data_hash = datum.hash
476		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
477		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
478		WHERE
479			tx_out.address = $1 AND ($2 IS NULL OR origin_block.block_no > $2) AND origin_block.block_no <= $3
480			AND multi_asset.policy = $4
481			AND multi_asset.name = $5)
482		UNION
483		(SELECT
484			datum.value as datum, consuming_block.block_no as block_no, consuming_tx.block_index as block_index, $7 as action, -1 as action_order
485		FROM tx_out
486		LEFT JOIN tx_in consuming_tx_in	ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
487		LEFT JOIN tx consuming_tx		ON consuming_tx_in.tx_in_id = consuming_tx.id
488		LEFT JOIN block consuming_block	ON consuming_tx.block_id = consuming_block.id
489		INNER JOIN datum				ON tx_out.data_hash = datum.hash
490		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
491		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
492		WHERE
493			tx_out.address = $1
494			AND (consuming_tx_in.id IS NOT NULL AND ($2 IS NULL OR consuming_block.block_no > $2) AND consuming_block.block_no <= $3)
495			AND multi_asset.policy = $4
496			AND multi_asset.name = $5))
497		ORDER BY block_no, block_index, action_order ASC";
498	Ok(sqlx::query_as::<_, DatumChangeOutput>(query)
499		.bind(&address.0)
500		.bind(after_block)
501		.bind(to_block)
502		.bind(&asset.policy_id.0)
503		.bind(&asset.asset_name.0)
504		.bind(GovernedMapAction::Create)
505		.bind(GovernedMapAction::Spend)
506		.fetch_all(pool)
507		.await?)
508}

918#[cfg(feature = "bridge")]
919pub(crate) async fn get_bridge_utxos_tx(
920	tx_in_configuration: TxInConfiguration,
921	pool: &Pool<Postgres>,
922	icp_address: &Address,
923	native_token: Asset,
924	checkpoint: ResolvedBridgeDataCheckpoint,
925	to_block: BlockNumber,
926	max_utxos: Option<u32>,
927) -> Result<Vec<BridgeUtxo>, SqlxError> {
928	use sqlx::QueryBuilder;
929
930	let mut query_builder = QueryBuilder::new(
931		"
932	SELECT
933	      block.block_no                          AS block_number
934	    , tx.block_index                          AS tx_ix
935	    , tx.hash                                 AS tx_hash
936	    , outputs.index                           AS utxo_ix
937	    , output_tokens.quantity                  AS tokens_out
938	    , coalesce(sum(input_tokens.quantity), 0) AS tokens_in
939	    , datum.value                             AS datum
940
941	FROM tx_out      outputs
942	JOIN tx                        ON outputs.tx_id = tx.id
943	JOIN block                     ON tx.block_id = block.id
944	JOIN ma_tx_out   output_tokens ON output_tokens.tx_out_id = outputs.id
945	JOIN multi_asset native_token  ON native_token.id = output_tokens.ident
946	LEFT JOIN datum                     ON datum.hash = outputs.data_hash
947",
948	);
949
950	query_builder.push(match tx_in_configuration {
951		TxInConfiguration::Consumed =>
952				"LEFT JOIN tx_out     inputs        ON inputs.consumed_by_tx_id = tx.id   AND inputs.address = $1",
953		TxInConfiguration::Enabled =>
954				"LEFT JOIN tx_in      inputs_join   ON tx.id = inputs_join.tx_in_id
955                 LEFT JOIN tx_out     inputs        ON inputs_join.tx_out_id = inputs.tx_id and inputs_join.tx_out_index = inputs.index AND inputs.address = $1",
956	});
957
958	query_builder.push(
959		"
960	LEFT JOIN ma_tx_out  input_tokens  ON input_tokens.tx_out_id = inputs.id AND input_tokens.ident = native_token.id
961	WHERE native_token.policy = $2
962	  AND native_token.name = $3
963	  AND outputs.address = $1
964	  AND block_no <= $4
965",
966	);
967
968	match checkpoint {
969		ResolvedBridgeDataCheckpoint::Block { number } => {
970			query_builder.push(&format!("AND block_no > {} ", number.0));
971		},
972		ResolvedBridgeDataCheckpoint::Utxo { block_number, tx_ix, tx_out_ix } => {
973			query_builder.push(&format!(
974				"AND (block_no, tx.block_index, outputs.index) > ({}, {}, {}) ",
975				block_number.0, tx_ix.0, tx_out_ix.0
976			));
977		},
978	}
979
980	query_builder
981		.push("GROUP BY tx.hash, outputs.id, output_tokens.quantity, datum.value, block.block_no, tx.block_index, outputs.index ")
982		.push("ORDER BY block.block_no, tx.block_index, outputs.index ");
983
984	if let Some(max_utxos) = max_utxos {
985		query_builder.push(&format!("LIMIT {max_utxos}"));
986	}
987	query_builder.push(";");
988
989	let query = query_builder
990		.build_query_as::<BridgeUtxo>()
991		.bind(&icp_address.0)
992		.bind(&native_token.policy_id.0)
993		.bind(&native_token.asset_name.0)
994		.bind(to_block);
995
996	Ok(query.fetch_all(pool).await?)
