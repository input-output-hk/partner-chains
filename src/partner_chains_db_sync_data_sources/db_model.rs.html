49impl TryFrom<MainchainTxOutputRow> for MainchainTxOutput {
50	type Error = sqlx::Error;
51	fn try_from(r: MainchainTxOutputRow) -> Result<Self, Self::Error> {
52		let tx_inputs: Result<Vec<UtxoId>, _> =
53			r.tx_inputs.into_iter().map(|i| UtxoId::from_str(i.as_str())).collect();
54		let tx_inputs = tx_inputs.map_err(|e| sqlx::Error::Decode(e.into()))?;
55		Ok(MainchainTxOutput {
56			utxo_id: UtxoId {
57				tx_hash: McTxHash(r.utxo_id_tx_hash),
58				index: UtxoIndex(r.utxo_id_index.0),
59			},
60			tx_block_no: r.tx_block_no,
61			tx_slot_no: r.tx_slot_no,
62			tx_epoch_no: r.tx_epoch_no,
63			tx_index_in_block: r.tx_index_in_block,
64			address: r.address,
65			datum: r.datum.map(|d| d.0),
66			tx_inputs,
67		})
68	}
69}

184	stake_pool_hashes: Vec<[u8; 28]>,

186	Ok(sqlx::query_as::<_, StakePoolDelegationOutputRow>(
187		"
188SELECT
189	epoch_stake.amount AS epoch_stake_amount,
190	pool_hash.hash_raw AS pool_hash_raw,
191	stake_address.hash_raw AS stake_address_hash_raw,
192	stake_address.script_hash AS stake_address_script_hash

196	INNER JOIN pool_hash          ON epoch_stake.pool_id = pool_hash.id

198	    epoch_stake.epoch_no = $1
199	AND epoch_stake.amount > 0
200	AND pool_hash.hash_raw IN (SELECT unnest($2))
201    ",
202	)
203	.bind(epoch)
204	.bind(stake_pool_hashes)
205	.fetch_all(pool)
206	.await?)

255/// Query to get the highest block in the given time range, not higher than a given block
256/// # Arguments
257/// * `max_block_number` - block number upper boundary
258/// * `min_time` - time lower boundary
259/// * `min_slot` - slot lower boundary, for efficiency, because time is not indexed
260/// * `max_time` - time upper boundary
261/// * `max_slot` - slot upper boundary, for efficiency, because time is not indexed
262#[cfg(feature = "block-source")]
263pub(crate) async fn get_highest_block(
264	pool: &Pool<Postgres>,
265	max_block_number: BlockNumber,
266	min_time: NaiveDateTime,
267	min_slot: SlotNumber,
268	max_time: NaiveDateTime,
269	max_slot: SlotNumber,
270) -> Result<Option<Block>, SqlxError> {
271	Ok(sqlx::query_as::<_, Block>(
272		"
273SELECT block_no, hash, epoch_no, slot_no, time
274FROM block
275WHERE block_no <= $1 AND $2 <= slot_no AND slot_no <= $3 AND $4 <= time AND time <= $5
276ORDER BY block_no DESC
277LIMIT 1",
278	)
279	.bind(max_block_number)
280	.bind(min_slot)
281	.bind(max_slot)
282	.bind(min_time)
283	.bind(max_time)
284	.fetch_optional(pool)
285	.await?)
286}

318/// Returns number of the latest epoch that is stable - no block in such an epoch can be rolled back.
319/// The latest stable epoch is one less than epoch of the highest stable block (HSB),
320/// because unstable part could be replaced with blocks sequence starting from the block that has
321/// the same epoch as HSB.
322pub(crate) async fn get_latest_stable_epoch(
323	pool: &Pool<Postgres>,
324	security_parameter: u32,
325) -> Result<Option<EpochNumber>, SqlxError> {
326	let sql = "SELECT stable_block.epoch_no - 1 as epoch_no
327FROM block INNER JOIN block as stable_block ON block.block_no - $1 = stable_block.block_no
328WHERE block.block_no IS NOT NULL
329ORDER BY block.block_no DESC
330LIMIT 1";
331	#[allow(deprecated)]
332	Ok(sqlx::query_as::<_, EpochNumberRow>(sql)
333		.bind(BlockNumber(security_parameter))
334		.fetch_optional(pool)
335		.await?
336		.map(EpochNumber::from))
337}

353/// Returns the token data of the given policy at the given slot.
354#[cfg(feature = "candidate-source")]
355pub(crate) async fn get_token_utxo_for_epoch(
356	pool: &Pool<Postgres>,
357	asset: &Asset,
358	epoch: EpochNumber,
359) -> Result<Option<TokenTxOutput>, SqlxError> {
360	// In practice queried assets always have empty name.
361	// However, it's important to keep multi_asset.name condition, to enable use of compound index on multi_asset policy and name.
362	let sql = "SELECT
363			origin_tx.hash        AS origin_tx_hash,
364        	tx_out.index          AS utxo_index,
365        	origin_block.epoch_no AS tx_epoch_no,
366        	origin_block.block_no AS tx_block_no,
367        	origin_block.slot_no  AS tx_slot_no,
368        	origin_tx.block_index AS tx_block_index,
369        	datum.value           AS datum
370        FROM ma_tx_out
371        INNER JOIN multi_asset          ON ma_tx_out.ident = multi_asset.id
372        INNER JOIN tx_out               ON ma_tx_out.tx_out_id = tx_out.id
373        INNER JOIN tx origin_tx         ON tx_out.tx_id = origin_tx.id
374        INNER JOIN block origin_block   ON origin_tx.block_id = origin_block.id
375        LEFT JOIN datum                 ON tx_out.data_hash = datum.hash
376        WHERE multi_asset.policy = $1
377		AND multi_asset.name = $2
378        AND origin_block.epoch_no <= $3
379        ORDER BY tx_block_no DESC, origin_tx.block_index DESC
380        LIMIT 1";
381	Ok(sqlx::query_as::<_, TokenTxOutput>(sql)
382		.bind(&asset.policy_id.0)
383		.bind(&asset.asset_name.0)
384		.bind(epoch)
385		.fetch_optional(pool)
386		.await?)
387}

390pub(crate) async fn get_changes(
391	pool: &Pool<Postgres>,
392	address: &Address,
393	after_block: Option<BlockNumber>,
394	to_block: BlockNumber,
395	asset: Asset,
396) -> Result<Vec<DatumChangeOutput>, SqlxError> {
397	let query = "
398		((SELECT
399			datum.value as datum, origin_block.block_no as block_no, origin_tx.block_index as block_index, $6 as action, 1 as action_order
400		FROM tx_out
401		INNER JOIN tx origin_tx			ON tx_out.tx_id = origin_tx.id
402		INNER JOIN block origin_block	ON origin_tx.block_id = origin_block.id
403		INNER JOIN datum				ON tx_out.data_hash = datum.hash
404		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
405		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
406		WHERE
407			tx_out.address = $1 AND ($2 IS NULL OR origin_block.block_no > $2) AND origin_block.block_no <= $3
408			AND multi_asset.policy = $4
409			AND multi_asset.name = $5)
410		UNION
411		(SELECT
412			datum.value as datum, consuming_block.block_no as block_no, consuming_tx.block_index as block_index, $7 as action, -1 as action_order
413		FROM tx_out
414		LEFT JOIN tx_in consuming_tx_in	ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
415		LEFT JOIN tx consuming_tx		ON consuming_tx_in.tx_in_id = consuming_tx.id
416		LEFT JOIN block consuming_block	ON consuming_tx.block_id = consuming_block.id
417		INNER JOIN datum				ON tx_out.data_hash = datum.hash
418		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
419		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
420		WHERE
421			tx_out.address = $1
422			AND (consuming_tx_in.id IS NOT NULL AND ($2 IS NULL OR consuming_block.block_no > $2) AND consuming_block.block_no <= $3)
423			AND multi_asset.policy = $4
424			AND multi_asset.name = $5))
425		ORDER BY block_no, block_index, action_order ASC";
426	Ok(sqlx::query_as::<_, DatumChangeOutput>(query)
427		.bind(&address.0)
428		.bind(after_block)
429		.bind(to_block)
430		.bind(&asset.policy_id.0)
431		.bind(&asset.asset_name.0)
432		.bind(GovernedMapAction::Create)
433		.bind(GovernedMapAction::Spend)
434		.fetch_all(pool)
435		.await?)
436}

483#[cfg(feature = "candidate-source")]
484pub(crate) async fn get_utxos_for_address(
485	pool: &Pool<Postgres>,
486	address: &Address,
487	block: BlockNumber,
488) -> Result<Vec<MainchainTxOutput>, SqlxError> {
489	let query = "SELECT
490          		origin_tx.hash as utxo_id_tx_hash,
491          		tx_out.index as utxo_id_index,
492          		origin_block.block_no as tx_block_no,
493          		origin_block.slot_no as tx_slot_no,
494          		origin_block.epoch_no as tx_epoch_no,
495          		origin_tx.block_index as tx_index_in_block,
496          		tx_out.address,
497          		datum.value as datum,
498          		array_agg(concat_ws('#', encode(consumes_tx.hash, 'hex'), consumes_tx_in.tx_out_index)) as tx_inputs
499			FROM tx_out
500			INNER JOIN tx    origin_tx       ON tx_out.tx_id = origin_tx.id
501			INNER JOIN block origin_block    ON origin_tx.block_id = origin_block.id
502          	LEFT JOIN tx_in consuming_tx_in  ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
503          	LEFT JOIN tx    consuming_tx     ON consuming_tx_in.tx_in_id = consuming_tx.id
504          	LEFT JOIN block consuming_block  ON consuming_tx.block_id = consuming_block.id
505			LEFT JOIN tx_in consumes_tx_in   ON consumes_tx_in.tx_in_id = origin_tx.id
506          	LEFT JOIN tx_out consumes_tx_out ON consumes_tx_out.tx_id = consumes_tx_in.tx_out_id AND consumes_tx_in.tx_out_index = consumes_tx_out.index
507          	LEFT JOIN tx consumes_tx         ON consumes_tx.id = consumes_tx_out.tx_id
508          	LEFT JOIN datum                  ON tx_out.data_hash = datum.hash
509          	WHERE
510          		tx_out.address = $1 AND origin_block.block_no <= $2
511          		AND (consuming_tx_in.id IS NULL OR consuming_block.block_no > $2)
512          		GROUP BY (
513					utxo_id_tx_hash,
514					utxo_id_index,
515					tx_block_no,
516					tx_slot_no,
517					tx_epoch_no,
518					tx_index_in_block,
519					tx_out.address,
520					datum
521				)";
522	let rows = sqlx::query_as::<_, MainchainTxOutputRow>(query)
523		.bind(&address.0)
524		.bind(block)
525		.fetch_all(pool)
526		.await?;
527	let result: Result<Vec<MainchainTxOutput>, sqlx::Error> =
528		rows.into_iter().map(MainchainTxOutput::try_from).collect();
529	Ok(result?)
530}

548pub(crate) async fn create_idx_tx_out_address(pool: &Pool<Postgres>) -> Result<(), SqlxError> {
549	let exists = index_exists(pool, "idx_tx_out_address").await?;
550	if exists {
551		info!("Index 'idx_tx_out_address' already exists");
552	} else {
553		let sql = "CREATE INDEX IF NOT EXISTS idx_tx_out_address ON tx_out USING hash (address)";
554		info!("Executing '{}', this might take a long time", sql);
555		sqlx::query(sql).execute(pool).await?;
556		info!("Index 'idx_tx_out_address' has been created");
557	}
558	Ok(())
559}

561/// Check if the index exists.
562async fn index_exists(pool: &Pool<Postgres>, index_name: &str) -> Result<bool, sqlx::Error> {
563	sqlx::query("select * from pg_indexes where indexname = $1")
564		.bind(index_name)
565		.fetch_all(pool)
566		.await
567		.map(|rows| rows.len() == 1)
568}

576/// Sums all transfers between genesis and the first block that is produced with the feature on.
577/// Used by `get_native_token_transfers` (NativeTokenDataSourceImpl).
578#[cfg(feature = "native-token")]
579pub(crate) async fn get_total_native_tokens_transfered(
580	pool: &Pool<Postgres>,
581	to_block: BlockNumber,
582	asset: Asset,
583	illiquid_supply_address: Address,
584) -> Result<NativeTokenAmount, SqlxError> {
585	let query = sqlx::query_as::<_, (NativeTokenAmount,)>(
586		"
587SELECT
588    COALESCE(SUM(ma_tx_out.quantity), 0)
589FROM tx_out
590LEFT JOIN ma_tx_out    ON ma_tx_out.tx_out_id = tx_out.id
591LEFT JOIN multi_asset  ON multi_asset.id = ma_tx_out.ident
592INNER JOIN tx          ON tx_out.tx_id = tx.id
593INNER JOIN block       ON tx.block_id = block.id
594WHERE address = $1
595AND multi_asset.policy = $2
596AND multi_asset.name = $3
597AND block.block_no <= $4;
598    ",
599	)
600	.bind(&illiquid_supply_address.0)
601	.bind(&asset.policy_id.0)
602	.bind(&asset.asset_name.0)
603	.bind(to_block);
604
605	Ok(query.fetch_one(pool).await?.0)
606}

608/// Returns the list of all blocks in given range,
609/// together with the sum of all token amounts transferred to illiquid supply address in this block.
610/// On devnet postgres it takes around 5ms to execute when querying 1000 blocks.
611/// Used by `get_native_token_transfers` (NativeTokenDataSourceImpl).
612#[cfg(feature = "native-token")]
613pub(crate) async fn get_native_token_transfers(
614	pool: &Pool<Postgres>,
615	from_block: BlockNumber,
616	to_block: BlockNumber,
617	asset: Asset,
618	illiquid_supply_address: Address,
619) -> Result<Vec<BlockTokenAmount>, SqlxError> {
620	let query = sqlx::query_as::<_, BlockTokenAmount>(
621		"
622SELECT
623    block.hash as block_hash, COALESCE(SUM(ma_tx_out.quantity), 0) as amount
624FROM block
625    LEFT JOIN tx ON block.id = tx.block_id
626    LEFT JOIN tx_out ON tx.id = tx_out.tx_id AND tx_out.address = $1
627    LEFT JOIN ma_tx_out    ON ma_tx_out.tx_out_id = tx_out.id
628    LEFT JOIN multi_asset  ON multi_asset.id = ma_tx_out.ident AND multi_asset.policy = $2 AND multi_asset.name = $3
629WHERE
630    $4 <= block.block_no AND block.block_no <= $5
631GROUP BY block.block_no, block.hash
632ORDER BY block.block_no ASC;
633    ",
634	)
635	.bind(&illiquid_supply_address.0)
636	.bind(&asset.policy_id.0)
637	.bind(&asset.asset_name.0)
638	.bind(from_block)
639	.bind(to_block);
640
641	Ok(query.fetch_all(pool).await?)
642}
