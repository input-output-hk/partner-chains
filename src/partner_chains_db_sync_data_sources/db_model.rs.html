18/// Db-Sync `tx_in.value` configuration field
19#[derive(Debug, PartialEq, Copy, Clone)]
20pub(crate) enum TxInConfiguration {
21	/// Transaction inputs are linked using `tx_in` table
22	Enabled,
23	/// Transaction inputs are linked using `consumed_by_tx_id` column in `tx_out` table
24	Consumed,
25}
26
27impl TxInConfiguration {
28	pub(crate) async fn from_connection(pool: &Pool<Postgres>) -> Result<Self, SqlxError> {
29		let tx_in_exists = sqlx::query_scalar::<_, i64>(
30			"select count(*) from information_schema.tables where table_name = 'tx_in';",
31		)
32		.fetch_one(pool)
33		.await? == 1;
34
35		if !tx_in_exists {
36			return Ok(Self::Consumed);
37		}
38
39		let tx_in_populated = sqlx::query_scalar::<_, bool>("SELECT EXISTS (SELECT 1 FROM tx_in);")
40			.fetch_one(pool)
41			.await?;
42
43		if tx_in_populated {
44			return Ok(Self::Enabled);
45		}
46
47		Ok(Self::Consumed)
48	}
49}

51/// Structure that queries, caches and provides Db-Sync configuration
52pub struct DbSyncConfigurationProvider {
53	/// Postgres connection pool
54	pub(crate) pool: PgPool,
55	/// Transaction input configuration used by Db-Sync
56	pub(crate) tx_in_config: Arc<Mutex<OnceCell<TxInConfiguration>>>,
57}
58
59impl DbSyncConfigurationProvider {
60	pub(crate) fn new(pool: PgPool) -> Self {
61		Self { tx_in_config: Arc::new(Mutex::new(OnceCell::new())), pool }
62	}
63
64	pub(crate) async fn get_tx_in_config(
65		&self,
66	) -> std::result::Result<TxInConfiguration, DataSourceError> {
67		let lock = self.tx_in_config.lock().await;
68		if let Some(tx_in_config) = lock.get() {
69			return Ok(*tx_in_config);
70		} else {
71			let tx_in_config = TxInConfiguration::from_connection(&self.pool).await?;
72			lock.set(tx_in_config).map_err(|_| {
73				DataSourceError::InternalDataSourceError(
74					"Failed to set tx_in_config in DbSyncConfigurationProvider".into(),
75				)
76			})?;
77			log::info!("Using configuration: {tx_in_config:?}");
78			return Ok(tx_in_config);
79		}
80	}
81}

257	AND pool_hash.hash_raw IN (SELECT unnest($2))
258    ",
259	)
260	.bind(epoch)
261	.bind(stake_pool_hashes)
262	.fetch_all(pool)
263	.await?)

312/// Query to get the highest block in the given time range, not higher than a given block
313/// # Arguments
314/// * `max_block_number` - block number upper boundary
315/// * `min_time` - time lower boundary
316/// * `min_slot` - slot lower boundary, for efficiency, because time is not indexed
317/// * `max_time` - time upper boundary
318/// * `max_slot` - slot upper boundary, for efficiency, because time is not indexed
319#[cfg(feature = "block-source")]
320pub(crate) async fn get_highest_block(
321	pool: &Pool<Postgres>,
322	max_block_number: BlockNumber,
323	min_time: NaiveDateTime,
324	min_slot: SlotNumber,
325	max_time: NaiveDateTime,
326	max_slot: SlotNumber,
327) -> Result<Option<Block>, SqlxError> {
328	Ok(sqlx::query_as::<_, Block>(
329		"
330SELECT block_no, hash, epoch_no, slot_no, time
331FROM block
332WHERE block_no <= $1 AND $2 <= slot_no AND slot_no <= $3 AND $4 <= time AND time <= $5
333ORDER BY block_no DESC
334LIMIT 1",
335	)
336	.bind(max_block_number)
337	.bind(min_slot)
338	.bind(max_slot)
339	.bind(min_time)
340	.bind(max_time)
341	.fetch_optional(pool)
342	.await?)
343}

374#[cfg(feature = "candidate-source")]
375/// Returns number of the latest epoch that is stable - no block in such an epoch can be rolled back.
376/// The latest stable epoch is one less than epoch of the highest stable block (HSB),
377/// because unstable part could be replaced with blocks sequence starting from the block that has
378/// the same epoch as HSB.
379pub(crate) async fn get_latest_stable_epoch(
380	pool: &Pool<Postgres>,
381	security_parameter: u32,
382) -> Result<Option<EpochNumber>, SqlxError> {
383	let sql = "SELECT stable_block.epoch_no - 1 as epoch_no
384FROM block INNER JOIN block as stable_block ON block.block_no - $1 = stable_block.block_no
385WHERE block.block_no IS NOT NULL
386ORDER BY block.block_no DESC
387LIMIT 1";
388	#[allow(deprecated)]
389	Ok(sqlx::query_as::<_, EpochNumberRow>(sql)
390		.bind(BlockNumber(security_parameter))
391		.fetch_optional(pool)
392		.await?
393		.map(EpochNumber::from))
394}

410/// Returns the token data of the given policy at the given slot.
411#[cfg(feature = "candidate-source")]
412pub(crate) async fn get_token_utxo_for_epoch(
413	pool: &Pool<Postgres>,
414	asset: &Asset,
415	epoch: EpochNumber,
416) -> Result<Option<TokenTxOutput>, SqlxError> {
417	// In practice queried assets always have empty name.
418	// However, it's important to keep multi_asset.name condition, to enable use of compound index on multi_asset policy and name.
419	let sql = "SELECT
420			origin_tx.hash        AS origin_tx_hash,
421        	tx_out.index          AS utxo_index,
422        	origin_block.epoch_no AS tx_epoch_no,
423        	origin_block.block_no AS tx_block_no,
424        	origin_block.slot_no  AS tx_slot_no,
425        	origin_tx.block_index AS tx_block_index,
426        	datum.value           AS datum
427        FROM ma_tx_out
428        INNER JOIN multi_asset          ON ma_tx_out.ident = multi_asset.id
429        INNER JOIN tx_out               ON ma_tx_out.tx_out_id = tx_out.id
430        INNER JOIN tx origin_tx         ON tx_out.tx_id = origin_tx.id
431        INNER JOIN block origin_block   ON origin_tx.block_id = origin_block.id
432        LEFT JOIN datum                 ON tx_out.data_hash = datum.hash
433        WHERE multi_asset.policy = $1
434		AND multi_asset.name = $2
435        AND origin_block.epoch_no <= $3
436        ORDER BY tx_block_no DESC, origin_tx.block_index DESC
437        LIMIT 1";
438	Ok(sqlx::query_as::<_, TokenTxOutput>(sql)
439		.bind(&asset.policy_id.0)
440		.bind(&asset.asset_name.0)
441		.bind(epoch)
442		.fetch_optional(pool)
443		.await?)
444}

467#[cfg(feature = "governed-map")]
468pub(crate) async fn get_governed_map_changes_tx_in_enabled(
469	pool: &Pool<Postgres>,
470	address: &Address,
471	after_block: Option<BlockNumber>,
472	to_block: BlockNumber,
473	asset: Asset,
474) -> Result<Vec<DatumChangeOutput>, SqlxError> {
475	let query = "
476		((SELECT
477			datum.value as datum, origin_block.block_no as block_no, origin_tx.block_index as block_index, $6 as action, 1 as action_order
478		FROM tx_out
479		INNER JOIN tx origin_tx			ON tx_out.tx_id = origin_tx.id
480		INNER JOIN block origin_block	ON origin_tx.block_id = origin_block.id
481		INNER JOIN datum				ON tx_out.data_hash = datum.hash
482		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
483		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
484		WHERE
485			tx_out.address = $1 AND ($2 IS NULL OR origin_block.block_no > $2) AND origin_block.block_no <= $3
486			AND multi_asset.policy = $4
487			AND multi_asset.name = $5)
488		UNION
489		(SELECT
490			datum.value as datum, consuming_block.block_no as block_no, consuming_tx.block_index as block_index, $7 as action, -1 as action_order
491		FROM tx_out
492		LEFT JOIN tx_in consuming_tx_in	ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
493		LEFT JOIN tx consuming_tx		ON consuming_tx_in.tx_in_id = consuming_tx.id
494		LEFT JOIN block consuming_block	ON consuming_tx.block_id = consuming_block.id
495		INNER JOIN datum				ON tx_out.data_hash = datum.hash
496		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
497		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
498		WHERE
499			tx_out.address = $1
500			AND (consuming_tx_in.id IS NOT NULL AND ($2 IS NULL OR consuming_block.block_no > $2) AND consuming_block.block_no <= $3)
501			AND multi_asset.policy = $4
502			AND multi_asset.name = $5))
503		ORDER BY block_no, block_index, action_order ASC";
504	Ok(sqlx::query_as::<_, DatumChangeOutput>(query)
505		.bind(&address.0)
506		.bind(after_block)
507		.bind(to_block)
508		.bind(&asset.policy_id.0)
509		.bind(&asset.asset_name.0)
510		.bind(GovernedMapAction::Create)
511		.bind(GovernedMapAction::Spend)
512		.fetch_all(pool)
513		.await?)
514}

849/// Returns the list of all blocks in given range,
850/// together with the sum of all token amounts transferred to illiquid supply address in this block.
851/// On devnet postgres it takes around 5ms to execute when querying 1000 blocks.
852/// Used by `get_native_token_transfers` (NativeTokenDataSourceImpl).
853#[cfg(feature = "native-token")]
854pub(crate) async fn get_native_token_transfers(
855	pool: &Pool<Postgres>,
856	from_block: BlockNumber,
857	to_block: BlockNumber,
858	asset: Asset,
859	illiquid_supply_address: Address,
860) -> Result<Vec<BlockTokenAmount>, SqlxError> {
861	let query = sqlx::query_as::<_, BlockTokenAmount>(
862		"
863SELECT
864	block.hash as block_hash, COALESCE(a.amount, 0) as amount
865FROM block
866	LEFT JOIN (SELECT tx.block_id as block_id, SUM(ma_tx_out.quantity) as amount FROM tx
867		INNER JOIN tx_out ON tx.id = tx_out.tx_id AND tx_out.address = $1
868		INNER JOIN ma_tx_out    ON ma_tx_out.tx_out_id = tx_out.id
869		INNER JOIN multi_asset  ON multi_asset.id = ma_tx_out.ident
870		WHERE multi_asset.policy = $2 AND multi_asset.name = $3
871		GROUP BY tx.block_id
872		) as a ON block.id = a.block_id
873	WHERE
874		$4 <= block.block_no AND block.block_no <= $5
875	ORDER BY block.block_no ASC;
876",
877	)
878	.bind(&illiquid_supply_address.0)
879	.bind(&asset.policy_id.0)
880	.bind(&asset.asset_name.0)
881	.bind(from_block)
882	.bind(to_block);
883
884	Ok(query.fetch_all(pool).await?)
885}

998#[cfg(feature = "bridge")]
999async fn get_bridge_utxos_tx_in_consumed(
1000	pool: &Pool<Postgres>,
1001	icp_address: &Address,
1002	native_token: Asset,
1003	checkpoint: BridgeCheckpoint,
1004	to_block: BlockNumber,
1005	max_utxos: u32,
1006) -> Result<Vec<BridgeUtxo>, SqlxError> {
1007	use sqlx::QueryBuilder;
1008	let mut query_builder = QueryBuilder::new("
1009	SELECT
1010	      block.block_no                          AS block_number
1011	    , tx.block_index                          AS tx_ix
1012	    , tx.hash                                 AS tx_hash
1013	    , outputs.index                           AS utxo_ix
1014	    , output_tokens.quantity                  AS tokens_out
1015	    , coalesce(sum(input_tokens.quantity), 0) AS tokens_in
1016	    , datum.value                             AS datum
1017
1018	FROM tx_out      outputs
1019	JOIN tx                        ON outputs.tx_id = tx.id
1020	JOIN block                     ON tx.block_id = block.id
1021	JOIN ma_tx_out   output_tokens ON output_tokens.tx_out_id = outputs.id
1022	JOIN multi_asset native_token  ON native_token.id = output_tokens.ident
1023	JOIN datum                     ON datum.tx_id = tx.id
1024
1025	LEFT JOIN tx_out     inputs        ON inputs.consumed_by_tx_id = tx.id   AND inputs.address = $1
1026	LEFT JOIN ma_tx_out  input_tokens  ON input_tokens.tx_out_id = inputs.id AND input_tokens.ident = native_token.id
1027
1028	WHERE native_token.policy = $2
1029	  AND native_token.name = $3
1030	  AND outputs.address = $1
1031	  AND block_no <= $4
1032	");
1033
1034	match checkpoint {
1035		BridgeCheckpoint::Block { number } => {
1036			query_builder.push(&format!("AND block_no > {number} "));
1037		},
1038		BridgeCheckpoint::Utxo { block_number, tx_ix, tx_out_ix } => {
1039			query_builder.push(&format!(
1040				"AND (block_no, tx.block_index, outputs.index) > ({block_number}, {tx_ix}, {tx_out_ix}) "
1041			));
1042		},
1043	}
1044
1045	let query = query_builder
1046		.push("GROUP BY tx.hash, outputs.id, output_tokens.quantity, datum.value, block.block_no, tx.block_index, outputs.index ")
1047		.push("ORDER BY block.block_no, tx.block_index, outputs.index ")
1048		.push(&format!("LIMIT {max_utxos};"))
1049		.build_query_as::<BridgeUtxo>()
1050		.bind(&icp_address.0)
1051		.bind(&native_token.policy_id.0)
1052		.bind(&native_token.asset_name.0)
1053		.bind(to_block);
1054
1055	Ok(query.fetch_all(pool).await?)
1056}

1060	pool: &Pool<Postgres>,
1061	icp_address: &Address,
1062	native_token: Asset,
1063	checkpoint: BridgeCheckpoint,
1064	to_block: BlockNumber,
1065	max_utxos: u32,
1066) -> Result<Vec<BridgeUtxo>, SqlxError> {
1067	use sqlx::QueryBuilder;
1068
1069	let mut query_builder = QueryBuilder::new("
1070	SELECT
1071		block.block_no                          AS block_number
1072		, tx.block_index                          AS tx_ix
1073		, tx.hash                                 AS tx_hash
1074		, outputs.index                           AS utxo_ix
1075		, output_tokens.quantity                  AS tokens_out
1076		, coalesce(sum(input_tokens.quantity), 0) AS tokens_in
1077		, datum.value                             AS datum
1078
1079	FROM tx_out      outputs
1080	JOIN tx                        ON outputs.tx_id = tx.id
1081	JOIN block                     ON tx.block_id = block.id
1082	JOIN ma_tx_out   output_tokens ON output_tokens.tx_out_id = outputs.id
1083	JOIN multi_asset native_token  ON native_token.id = output_tokens.ident
1084	JOIN datum                     ON datum.tx_id = tx.id
1085
1086	LEFT JOIN tx_in      inputs_join   ON tx.id = inputs_join.tx_in_id
1087	LEFT JOIN tx_out     inputs        ON inputs_join.tx_out_id = inputs.tx_id and inputs_join.tx_out_index = inputs.index AND inputs.address = $1
1088	LEFT JOIN ma_tx_out  input_tokens  ON input_tokens.tx_out_id = inputs.id AND input_tokens.ident = native_token.id
1089
1090	WHERE native_token.policy = $2
1091      AND native_token.name = $3
1092      AND outputs.address = $1
1093      AND block_no <= $4
1094	");
1095
1096	match checkpoint {

1099		},
1100		BridgeCheckpoint::Utxo { block_number, tx_ix, tx_out_ix } => {
1101			query_builder.push(&format!(
1102				"AND (block_no, tx.block_index, outputs.index) > ({block_number}, {tx_ix}, {tx_out_ix}) "

1104		},
1105	}

1108		.push("GROUP BY tx.hash, outputs.id, output_tokens.quantity, datum.value, block.block_no, tx.block_index, outputs.index ")
1109		.push("ORDER BY block.block_no, tx.block_index, outputs.index ")
1110		.push(&format!("LIMIT {max_utxos};"))
1111		.build_query_as::<BridgeUtxo>()
1112		.bind(&icp_address.0)
