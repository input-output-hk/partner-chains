18/// Db-Sync `tx_in.value` configuration field
19#[derive(Debug, PartialEq, Copy, Clone)]
20pub(crate) enum TxInConfiguration {
21	/// Transaction inputs are linked using `tx_in` table
22	Enabled,
23	/// Transaction inputs are linked using `consumed_by_tx_id` column in `tx_out` table
24	Consumed,
25}
26
27impl TxInConfiguration {
28	pub(crate) async fn from_connection(pool: &Pool<Postgres>) -> Result<Self, SqlxError> {
29		let tx_in_exists = sqlx::query_scalar::<_, i64>(
30			"select count(*) from information_schema.tables where table_name = 'tx_in';",
31		)
32		.fetch_one(pool)
33		.await? == 1;
34
35		if !tx_in_exists {
36			return Ok(Self::Consumed);
37		}
38
39		let tx_in_populated = sqlx::query_scalar::<_, bool>("SELECT EXISTS (SELECT 1 FROM tx_in);")
40			.fetch_one(pool)
41			.await?;
42
43		if tx_in_populated {
44			return Ok(Self::Enabled);
45		}
46
47		Ok(Self::Consumed)
48	}
49}

51/// Structure that queries, caches and provides Db-Sync configuration
52pub struct DbSyncConfigurationProvider {
53	/// Postgres connection pool
54	pub(crate) pool: PgPool,
55	/// Transaction input configuration used by Db-Sync
56	pub(crate) tx_in_config: Arc<Mutex<OnceCell<TxInConfiguration>>>,
57}
58
59impl DbSyncConfigurationProvider {
60	pub(crate) fn new(pool: PgPool) -> Self {
61		Self { tx_in_config: Arc::new(Mutex::new(OnceCell::new())), pool }
62	}
63
64	pub(crate) async fn get_tx_in_config(
65		&self,
66	) -> std::result::Result<TxInConfiguration, DataSourceError> {
67		let lock = self.tx_in_config.lock().await;
68		if let Some(tx_in_config) = lock.get() {
69			return Ok(*tx_in_config);
70		} else {
71			let tx_in_config = TxInConfiguration::from_connection(&self.pool).await?;
72			lock.set(tx_in_config).map_err(|_| {
73				DataSourceError::InternalDataSourceError(
74					"Failed to set tx_in_config in DbSyncConfigurationProvider".into(),
75				)
76			})?;
77			log::info!("Using configuration: {tx_in_config:?}");
78			return Ok(tx_in_config);
79		}
80	}
81}

248    ",
249	)
250	.bind(epoch)
251	.bind(stake_pool_hashes)
252	.fetch_all(pool)
253	.await?)

302/// Query to get the highest block in the given time range, not higher than a given block
303/// # Arguments
304/// * `max_block_number` - block number upper boundary
305/// * `min_time` - time lower boundary
306/// * `min_slot` - slot lower boundary, for efficiency, because time is not indexed
307/// * `max_time` - time upper boundary
308/// * `max_slot` - slot upper boundary, for efficiency, because time is not indexed
309#[cfg(feature = "block-source")]
310pub(crate) async fn get_highest_block(
311	pool: &Pool<Postgres>,
312	max_block_number: BlockNumber,
313	min_time: NaiveDateTime,
314	min_slot: SlotNumber,
315	max_time: NaiveDateTime,
316	max_slot: SlotNumber,
317) -> Result<Option<Block>, SqlxError> {
318	Ok(sqlx::query_as::<_, Block>(
319		"
320SELECT block_no, hash, epoch_no, slot_no, time
321FROM block
322WHERE block_no <= $1 AND $2 <= slot_no AND slot_no <= $3 AND $4 <= time AND time <= $5
323ORDER BY block_no DESC
324LIMIT 1",
325	)
326	.bind(max_block_number)
327	.bind(min_slot)
328	.bind(max_slot)
329	.bind(min_time)
330	.bind(max_time)
331	.fetch_optional(pool)
332	.await?)

364#[cfg(feature = "candidate-source")]
365/// Returns number of the latest epoch that is stable - no block in such an epoch can be rolled back.
366/// The latest stable epoch is one less than epoch of the highest stable block (HSB),
367/// because unstable part could be replaced with blocks sequence starting from the block that has
368/// the same epoch as HSB.
369pub(crate) async fn get_latest_stable_epoch(
370	pool: &Pool<Postgres>,
371	security_parameter: u32,
372) -> Result<Option<EpochNumber>, SqlxError> {
373	let sql = "SELECT stable_block.epoch_no - 1 as epoch_no
374FROM block INNER JOIN block as stable_block ON block.block_no - $1 = stable_block.block_no
375WHERE block.block_no IS NOT NULL
376ORDER BY block.block_no DESC
377LIMIT 1";
378	#[allow(deprecated)]
379	Ok(sqlx::query_as::<_, EpochNumberRow>(sql)
380		.bind(BlockNumber(security_parameter))
381		.fetch_optional(pool)
382		.await?
383		.map(EpochNumber::from))
384}

400/// Returns the token data of the given policy at the given slot.
401#[cfg(feature = "candidate-source")]
402pub(crate) async fn get_token_utxo_for_epoch(
403	pool: &Pool<Postgres>,
404	asset: &Asset,
405	epoch: EpochNumber,
406) -> Result<Option<TokenTxOutput>, SqlxError> {
407	// In practice queried assets always have empty name.
408	// However, it's important to keep multi_asset.name condition, to enable use of compound index on multi_asset policy and name.
409	let sql = "SELECT
410			origin_tx.hash        AS origin_tx_hash,
411        	tx_out.index          AS utxo_index,
412        	origin_block.epoch_no AS tx_epoch_no,
413        	origin_block.block_no AS tx_block_no,
414        	origin_block.slot_no  AS tx_slot_no,
415        	origin_tx.block_index AS tx_block_index,
416        	datum.value           AS datum
417        FROM ma_tx_out
418        INNER JOIN multi_asset          ON ma_tx_out.ident = multi_asset.id
419        INNER JOIN tx_out               ON ma_tx_out.tx_out_id = tx_out.id
420        INNER JOIN tx origin_tx         ON tx_out.tx_id = origin_tx.id
421        INNER JOIN block origin_block   ON origin_tx.block_id = origin_block.id
422        LEFT JOIN datum                 ON tx_out.data_hash = datum.hash
423        WHERE multi_asset.policy = $1
424		AND multi_asset.name = $2
425        AND origin_block.epoch_no <= $3
426        ORDER BY tx_block_no DESC, origin_tx.block_index DESC
427        LIMIT 1";
428	Ok(sqlx::query_as::<_, TokenTxOutput>(sql)
429		.bind(&asset.policy_id.0)
430		.bind(&asset.asset_name.0)
431		.bind(epoch)
432		.fetch_optional(pool)
433		.await?)
434}

457#[cfg(feature = "governed-map")]
458pub(crate) async fn get_governed_map_changes_tx_in_enabled(
459	pool: &Pool<Postgres>,
460	address: &Address,
461	after_block: Option<BlockNumber>,
462	to_block: BlockNumber,
463	asset: Asset,
464) -> Result<Vec<DatumChangeOutput>, SqlxError> {
465	let query = "
466		((SELECT
467			datum.value as datum, origin_block.block_no as block_no, origin_tx.block_index as block_index, $6 as action, 1 as action_order
468		FROM tx_out
469		INNER JOIN tx origin_tx			ON tx_out.tx_id = origin_tx.id
470		INNER JOIN block origin_block	ON origin_tx.block_id = origin_block.id
471		INNER JOIN datum				ON tx_out.data_hash = datum.hash
472		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
473		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
474		WHERE
475			tx_out.address = $1 AND ($2 IS NULL OR origin_block.block_no > $2) AND origin_block.block_no <= $3
476			AND multi_asset.policy = $4
477			AND multi_asset.name = $5)
478		UNION
479		(SELECT
480			datum.value as datum, consuming_block.block_no as block_no, consuming_tx.block_index as block_index, $7 as action, -1 as action_order
481		FROM tx_out
482		LEFT JOIN tx_in consuming_tx_in	ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
483		LEFT JOIN tx consuming_tx		ON consuming_tx_in.tx_in_id = consuming_tx.id
484		LEFT JOIN block consuming_block	ON consuming_tx.block_id = consuming_block.id
485		INNER JOIN datum				ON tx_out.data_hash = datum.hash
486		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
487		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
488		WHERE
489			tx_out.address = $1
490			AND (consuming_tx_in.id IS NOT NULL AND ($2 IS NULL OR consuming_block.block_no > $2) AND consuming_block.block_no <= $3)
491			AND multi_asset.policy = $4
492			AND multi_asset.name = $5))
493		ORDER BY block_no, block_index, action_order ASC";
494	Ok(sqlx::query_as::<_, DatumChangeOutput>(query)
495		.bind(&address.0)
496		.bind(after_block)
497		.bind(to_block)
498		.bind(&asset.policy_id.0)
499		.bind(&asset.asset_name.0)
500		.bind(GovernedMapAction::Create)
501		.bind(GovernedMapAction::Spend)
502		.fetch_all(pool)
503		.await?)
504}

839/// Returns the list of all blocks in given range,
840/// together with the sum of all token amounts transferred to illiquid supply address in this block.
841/// On devnet postgres it takes around 5ms to execute when querying 1000 blocks.
842/// Used by `get_native_token_transfers` (NativeTokenDataSourceImpl).
843#[cfg(feature = "native-token")]
844pub(crate) async fn get_native_token_transfers(
845	pool: &Pool<Postgres>,
846	from_block: BlockNumber,
847	to_block: BlockNumber,
848	asset: Asset,
849	illiquid_supply_address: Address,
850) -> Result<Vec<BlockTokenAmount>, SqlxError> {
851	let query = sqlx::query_as::<_, BlockTokenAmount>(
852		"
853SELECT
854	block.hash as block_hash, COALESCE(a.amount, 0) as amount
855FROM block
856	LEFT JOIN (SELECT tx.block_id as block_id, SUM(ma_tx_out.quantity) as amount FROM tx
857		INNER JOIN tx_out ON tx.id = tx_out.tx_id AND tx_out.address = $1
858		INNER JOIN ma_tx_out    ON ma_tx_out.tx_out_id = tx_out.id
859		INNER JOIN multi_asset  ON multi_asset.id = ma_tx_out.ident
860		WHERE multi_asset.policy = $2 AND multi_asset.name = $3
861		GROUP BY tx.block_id
862		) as a ON block.id = a.block_id
863	WHERE
864		$4 <= block.block_no AND block.block_no <= $5
865	ORDER BY block.block_no ASC;
866",
867	)
868	.bind(&illiquid_supply_address.0)
869	.bind(&asset.policy_id.0)
870	.bind(&asset.asset_name.0)
871	.bind(from_block)
872	.bind(to_block);
873
874	Ok(query.fetch_all(pool).await?)
875}

988#[cfg(feature = "bridge")]
989async fn get_bridge_utxos_tx_in_consumed(
990	pool: &Pool<Postgres>,
991	icp_address: &Address,
992	native_token: Asset,
993	checkpoint: BridgeCheckpoint,
994	to_block: BlockNumber,
995	max_utxos: u32,
996) -> Result<Vec<BridgeUtxo>, SqlxError> {
997	use sqlx::QueryBuilder;
998	let mut query_builder = QueryBuilder::new("
999	SELECT
1000	      block.block_no                          AS block_number
1001	    , tx.block_index                          AS tx_ix
1002	    , tx.hash                                 AS tx_hash
1003	    , outputs.index                           AS utxo_ix
1004	    , output_tokens.quantity                  AS tokens_out
1005	    , coalesce(sum(input_tokens.quantity), 0) AS tokens_in
1006	    , datum.value                             AS datum
1007
1008	FROM tx_out      outputs
1009	JOIN tx                        ON outputs.tx_id = tx.id
1010	JOIN block                     ON tx.block_id = block.id
1011	JOIN ma_tx_out   output_tokens ON output_tokens.tx_out_id = outputs.id
1012	JOIN multi_asset native_token  ON native_token.id = output_tokens.ident
1013	JOIN datum                     ON datum.tx_id = tx.id
1014
1015	LEFT JOIN tx_out     inputs        ON inputs.consumed_by_tx_id = tx.id   AND inputs.address = $1
1016	LEFT JOIN ma_tx_out  input_tokens  ON input_tokens.tx_out_id = inputs.id AND input_tokens.ident = native_token.id
1017
1018	WHERE native_token.policy = $2
1019	  AND native_token.name = $3
1020	  AND outputs.address = $1
1021	  AND block_no <= $4
1022	");
1023
1024	match checkpoint {
1025		BridgeCheckpoint::Block { number } => {
1026			query_builder.push(&format!("AND block_no > {number} "));
1027		},
1028		BridgeCheckpoint::Utxo { block_number, tx_ix, tx_out_ix } => {
1029			query_builder.push(&format!(
1030				"AND (block_no, tx.block_index, outputs.index) > ({block_number}, {tx_ix}, {tx_out_ix}) "
1031			));
1032		},
1033	}
1034
1035	let query = query_builder
1036		.push("GROUP BY tx.hash, outputs.id, output_tokens.quantity, datum.value, block.block_no, tx.block_index, outputs.index ")
1037		.push("ORDER BY block.block_no, tx.block_index, outputs.index ")
1038		.push(&format!("LIMIT {max_utxos};"))
1039		.build_query_as::<BridgeUtxo>()
1040		.bind(&icp_address.0)
1041		.bind(&native_token.policy_id.0)
1042		.bind(&native_token.asset_name.0)
1043		.bind(to_block);
1044
1045	Ok(query.fetch_all(pool).await?)
1046}

1048#[cfg(feature = "bridge")]
1049async fn get_bridge_utxos_tx_in_enabled(
1050	pool: &Pool<Postgres>,
1051	icp_address: &Address,
1052	native_token: Asset,
1053	checkpoint: BridgeCheckpoint,
1054	to_block: BlockNumber,
1055	max_utxos: u32,
1056) -> Result<Vec<BridgeUtxo>, SqlxError> {
1057	use sqlx::QueryBuilder;
1058
1059	let mut query_builder = QueryBuilder::new("
1060	SELECT
1061		block.block_no                          AS block_number
1062		, tx.block_index                          AS tx_ix
1063		, tx.hash                                 AS tx_hash
1064		, outputs.index                           AS utxo_ix
1065		, output_tokens.quantity                  AS tokens_out
1066		, coalesce(sum(input_tokens.quantity), 0) AS tokens_in
1067		, datum.value                             AS datum
1068
1069	FROM tx_out      outputs
1070	JOIN tx                        ON outputs.tx_id = tx.id
1071	JOIN block                     ON tx.block_id = block.id
1072	JOIN ma_tx_out   output_tokens ON output_tokens.tx_out_id = outputs.id
1073	JOIN multi_asset native_token  ON native_token.id = output_tokens.ident
1074	JOIN datum                     ON datum.tx_id = tx.id
1075
1076	LEFT JOIN tx_in      inputs_join   ON tx.id = inputs_join.tx_in_id
1077	LEFT JOIN tx_out     inputs        ON inputs_join.tx_out_id = inputs.tx_id and inputs_join.tx_out_index = inputs.index AND inputs.address = $1
1078	LEFT JOIN ma_tx_out  input_tokens  ON input_tokens.tx_out_id = inputs.id AND input_tokens.ident = native_token.id
1079
1080	WHERE native_token.policy = $2
1081      AND native_token.name = $3
1082      AND outputs.address = $1
1083      AND block_no <= $4
1084	");
1085
1086	match checkpoint {

1094		},
1095	}

1098		.push("GROUP BY tx.hash, outputs.id, output_tokens.quantity, datum.value, block.block_no, tx.block_index, outputs.index ")
1099		.push("ORDER BY block.block_no, tx.block_index, outputs.index ")
1100		.push(&format!("LIMIT {max_utxos};"))
1101		.build_query_as::<BridgeUtxo>()
1102		.bind(&icp_address.0)
1103		.bind(&native_token.policy_id.0)
1104		.bind(&native_token.asset_name.0)
