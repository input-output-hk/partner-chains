51impl TryFrom<MainchainTxOutputRow> for MainchainTxOutput {
52	type Error = sqlx::Error;
53	fn try_from(r: MainchainTxOutputRow) -> Result<Self, Self::Error> {
54		let tx_inputs: Result<Vec<UtxoId>, _> =
55			r.tx_inputs.into_iter().map(|i| UtxoId::from_str(i.as_str())).collect();
56		let tx_inputs = tx_inputs.map_err(|e| sqlx::Error::Decode(e.into()))?;
57		Ok(MainchainTxOutput {
58			utxo_id: UtxoId {
59				tx_hash: McTxHash(r.utxo_id_tx_hash),
60				index: UtxoIndex(r.utxo_id_index.0),
61			},
62			tx_block_no: r.tx_block_no,
63			tx_slot_no: r.tx_slot_no,
64			tx_epoch_no: r.tx_epoch_no,
65			tx_index_in_block: r.tx_index_in_block,
66			address: r.address,
67			datum: r.datum.map(|d| d.0),
68			tx_inputs,
69		})
70	}
71}

175	epoch: EpochNumber,
176	stake_pool_hashes: Vec<[u8; 28]>,

178	Ok(sqlx::query_as::<_, StakePoolDelegationOutputRow>(
179		"
180SELECT
181	epoch_stake.amount AS epoch_stake_amount,
182	pool_hash.hash_raw AS pool_hash_raw,
183	stake_address.hash_raw AS stake_address_hash_raw,
184	stake_address.script_hash AS stake_address_script_hash
185FROM
186			   epoch_stake
187	INNER JOIN stake_address      ON epoch_stake.addr_id = stake_address.id
188	INNER JOIN pool_hash          ON epoch_stake.pool_id = pool_hash.id

190	    epoch_stake.epoch_no = $1
191	AND epoch_stake.amount > 0
192	AND pool_hash.hash_raw IN (SELECT unnest($2))
193    ",
194	)
195	.bind(epoch)
196	.bind(stake_pool_hashes)
197	.fetch_all(pool)
198	.await?)

247/// Query to get the highest block in the given time range, not higher than a given block
248/// # Arguments
249/// * `max_block_number` - block number upper boundary
250/// * `min_time` - time lower boundary
251/// * `min_slot` - slot lower boundary, for efficiency, because time is not indexed
252/// * `max_time` - time upper boundary
253/// * `max_slot` - slot upper boundary, for efficiency, because time is not indexed
254#[cfg(feature = "block-source")]
255pub(crate) async fn get_highest_block(
256	pool: &Pool<Postgres>,
257	max_block_number: BlockNumber,
258	min_time: NaiveDateTime,
259	min_slot: SlotNumber,
260	max_time: NaiveDateTime,
261	max_slot: SlotNumber,
262) -> Result<Option<Block>, SqlxError> {
263	Ok(sqlx::query_as::<_, Block>(
264		"
265SELECT block_no, hash, epoch_no, slot_no, time
266FROM block
267WHERE block_no <= $1 AND $2 <= slot_no AND slot_no <= $3 AND $4 <= time AND time <= $5
268ORDER BY block_no DESC
269LIMIT 1",
270	)
271	.bind(max_block_number)
272	.bind(min_slot)
273	.bind(max_slot)
274	.bind(min_time)
275	.bind(max_time)
276	.fetch_optional(pool)
277	.await?)

310/// Returns number of the latest epoch that is stable - no block in such an epoch can be rolled back.
311/// The latest stable epoch is one less than epoch of the highest stable block (HSB),
312/// because unstable part could be replaced with blocks sequence starting from the block that has
313/// the same epoch as HSB.
314pub(crate) async fn get_latest_stable_epoch(
315	pool: &Pool<Postgres>,
316	security_parameter: u32,
317) -> Result<Option<EpochNumber>, SqlxError> {
318	let sql = "SELECT stable_block.epoch_no - 1 as epoch_no
319FROM block INNER JOIN block as stable_block ON block.block_no - $1 = stable_block.block_no
320WHERE block.block_no IS NOT NULL
321ORDER BY block.block_no DESC
322LIMIT 1";
323	Ok(sqlx::query_as::<_, EpochNumberRow>(sql)
324		.bind(BlockNumber(security_parameter))
325		.fetch_optional(pool)
326		.await?
327		.map(EpochNumber::from))
328}

342/// Returns the token data of the given policy at the given slot.
343#[cfg(feature = "candidate-source")]
344pub(crate) async fn get_token_utxo_for_epoch(
345	pool: &Pool<Postgres>,
346	asset: &Asset,
347	epoch: EpochNumber,
348) -> Result<Option<TokenTxOutput>, SqlxError> {
349	// In practice queried assets always have empty name.
350	// However, it's important to keep multi_asset.name condition, to enable use of compound index on multi_asset policy and name.
351	let sql = "SELECT
352			origin_tx.hash        AS origin_tx_hash,
353        	tx_out.index          AS utxo_index,
354        	origin_block.epoch_no AS tx_epoch_no,
355        	origin_block.block_no AS tx_block_no,
356        	origin_block.slot_no  AS tx_slot_no,
357        	origin_tx.block_index AS tx_block_index,
358        	datum.value           AS datum
359        FROM ma_tx_out
360        INNER JOIN multi_asset          ON ma_tx_out.ident = multi_asset.id
361        INNER JOIN tx_out               ON ma_tx_out.tx_out_id = tx_out.id
362        INNER JOIN tx origin_tx         ON tx_out.tx_id = origin_tx.id
363        INNER JOIN block origin_block   ON origin_tx.block_id = origin_block.id
364        LEFT JOIN datum                 ON tx_out.data_hash = datum.hash
365        WHERE multi_asset.policy = $1
366		AND multi_asset.name = $2
367        AND origin_block.epoch_no <= $3
368        ORDER BY tx_block_no DESC, origin_tx.block_index DESC
369        LIMIT 1";
370	Ok(sqlx::query_as::<_, TokenTxOutput>(sql)
371		.bind(&asset.policy_id.0)
372		.bind(&asset.asset_name.0)
373		.bind(epoch)
374		.fetch_optional(pool)
375		.await?)
376}

379pub(crate) async fn get_changes(
380	pool: &Pool<Postgres>,
381	address: &Address,
382	after_block: Option<BlockNumber>,
383	to_block: BlockNumber,
384	asset: Asset,
385) -> Result<Vec<DatumChangeOutput>, SqlxError> {
386	let query = "
387		((SELECT
388			datum.value as datum, origin_block.block_no as block_no, origin_tx.block_index as block_index, 'upsert' as action
389		FROM tx_out
390		INNER JOIN tx origin_tx			ON tx_out.tx_id = origin_tx.id
391		INNER JOIN block origin_block	ON origin_tx.block_id = origin_block.id
392		INNER JOIN datum				ON tx_out.data_hash = datum.hash
393		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
394		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
395		WHERE
396			tx_out.address = $1 AND ($2 IS NULL OR origin_block.block_no > $2) AND origin_block.block_no <= $3
397			AND multi_asset.policy = $4
398			AND multi_asset.name = $5)
399		UNION
400		(SELECT
401			datum.value as datum, consuming_block.block_no as block_no, consuming_tx.block_index as block_index, 'remove' as action
402		FROM tx_out
403		LEFT JOIN tx_in consuming_tx_in	ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
404		LEFT JOIN tx consuming_tx		ON consuming_tx_in.tx_in_id = consuming_tx.id
405		LEFT JOIN block consuming_block	ON consuming_tx.block_id = consuming_block.id
406		INNER JOIN datum				ON tx_out.data_hash = datum.hash
407		INNER JOIN ma_tx_out			ON tx_out.id = ma_tx_out.tx_out_id
408		INNER JOIN multi_asset			ON multi_asset.id = ma_tx_out.ident
409		WHERE
410			tx_out.address = $1
411			AND (consuming_tx_in.id IS NOT NULL AND ($2 IS NULL OR consuming_block.block_no > $2) AND consuming_block.block_no <= $3)
412			AND multi_asset.policy = $4
413			AND multi_asset.name = $5))
414		ORDER BY block_no, block_index ASC";
415	Ok(sqlx::query_as::<_, DatumChangeOutput>(query)
416		.bind(&address.0)
417		.bind(after_block)
418		.bind(to_block)
419		.bind(&asset.policy_id.0)
420		.bind(&asset.asset_name.0)
421		.fetch_all(pool)
422		.await?)
423}

470#[cfg(feature = "candidate-source")]
471pub(crate) async fn get_utxos_for_address(
472	pool: &Pool<Postgres>,
473	address: &Address,
474	block: BlockNumber,
475) -> Result<Vec<MainchainTxOutput>, SqlxError> {
476	let query = "SELECT
477          		origin_tx.hash as utxo_id_tx_hash,
478          		tx_out.index as utxo_id_index,
479          		origin_block.block_no as tx_block_no,
480          		origin_block.slot_no as tx_slot_no,
481          		origin_block.epoch_no as tx_epoch_no,
482          		origin_tx.block_index as tx_index_in_block,
483          		tx_out.address,
484          		datum.value as datum,
485          		array_agg(concat_ws('#', encode(consumes_tx.hash, 'hex'), consumes_tx_in.tx_out_index)) as tx_inputs
486			FROM tx_out
487			INNER JOIN tx    origin_tx       ON tx_out.tx_id = origin_tx.id
488			INNER JOIN block origin_block    ON origin_tx.block_id = origin_block.id
489          	LEFT JOIN tx_in consuming_tx_in  ON tx_out.tx_id = consuming_tx_in.tx_out_id AND tx_out.index = consuming_tx_in.tx_out_index
490          	LEFT JOIN tx    consuming_tx     ON consuming_tx_in.tx_in_id = consuming_tx.id
491          	LEFT JOIN block consuming_block  ON consuming_tx.block_id = consuming_block.id
492			LEFT JOIN tx_in consumes_tx_in   ON consumes_tx_in.tx_in_id = origin_tx.id
493          	LEFT JOIN tx_out consumes_tx_out ON consumes_tx_out.tx_id = consumes_tx_in.tx_out_id AND consumes_tx_in.tx_out_index = consumes_tx_out.index
494          	LEFT JOIN tx consumes_tx         ON consumes_tx.id = consumes_tx_out.tx_id
495          	LEFT JOIN datum                  ON tx_out.data_hash = datum.hash
496          	WHERE
497          		tx_out.address = $1 AND origin_block.block_no <= $2
498          		AND (consuming_tx_in.id IS NULL OR consuming_block.block_no > $2)
499          		GROUP BY (
500					utxo_id_tx_hash,
501					utxo_id_index,
502					tx_block_no,
503					tx_slot_no,
504					tx_epoch_no,
505					tx_index_in_block,
506					tx_out.address,
507					datum
508				)";
509	let rows = sqlx::query_as::<_, MainchainTxOutputRow>(query)
510		.bind(&address.0)
511		.bind(block)
512		.fetch_all(pool)
513		.await?;
514	let result: Result<Vec<MainchainTxOutput>, sqlx::Error> =
515		rows.into_iter().map(MainchainTxOutput::try_from).collect();
516	Ok(result?)
517}

549		"
550SELECT
551    COALESCE(SUM(ma_tx_out.quantity), 0)
552FROM tx_out
553LEFT JOIN ma_tx_out    ON ma_tx_out.tx_out_id = tx_out.id
554LEFT JOIN multi_asset  ON multi_asset.id = ma_tx_out.ident
555INNER JOIN tx          ON tx_out.tx_id = tx.id
556INNER JOIN block       ON tx.block_id = block.id
557WHERE address = $1
558AND multi_asset.policy = $2
559AND multi_asset.name = $3
560AND block.block_no <= $4;
561    ",
562	)
563	.bind(&illiquid_supply_address.0)
564	.bind(&asset.policy_id.0)
565	.bind(&asset.asset_name.0)
566	.bind(to_block);
567
568	Ok(query.fetch_one(pool).await?.0)
569}

571/// Returns the list of all blocks in given range,
572/// together with the sum of all token amounts transferred to illiquid supply address in this block.
573/// On devnet postgres it takes around 5ms to execute when querying 1000 blocks.
574/// Used by `get_native_token_transfers` (NativeTokenDataSourceImpl).
575#[cfg(feature = "native-token")]
576pub(crate) async fn get_native_token_transfers(
577	pool: &Pool<Postgres>,
578	from_block: BlockNumber,
579	to_block: BlockNumber,
580	asset: Asset,
581	illiquid_supply_address: Address,
582) -> Result<Vec<BlockTokenAmount>, SqlxError> {
583	let query = sqlx::query_as::<_, BlockTokenAmount>(
584		"
585SELECT
586    block.hash as block_hash, COALESCE(SUM(ma_tx_out.quantity), 0) as amount
587FROM block
588    LEFT JOIN tx ON block.id = tx.block_id
589    LEFT JOIN tx_out ON tx.id = tx_out.tx_id AND tx_out.address = $1
590    LEFT JOIN ma_tx_out    ON ma_tx_out.tx_out_id = tx_out.id
591    LEFT JOIN multi_asset  ON multi_asset.id = ma_tx_out.ident AND multi_asset.policy = $2 AND multi_asset.name = $3
592WHERE
593    $4 <= block.block_no AND block.block_no <= $5
594GROUP BY block.block_no, block.hash
595ORDER BY block.block_no ASC;
596    ",
597	)
598	.bind(&illiquid_supply_address.0)
599	.bind(&asset.policy_id.0)
600	.bind(&asset.asset_name.0)
601	.bind(from_block)
602	.bind(to_block);
603
604	Ok(query.fetch_all(pool).await?)
605}
