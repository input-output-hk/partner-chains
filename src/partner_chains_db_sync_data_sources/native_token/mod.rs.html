22observed_async_trait!(
23impl NativeTokenManagementDataSource for NativeTokenManagementDataSourceImpl {
24	// after_block is always less or equal to_block
25	// to_block is always a stable block
26	async fn get_total_native_token_transfer(
27		&self,
28		after_block: Option<McBlockHash>,
29		to_block: McBlockHash,
30		scripts: MainChainScripts,
31	) -> std::result::Result<NativeTokenAmount, Box<dyn std::error::Error + Send + Sync>> {
32		if let Some(after_block) = after_block {
33			if after_block == to_block {
34				Ok(NativeTokenAmount(0))
35			} else if let Some(amount) = self.get_from_cache(&after_block, &to_block, &scripts) {
36				log::debug!(
37					"Illiquid supply transfers sum from cache after block '{:?}' to block '{:?}' is {}",
38					after_block, to_block, amount.0
39				);
40				Ok(amount)
41			} else {
42				log::debug!("Illiquid supply transfers after block '{:?}' to block '{:?}' not found in cache.", after_block, to_block);
43				let block_to_amount = self
44					.get_data_to_cache(&after_block, &to_block, &scripts)
45					.await?;
46				log::debug!("Caching illiquid supply transfers from {} blocks", block_to_amount.len());
47
48				let amount = block_to_amount
49					.iter()
50					.skip(1) // the first element is the 'after_block' which is not included in the sum
51					.take_while_inclusive(|(block_hash, _)| *block_hash != to_block)
52					.map(|(_, amount)| amount)
53					.sum();
54				log::debug!("Amount of illiquid supply transfers is {}", amount);
55
56				if let Ok(mut cache) = self.cache.lock() {
57					cache.update(block_to_amount, scripts)
58				}
59				Ok(NativeTokenAmount(amount))
60			}
61		} else {
62			let amount = self
63				.query_transfers_from_genesis(&to_block, &scripts)
64				.await?;
65			log::debug!("Amount of illiquid supply transfers from genesis to {} is {}", to_block, amount.0);
66			Ok(amount)
67		}
68	}
69});

72	pub async fn new(
73		pool: PgPool,
74		metrics_opt: Option<McFollowerMetrics>,

76		cache_size: u16,

79		let cache = Default::default();
80		Ok(Self { pool, metrics_opt, security_parameter, cache_size, cache })
81	}

83	pub async fn new_from_env(
84		pool: PgPool,
85		metrics_opt: Option<McFollowerMetrics>,
86	) -> std::result::Result<Self, Box<dyn std::error::Error + Send + Sync>> {
87		let security_parameter: u32 = std::env::var("CARDANO_SECURITY_PARAMETER")
88			.ok()
89			.and_then(|s| s.parse().ok())
90			.ok_or("Couldn't read env variable CARDANO_SECURITY_PARAMETER as u32")?;
91		Self::new(pool, metrics_opt, security_parameter, 1000).await
92	}
93
94	fn get_from_cache(
95		&self,
96		after_block: &McBlockHash,
97		to_block: &McBlockHash,
98		scripts: &MainChainScripts,
99	) -> Option<NativeTokenAmount> {
100		let cache = self.cache.lock().ok()?;
101		if cache.scripts.as_ref() == Some(scripts) {
102			cache.get_sum_in_range(after_block, to_block).map(NativeTokenAmount)
103		} else {
104			None
105		}
106	}
107
108	// invariant: to_block is always a stable block
109	// Returned data contains the 'from_block', it is required as guard for the cache
110	async fn get_data_to_cache(
111		&self,
112		from_block: &McBlockHash,
113		to_block: &McBlockHash,
114		scripts: &MainChainScripts,
115	) -> Result<Vec<(McBlockHash, u128)>> {
116		let (from_block_no, to_block_no, latest_block) = futures::try_join!(
117			get_from_block_no(from_block, &self.pool),
118			get_to_block_no(to_block, &self.pool),
119			get_latest_block(&self.pool),
120		)?;
121		let latest_stable_block = latest_block.block_no.0.saturating_sub(self.security_parameter);
122
123		// to_block_no is always a stable block, so it is not above latest_stable_block,
124		// but from_block_no + cache_size could be above latest_stable_block, so min has to be applied
125		let cache_to_block_no = BlockNumber(std::cmp::min(
126			latest_stable_block,
127			std::cmp::max(to_block_no.0, from_block_no.0.saturating_add(self.cache_size.into())),
128		));
129		// transfers starts with block having hash equal to after_block or genesis
130		let transfers = self.query_db(from_block_no, cache_to_block_no, scripts).await?;
131		Ok(transfers.iter().map(|t| (McBlockHash(t.block_hash), t.amount.0)).collect())
132	}

134	async fn query_db(

146	}

149		&self,

151		scripts: &MainChainScripts,
152	) -> Result<NativeTokenAmount> {
153		let to_block = get_to_block_no(to_block, &self.pool).await?;
154		Ok(crate::db_model::get_total_native_tokens_transfered(
155			&self.pool,
156			to_block,
157			to_db_asset(scripts),
158			scripts.illiquid_supply_validator_address.clone().into(),
159		)
160		.await?
161		.into())
162	}

198#[derive(Default)]
199pub(crate) struct Cache {
200	/// Continous blocks with their respective total native token transfer amount
201	block_hash_to_amount: Vec<(McBlockHash, u128)>,
202	pub(crate) scripts: Option<MainChainScripts>,
203}
204
205impl Cache {
206	/// Returns the sum of native token transfer amounts after `after` and to `to` block
207	/// Returns None if `after` or `to` block is not found, because it indicates that the cache
208	/// doesn't contain the required blocks interval.
209	fn get_sum_in_range(&self, after: &McBlockHash, to: &McBlockHash) -> Option<u128> {
210		let after_idx = self.block_hash_to_amount.iter().position(|(block, _)| block == after)?;
211		let to_idx = self.block_hash_to_amount.iter().position(|(block, _)| block == to)?;
212		let after_to = self.block_hash_to_amount.get((after_idx + 1)..=to_idx)?;
213		Some(after_to.iter().map(|(_, amount)| amount).sum())
214	}
215
216	pub fn update(
217		&mut self,
218		block_hash_to_amount: Vec<(McBlockHash, u128)>,
219		scripts: MainChainScripts,
220	) {
221		self.block_hash_to_amount = block_hash_to_amount;
222		self.scripts = Some(scripts);
223	}
224}
