1use crate::SessionValidatorManagementQueryApi;
2use sidechain_domain::{McEpochNumber, StakePoolPublicKey};

4pub async fn cli_get_ariadne_parameters(
5	query: impl SessionValidatorManagementQueryApi,
6	mc_epoch_number: McEpochNumber,
7) -> Result<String, String> {
8	let ariadne_parameters = query
9		.get_ariadne_parameters(mc_epoch_number)
10		.await
11		.map_err(|err| err.to_string())?;
12	let json = serde_json::to_value(ariadne_parameters).map_err(|err| err.to_string())?;
13	serde_json::to_string_pretty(&json).map_err(|err| err.to_string())
14}

16pub async fn cli_get_registration_status(
17	query: impl SessionValidatorManagementQueryApi,
18	mc_epoch_number: McEpochNumber,
19	stake_pool_public_key: StakePoolPublicKey,
20) -> Result<String, String> {
21	let registrations = query
22		.get_registrations(mc_epoch_number, stake_pool_public_key.clone())
23		.await
24		.map_err(|err| err.to_string())?;
25	let registrations_json = serde_json::to_value(registrations).map_err(|err| err.to_string())?;
26	serde_json::to_string_pretty(&registrations_json).map_err(|err| err.to_string())
27}
