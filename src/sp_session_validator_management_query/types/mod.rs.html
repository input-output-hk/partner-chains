25#[serde(rename_all = "camelCase")]
26pub struct GetCommitteeResponse {
27	pub sidechain_epoch: u64,
28	pub committee: Vec<CommitteeMember>,
29}
30
31impl GetCommitteeResponse {
32	pub fn new<Member: CommitteeMemberT>(
33		sidechain_epoch: sidechain_domain::ScEpochNumber,
34		committee: Vec<Member>,
35	) -> GetCommitteeResponse
36	where
37		Member::AuthorityId: AsRef<[u8]>,
38	{
39		let committee = committee
40			.into_iter()
41			.map(|member| CommitteeMember::new(member.authority_id()))
42			.collect();
43		GetCommitteeResponse { sidechain_epoch: sidechain_epoch.0, committee }
44	}
45
46	pub fn new_legacy<AuthorityId>(
47		sidechain_epoch: sidechain_domain::ScEpochNumber,
48		committee: Vec<AuthorityId>,
49	) -> GetCommitteeResponse
50	where
51		AuthorityId: AsRef<[u8]>,
52	{
53		let committee = committee.into_iter().map(|member| CommitteeMember::new(member)).collect();
54		GetCommitteeResponse { sidechain_epoch: sidechain_epoch.0, committee }
55	}
56}
