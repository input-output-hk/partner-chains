19#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
20#[serde(rename_all = "camelCase")]
21/// The D-parameter describing the ratio of permissioned and registered candidates.
22pub struct DParameter {
23	/// The number of permissioned candidates

29impl From<sidechain_domain::DParameter> for DParameter {

31		sidechain_domain::DParameter {

37	}
38}

40#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
41#[serde(rename_all = "camelCase")]
42/// Represents data associated with a permissioned candidate
43pub struct PermissionedCandidateData {
44	/// Sidechain public key of the candidate
45	pub sidechain_public_key: SidechainPublicKey,
46	/// All keys specific to the particular partner chain
47	pub keys: HashMap<String, ByteString>,
48	/// Is the registration valid
49	pub is_valid: bool,
50	/// Human-readable reasons of registration being invalid. Present only for invalid entries.
51	#[serde(skip_serializing_if = "Option::is_none")]
52	pub invalid_reasons: Option<PermissionedCandidateDataError>,
53}
54
55impl PermissionedCandidateData {
56	/// Constructor for [PermissionedCandidateData]
57	pub fn new(
58		data: sidechain_domain::PermissionedCandidateData,
59		invalid_reasons: Option<PermissionedCandidateDataError>,
60	) -> Self {
61		Self {
62			sidechain_public_key: data.sidechain_public_key,
63			keys: keys_to_map(&data.keys),
64			is_valid: invalid_reasons.is_none(),
65			invalid_reasons,
66		}
67	}
