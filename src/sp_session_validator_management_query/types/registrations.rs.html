17#[derive(Debug, PartialEq, Eq, Clone, Decode, thiserror::Error, Serialize, Deserialize)]
18pub enum RegistrationError {
19	#[error("{0}")]
20	StakeError(#[from] StakeError),
21	#[error("{0}")]
22	InvalidRegistrationData(#[from] RegistrationDataError),
23}
24
25#[derive(Default, Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
26#[serde(rename_all = "camelCase")]
27pub struct CandidateRegistrationEntry {
28	pub sidechain_pub_key: String,
29	/// SS58 address derived from public key. ss58(blake2b32(sidechainPubKey))
30	pub sidechain_account_id: String,
31	/// Stake Pool public key
32	pub mainchain_pub_key: String,
33	pub cross_chain_pub_key: String,
34	pub aura_pub_key: String,
35	pub grandpa_pub_key: String,
36	pub sidechain_signature: String,
37	/// Signature made with Stake Pool key
38	pub mainchain_signature: String,
39	pub cross_chain_signature: String,
40	/// Data of Utxo that contained this registration"
41	pub utxo: UtxoInfo,
42	#[serde(skip_serializing_if = "Option::is_none")]
43	pub stake_delegation: Option<u64>,
44	pub is_valid: bool,
45	/// Human readable reasons of registration being invalid. Present only for invalid entries.
46	#[serde(skip_serializing_if = "Option::is_none")]
47	pub invalid_reasons: Option<RegistrationError>,
48}

50pub type GetRegistrationsResponseMap = HashMap<String, Vec<CandidateRegistrationEntry>>;

52impl CandidateRegistrationEntry {
53	pub fn new(
54		registration_data: RegistrationData,
55		stake_pool_public_key: StakePoolPublicKey,
56		stake_delegation: Option<StakeDelegation>,
57		invalid_reasons: Option<RegistrationDataError>,
58	) -> Self {
59		Self {
60			sidechain_pub_key: to_hex(&registration_data.sidechain_pub_key.0, false),
61			sidechain_account_id: Self::sidechain_account_ss58(
62				registration_data.sidechain_pub_key.clone(),
63			)
64			.unwrap_or("Invalid Sidechain Public Key. Could not decode...".into()),
65			mainchain_pub_key: to_hex(&stake_pool_public_key.0.clone(), false),
66			cross_chain_pub_key: to_hex(&registration_data.cross_chain_pub_key.0, false),
67			aura_pub_key: to_hex(&registration_data.aura_pub_key.0, false),
68			grandpa_pub_key: to_hex(&registration_data.grandpa_pub_key.0, false),
69			sidechain_signature: to_hex(&registration_data.sidechain_signature.0, false),
70			mainchain_signature: to_hex(&registration_data.mainchain_signature.0, false),
71			cross_chain_signature: to_hex(&registration_data.cross_chain_signature.0, false),
72			utxo: registration_data.utxo_info,
73			stake_delegation: stake_delegation.map(|sd| sd.0),
74			is_valid: invalid_reasons.is_none(),
75			invalid_reasons: invalid_reasons.map(|e| e.into()),
76		}
77	}
78
79	fn sidechain_account_ss58(sidechain_pub_key: SidechainPublicKey) -> Result<String, Vec<u8>> {
80		let sidechain_ecdsa_pub_key =
81			ecdsa::Public::from(<[u8; 33]>::try_from(sidechain_pub_key.0)?);
82		let sidechain_account: AccountId32 =
83			MultiSigner::from(sidechain_ecdsa_pub_key).into_account();
84		let sidechain_account_ss58 = sidechain_account.to_ss58check();
85		Ok(sidechain_account_ss58)
86	}
87}
