16/// Interface for retrieving information about slot and epoch of Partner Chain blocks
17pub trait SidechainInfo<Block: BlockT>: Client<Block> {
18	/// Finds the Partner Chain eopch number for a given block number
19	fn get_epoch_of_block(&self, block_number: NumberFor<Block>)
20	-> Result<ScEpochNumber, ApiError>;
21}
22
23#[allow(deprecated)]
24impl<C, Block> SidechainInfo<Block> for C
25where
26	C: Client<Block> + Send + Sync + 'static,
27	C::Api: GetSidechainStatus<Block>,
28	Block: BlockT,
29	NumberFor<Block>: From<u32> + Into<u32>,
30{
31	fn get_epoch_of_block(
32		&self,
33		block_number: NumberFor<Block>,
34	) -> Result<ScEpochNumber, ApiError> {
35		let api = self.runtime_api();
36		let block_hash = self
37			.hash(block_number)?
38			.ok_or(ApiError::UnknownBlock(format!("Block Number {block_number} does not exist")))?;
39		let sidechain_status = api.get_sidechain_status(block_hash)?;
40		Ok(sidechain_status.epoch)
41	}
42}

44/// Runtime client capable of finding Partner Chain blocks via binary search
45pub trait FindSidechainBlock<Block: BlockT>: Client<Block> + Sized {
46	/// Finds any block in the given epoch if it exists
47	fn find_any_block_in_epoch(&self, epoch: ScEpochNumber) -> Result<Block::Hash, ApiError>;
48}
49
50#[allow(deprecated)]
51impl<C, Block> FindSidechainBlock<Block> for C
52where
53	C: Client<Block> + Send + Sync + 'static,
54	Block: BlockT,
55	NumberFor<Block>: From<u32> + Into<u32>,
56	C::Api: GetSidechainStatus<Block>,
57{
58	/// Finds any block in the given epoch if it exists
59	fn find_any_block_in_epoch(&self, epoch: ScEpochNumber) -> Result<Block::Hash, ApiError> {
60		let mut left = 1u32;
61		let mut right: u32 = self.info().best_number.into();
62
63		while left <= right {
64			let middle = (left + right) / 2;
65			let block_epoch = self.get_epoch_of_block(middle.into())?;
66
67			match block_epoch.cmp(&epoch) {
68				Ordering::Less => left = middle + 1,
69				Ordering::Greater => right = middle - 1,
70				Ordering::Equal => {
71					return Ok(self.hash(middle.into())?.expect(
72						"Block with given number exists, so its hash should exists as well",
73					));
74				},
75			}
76		}
77
78		return Err(ApiError::Application("Could not find block".to_string().into()));
79	}
