37impl<T> Register for T
38where
39	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,

42		&self,

45		candidate_registration: &CandidateRegistration,
46		payment_signing_key: &CardanoPaymentSigningKey,
47	) -> Result<Option<McTxHash>, OffchainError> {
48		run_register(genesis_utxo, candidate_registration, payment_signing_key, self, retries)

51	}

54pub async fn run_register<
55	C: QueryLedgerState + QueryNetwork + QueryUtxoByUtxoId + Transactions,
56	A: AwaitTx,
57>(
58	genesis_utxo: UtxoId,
59	candidate_registration: &CandidateRegistration,
60	payment_signing_key: &CardanoPaymentSigningKey,
61	client: &C,
62	await_tx: A,
63) -> anyhow::Result<Option<McTxHash>> {
64	let ctx = TransactionContext::for_payment_key(payment_signing_key, client).await?;
65	let validator = crate::scripts_data::registered_candidates_scripts(genesis_utxo)?;
66	let validator_address = validator.address_bech32(ctx.network)?;
67	let registration_utxo = ctx
68		.payment_key_utxos
69		.iter()
70		.find(|u| u.utxo_id() == candidate_registration.registration_utxo)
71		.ok_or(anyhow!("registration utxo not found at payment address"))?;
72	let all_registration_utxos = client.query_utxos(&[validator_address]).await?;
73	let own_registrations = get_own_registrations(
74		candidate_registration.own_pkh,
75		candidate_registration.stake_ownership.pub_key.clone(),
76		&all_registration_utxos,
77	);
78
79	if own_registrations.iter().any(|(_, existing_registration)| {
80		candidate_registration.matches_keys(existing_registration)
81	}) {
82		log::info!("✅ Candidate already registered with same keys.");
83		return Ok(None);
84	}
85	let own_registration_utxos = own_registrations.iter().map(|r| r.0.clone()).collect::<Vec<_>>();
86
87	let tx = Costs::calculate_costs(
88		|costs| {
89			register_tx(
90				&validator,
91				candidate_registration,
92				registration_utxo,
93				&own_registration_utxos,
94				costs,
95				&ctx,
96			)
97		},
98		client,
99	)
100	.await?;
101
102	let signed_tx = ctx.sign(&tx).to_bytes();
103	let result = client.submit_transaction(&signed_tx).await.map_err(|e| {
104		anyhow!(
105			"Submit candidate registration transaction request failed: {}, bytes: {}",
106			e,
107			hex::encode(tx.to_bytes())
108		)
109	})?;
110	let tx_id = result.transaction.id;
111	log::info!("✅ Transaction submitted. ID: {}", hex::encode(result.transaction.id));
112	await_tx.await_tx_output(client, UtxoId::new(tx_id, 0)).await?;
113
114	Ok(Some(McTxHash(result.transaction.id)))
115}

145pub async fn run_deregister<
146	C: QueryLedgerState + QueryNetwork + QueryUtxoByUtxoId + Transactions,
147	A: AwaitTx,
148>(
149	genesis_utxo: UtxoId,
150	payment_signing_key: &CardanoPaymentSigningKey,
151	stake_ownership_pub_key: StakePoolPublicKey,
152	client: &C,
153	await_tx: A,
154) -> anyhow::Result<Option<McTxHash>> {
155	let ctx = TransactionContext::for_payment_key(payment_signing_key, client).await?;
156	let validator = crate::scripts_data::registered_candidates_scripts(genesis_utxo)?;
157	let validator_address = validator.address_bech32(ctx.network)?;
158	let all_registration_utxos = client.query_utxos(&[validator_address]).await?;
159	let own_registrations = get_own_registrations(
160		payment_signing_key.to_pub_key_hash(),
161		stake_ownership_pub_key.clone(),
162		&all_registration_utxos,
163	);
164
165	if own_registrations.is_empty() {
166		log::info!("✅ Candidate is not registered.");
167		return Ok(None);
168	}
169
170	let own_registration_utxos = own_registrations.iter().map(|r| r.0.clone()).collect::<Vec<_>>();
171
172	let tx = Costs::calculate_costs(
173		|costs| deregister_tx(&validator, &own_registration_utxos, costs, &ctx),
174		client,
175	)
176	.await?;
177
178	let signed_tx = ctx.sign(&tx).to_bytes();
179	let result = client.submit_transaction(&signed_tx).await.map_err(|e| {
180		anyhow!(
181			"Submit candidate deregistration transaction request failed: {}, bytes: {}",
182			e,
183			hex::encode(tx.to_bytes())
184		)
185	})?;
186	let tx_id = result.transaction.id;
187	log::info!("✅ Transaction submitted. ID: {}", hex::encode(result.transaction.id));
188	await_tx.await_tx_output(client, UtxoId::new(tx_id, 0)).await?;
189
190	Ok(Some(McTxHash(result.transaction.id)))
191}

193fn get_own_registrations(
194	own_pkh: MainchainKeyHash,
195	spo_pub_key: StakePoolPublicKey,
196	validator_utxos: &[OgmiosUtxo],
197) -> Vec<(OgmiosUtxo, CandidateRegistration)> {
198	let mut own_registrations = Vec::new();
199	for validator_utxo in validator_utxos {
200		match get_candidate_registration(validator_utxo.clone()) {
201			Ok(candidate_registration) => {
202				if candidate_registration.stake_ownership.pub_key == spo_pub_key
203					&& candidate_registration.own_pkh == own_pkh
204				{
205					own_registrations.push((validator_utxo.clone(), candidate_registration.clone()))
206				}
207			},
208			Err(e) => log::debug!("Found invalid UTXO at validator address: {}", e),
209		}
210	}
211	own_registrations
212}

214fn get_candidate_registration(validator_utxo: OgmiosUtxo) -> anyhow::Result<CandidateRegistration> {
215	let datum = validator_utxo.datum.ok_or_else(|| anyhow!("UTXO does not have a datum"))?;
216	let datum_plutus_data = PlutusData::from_bytes(datum.bytes)
217		.map_err(|e| anyhow!("Could not decode datum of validator script: {}", e))?;
218	let register_validator_datum = RegisterValidatorDatum::try_from(datum_plutus_data)
219		.map_err(|e| anyhow!("Could not decode datum of validator script: {}", e))?;
220	Ok(register_validator_datum.into())
221}

223fn register_tx(
224	validator: &PlutusScript,
225	candidate_registration: &CandidateRegistration,
226	registration_utxo: &OgmiosUtxo,
227	own_registration_utxos: &[OgmiosUtxo],
228	costs: Costs,
229	ctx: &TransactionContext,
230) -> anyhow::Result<Transaction> {
231	let config = crate::csl::get_builder_config(ctx)?;
232	let mut tx_builder = TransactionBuilder::new(&config);
233
234	{
235		let mut inputs = TxInputsBuilder::new();
236		for own_registration_utxo in own_registration_utxos {
237			inputs.add_script_utxo_input(
238				own_registration_utxo,
239				validator,
240				&register_redeemer_data(),
241				&costs.get_one_spend(),
242			)?;
243		}
244		inputs.add_regular_inputs(&[registration_utxo.clone()])?;
245		tx_builder.set_inputs(&inputs);
246	}
247
248	{
249		let datum = candidate_registration_to_plutus_data(candidate_registration);
250		let amount_builder = TransactionOutputBuilder::new()
251			.with_address(&validator.address(ctx.network))
252			.with_plutus_data(&datum)
253			.next()?;
254		let output = amount_builder.with_minimum_ada(ctx)?.build()?;
255		tx_builder.add_output(&output)?;
256	}
257
258	Ok(tx_builder.balance_update_and_build(ctx)?)
259}

439	fn match_inputs(inputs: &TransactionInputs, payment_utxos: &[OgmiosUtxo]) -> Vec<OgmiosUtxo> {
440		inputs
441			.into_iter()
442			.map(|input| {
443				payment_utxos
444					.iter()
445					.find(|utxo| utxo.to_csl_tx_input() == *input)
446					.unwrap()
447					.clone()
448			})
449			.collect()
450	}

452	fn used_inputs_lovelace_equals_outputs_and_fee(tx: &Transaction, payment_utxos: &[OgmiosUtxo]) {
453		let used_inputs: Vec<OgmiosUtxo> = match_inputs(&tx.body().inputs(), payment_utxos);
454		let used_inputs_value: u64 = sum_lovelace(&used_inputs);
455		let outputs_lovelace_sum: u64 = tx
456			.body()
457			.outputs()
458			.into_iter()
459			.map(|output| {
460				let value: u64 = output.amount().coin().into();
461				value
462			})
463			.sum();
464		let fee: u64 = tx.body().fee().into();
465		// Used inputs are qual to the sum of the outputs plus the fee
466		assert_eq!(used_inputs_value, outputs_lovelace_sum + fee);
467	}
