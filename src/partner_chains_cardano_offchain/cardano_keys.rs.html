1use anyhow::anyhow;
2use cardano_serialization_lib::{PrivateKey, PublicKey};

5/// Signing (payment) key abstraction layer. Hides internal crytpo library details.
6/// It is either:
7/// * 32 bytes regural private key
8/// * 64 bytes extended private key
9pub struct CardanoPaymentSigningKey(pub(crate) PrivateKey);
10
11impl CardanoPaymentSigningKey {
12	pub fn from_extended_128_bytes(bytes: [u8; 128]) -> anyhow::Result<Self> {
13		// All 128 bytes are: 64 bytes of prefix, 32 bytes of verification key, 32 bytes of chain code
14		let prefix: [u8; 64] = bytes[0..64].try_into().unwrap();
15		Ok(Self(
16			PrivateKey::from_extended_bytes(&prefix)
17				.map_err(|e| anyhow!("Couldn't parse 128 bytes into a BIP32 Private Key: {e}"))?,
18		))
19	}
20
21	pub fn from_normal_bytes(bytes: [u8; 32]) -> Result<Self, anyhow::Error> {
22		Ok(Self(
23			PrivateKey::from_normal_bytes(&bytes)
24				.map_err(|e| anyhow!("Couldn't parse 32 bytes into a Private Key: {e}"))?,
25		))
26	}
27
28	pub fn sign(&self, message: &[u8]) -> Result<Vec<u8>, anyhow::Error> {
29		Ok(self.0.sign(message).to_bytes())
30	}
31
32	pub fn to_pub_key_hash(&self) -> MainchainKeyHash {
33		MainchainKeyHash(
34			self.0
35				.to_public()
36				.hash()
37				.to_bytes()
38				.as_slice()
39				.try_into()
40				.expect("CSL PublicKeyHash is 28 bytes"),
41		)
42	}
43
44	pub fn to_csl_pub_key(&self) -> PublicKey {
45		self.0.to_public()
46	}
47
48	pub fn to_bytes(&self) -> Vec<u8> {
49		self.0.as_bytes()
50	}
51}

53impl TryFrom<CardanoKeyFileContent> for CardanoPaymentSigningKey {
54	type Error = anyhow::Error;
55
56	fn try_from(key: CardanoKeyFileContent) -> Result<Self, Self::Error> {
57		let key_type = key.r#type.clone();
58		if key_type == "PaymentSigningKeyShelley_ed25519" {
59			Ok(CardanoPaymentSigningKey::from_normal_bytes(key.raw_key_bytes()?)?)
60		} else if key_type == "PaymentExtendedSigningKeyShelley_ed25519_bip32" {
61			Ok(CardanoPaymentSigningKey::from_extended_128_bytes(key.raw_key_bytes()?)?)
62		} else {
63			Err(anyhow!("Unsupported key type: {}. Expected a signing key", key_type))
64		}
65	}
66}
67
68impl Clone for CardanoPaymentSigningKey {
69	fn clone(&self) -> Self {
70		let bytes = self.0.as_bytes();
71		let private_key = if bytes.len() == 32 {
72			PrivateKey::from_normal_bytes(&bytes)
73				.expect("PrivateKey bytes are valid to clone the key")
74		} else {
75			PrivateKey::from_extended_bytes(&bytes)
76				.expect("PrivateKey bytes are valid to clone the key")
77		};
78		Self(private_key)
79	}
80}

82#[derive(serde::Serialize, serde::Deserialize, Debug, Clone)]
83#[serde(rename_all = "camelCase")]
84pub struct CardanoKeyFileContent {
85	pub r#type: String,
86	pub cbor_hex: String,
87}
88
89impl CardanoKeyFileContent {
90	pub fn parse_file(path: &str) -> anyhow::Result<Self> {
91		let file_content = std::fs::read_to_string(path)
92			.map_err(|e| anyhow!("Could not read Cardano key file at: {path}. Cause: {e}"))?;
93		serde_json::from_str::<Self>(&file_content)
94			.map_err(|e| anyhow!("{path} is not valid Cardano key JSON file. {e}"))
95	}
96	/// Parses raw bytes of 'cborHex' field of Cardano key file content.
97	/// Works for 32, 64 and 128 bytes hex strings (68, 132, 260 hex digits) - assumes CBOR prefix has 4 hex digits.
98	pub fn raw_key_bytes<const N: usize>(&self) -> anyhow::Result<[u8; N]> {
99		let (_cbor_prefix, vkey) = self.cbor_hex.split_at(4);
100		let bytes: [u8; N] = hex::decode(vkey)
101			.map_err(|err| {
102				anyhow!(
103					"Invalid cborHex value of Cardano key - not valid hex: {}\n{err:?}",
104					self.cbor_hex
105				)
106			})?
107			.try_into()
108			.map_err(|_| {
109				anyhow!(
110					"Invalid cborHex value of Cardano key - incorrect length: {}",
111					self.cbor_hex
112				)
113			})?;
114		Ok(bytes)
115	}
116}
