9pub(crate) fn init_governance_transaction(
10	governance_parameters: &MultiSigParameters,
11	genesis_utxo: OgmiosUtxo,
12	costs: Costs,
13	ctx: &TransactionContext,
14) -> anyhow::Result<Transaction> {
15	let multi_sig_policy = governance_parameters.as_simple_at_least_n().to_csl_native_script();
16	let version_oracle = version_oracle(genesis_utxo.utxo_id(), ctx.network)?;
17	let config = crate::csl::get_builder_config(ctx)?;
18	let mut tx_builder = TransactionBuilder::new(&config);
19
20	tx_builder.add_mint_one_script_token(
21		&version_oracle.policy,
22		&version_oracle_asset_name(),
23		&VersionOraclePolicyRedeemer::InitializeVersionOracle(
24			raw_scripts::ScriptId::GovernancePolicy.into(),
25			multi_sig_policy.hash().into(),
26		)
27		.into(),
28		&costs.get_mint(&version_oracle.policy.clone()),
29	)?;
30
31	tx_builder.add_output(&version_oracle_datum_output(
32		version_oracle.validator.clone(),
33		version_oracle.policy.clone(),
34		multi_sig_policy,
35		ctx.network,
36		ctx,
37	)?)?;
38
39	tx_builder.set_inputs(&{ TxInputsBuilder::with_regular_inputs(&[genesis_utxo])? });
40
41	Ok(tx_builder.balance_update_and_build(ctx)?)
42}
43
44fn version_oracle_asset_name() -> AssetName {
45	AssetName::new(b"Version oracle".to_vec()).expect("Constant asset name should work")
46}

48pub(crate) fn version_oracle_datum_output(
49	version_oracle_validator: PlutusScript,
50	version_oracle_policy: PlutusScript,
51	multi_sig_policy: NativeScript,
52	network: NetworkIdKind,
53	tx_context: &TransactionContext,
54) -> anyhow::Result<cardano_serialization_lib::TransactionOutput> {
55	let datum: PlutusData = VersionOracleDatum {
56		version_oracle: raw_scripts::ScriptId::GovernancePolicy.into(),
57		currency_symbol: version_oracle_policy.policy_id().0.into(),
58	}
59	.into();
60
61	let amount_builder = TransactionOutputBuilder::new()
62		.with_address(&version_oracle_validator.address(network))
63		.with_plutus_data(&datum)
64		.with_script_ref(&ScriptRef::new_native_script(&multi_sig_policy))
65		.next()?;
66
67	let ma = MultiAsset::new()
68		.with_asset_amount(&version_oracle_policy.asset(version_oracle_asset_name())?, 1u64)?;
69
70	let output = amount_builder.with_minimum_ada_and_asset(&ma, tx_context)?.build()?;
71	Ok(output)
72}
