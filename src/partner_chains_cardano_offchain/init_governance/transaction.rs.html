9pub(crate) fn init_governance_transaction(
10	governance_parameters: &MultiSigParameters,
11	genesis_utxo: OgmiosUtxo,
12	costs: Costs,
13	ctx: &TransactionContext,
14) -> anyhow::Result<Transaction> {
15	let multi_sig_policy = governance_parameters.as_simple_at_least_n().to_csl_native_script();
16	let version_oracle = version_oracle(genesis_utxo.utxo_id(), ctx.network)?;
17	let config = crate::csl::get_builder_config(ctx)?;
18	let mut tx_builder = TransactionBuilder::new(&config);
19
20	tx_builder.add_mint_one_script_token(
21		&version_oracle.policy,
22		&version_oracle_asset_name(),
23		&mint_redeemer(&multi_sig_policy),
24		&costs.get_mint(&version_oracle.policy.clone()),
25	)?;
26
27	tx_builder.add_output(&version_oracle_datum_output(
28		version_oracle.validator.clone(),
29		version_oracle.policy.clone(),
30		multi_sig_policy,
31		ctx.network,
32		ctx,
33	)?)?;
34
35	tx_builder.set_inputs(&{ TxInputsBuilder::with_regular_inputs(&[genesis_utxo])? });
36
37	Ok(tx_builder.balance_update_and_build(ctx)?)
38}
39
40fn version_oracle_asset_name() -> AssetName {
41	AssetName::new(b"Version oracle".to_vec()).expect("Constant asset name should work")
42}
43
44fn mint_redeemer(multi_sig_policy: &NativeScript) -> PlutusData {
45	PlutusData::new_constr_plutus_data(&ConstrPlutusData::new(&0u64.into(), &{
46		let mut list = PlutusList::new();
47		list.add(&PlutusData::new_integer(
48			&(raw_scripts::ScriptId::GovernancePolicy as u32).into(),
49		));
50		list.add(&PlutusData::new_bytes(multi_sig_policy.hash().to_bytes()));
51		list
52	}))
53}

55pub(crate) fn version_oracle_datum_output(
56	version_oracle_validator: PlutusScript,
57	version_oracle_policy: PlutusScript,
58	multi_sig_policy: NativeScript,
59	network: NetworkIdKind,
60	tx_context: &TransactionContext,
61) -> anyhow::Result<cardano_serialization_lib::TransactionOutput> {
62	let datum: PlutusData = VersionOracleDatum {
63		version_oracle: raw_scripts::ScriptId::GovernancePolicy as u32,
64		currency_symbol: version_oracle_policy.policy_id().0,
65	}
66	.into();
67
68	let amount_builder = TransactionOutputBuilder::new()
69		.with_address(&version_oracle_validator.address(network))
70		.with_plutus_data(&datum)
71		.with_script_ref(&ScriptRef::new_native_script(&multi_sig_policy))
72		.next()?;
73
74	let ma = MultiAsset::new()
75		.with_asset_amount(&version_oracle_policy.asset(version_oracle_asset_name())?, 1u64)?;
76
77	let output = amount_builder.with_minimum_ada_and_asset(&ma, tx_context)?.build()?;
78	Ok(output)
79}
