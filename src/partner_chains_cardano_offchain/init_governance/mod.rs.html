22pub trait InitGovernance {
23	/// Initializes goveranance mechanism with the authority being `governance_authority`,
24	/// for the chain identified by `genesis_utxo_id`.
25	#[allow(async_fn_in_trait)]
26	async fn init_governance(
27		&self,
28		governance_parameters: &MultiSigParameters,
29		payment_key: &CardanoPaymentSigningKey,
30		genesis_utxo_id: UtxoId,
31	) -> Result<McTxHash, OffchainError>;
32}
33
34impl<T> InitGovernance for T
35where
36	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,
37{
38	async fn init_governance(
39		&self,
40		governance_parameters: &MultiSigParameters,
41		payment_key: &CardanoPaymentSigningKey,
42		genesis_utxo_id: UtxoId,
43	) -> Result<McTxHash, OffchainError> {
44		run_init_governance(
45			governance_parameters,
46			payment_key,
47			Some(genesis_utxo_id),
48			self,
49			FixedDelayRetries::five_minutes(),
50		)
51		.await
52		.map(|result| result.tx_hash)
53		.map_err(|e| OffchainError::InternalError(e.to_string()))
54	}
55}

58pub struct InitGovernanceResult {

60	pub genesis_utxo: UtxoId,
61}

63pub async fn run_init_governance<
64	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,
65	A: AwaitTx,
66>(
67	governance_parameters: &MultiSigParameters,
68	payment_key: &CardanoPaymentSigningKey,
69	genesis_utxo_id: Option<UtxoId>,
70	client: &T,
71	await_tx: A,
72) -> anyhow::Result<InitGovernanceResult> {
73	let ctx = crate::csl::TransactionContext::for_payment_key(payment_key, client).await?;
74
75	let own_address = key_hash_address(&ctx.payment_key_hash(), ctx.network);
76	log::info!("‚úâÔ∏è Submitter address: {}", own_address.to_bech32(None).unwrap());
77
78	let own_utxos = ctx.payment_key_utxos.clone();
79	log::info!("üí± {} UTXOs available", own_utxos.len());
80
81	let genesis_utxo = match genesis_utxo_id {
82		None => {
83			log::info!("‚öôÔ∏è No genesis UTXO provided, will select one automatically...");
84			let utxo = own_utxos.first().ok_or(anyhow!("No UTXOs to choose from"))?.clone();
85			log::info!("‚òëÔ∏è UTXO selected: {}", utxo);
86			utxo
87		},
88		Some(utxo_id) => own_utxos
89			.iter()
90			.find(|utxo| utxo.transaction.id == utxo_id.tx_hash.0 && utxo.index == utxo_id.index.0)
91			.ok_or(anyhow!("Could not find genesis UTXO: {utxo_id}"))?
92			.clone(),
93	};
94
95	let tx = Costs::calculate_costs(
96		|costs| {
97			transaction::init_governance_transaction(
98				governance_parameters,
99				genesis_utxo.clone(),
100				costs,
101				&ctx,
102			)
103		},
104		client,
105	)
106	.await?;
107
108	let signed_transaction = ctx.sign(&tx);
109
110	let result = client.submit_transaction(&signed_transaction.to_bytes()).await?;
111	let tx_id = result.transaction.id;
112	log::info!("‚úÖ Transaction submitted. ID: {}", hex::encode(tx_id));
113	await_tx.await_tx_output(client, UtxoId::new(tx_id, 0)).await?;
114	Ok(InitGovernanceResult { tx_hash: McTxHash(tx_id), genesis_utxo: genesis_utxo.utxo_id() })
115}
