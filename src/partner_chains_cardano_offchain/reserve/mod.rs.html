37impl ReserveData {
38	pub(crate) async fn get<T: QueryLedgerState>(
39		genesis_utxo: UtxoId,
40		ctx: &TransactionContext,
41		client: &T,
42	) -> Result<Self, anyhow::Error> {
43		let version_oracle = scripts_data::version_oracle(genesis_utxo, ctx.network)?;
44		let auth_policy_version_utxo =
45			find_script_utxo(
46				raw_scripts::ScriptId::ReserveAuthPolicy as u32,
47				&version_oracle,
48				ctx,
49				client,
50			)
51			.await?
52			.ok_or_else(|| {
53				anyhow!("Reserve Auth Version Utxo not found, is the Reserve Token Management initialized?")
54			})?;
55		let validator_version_utxo = find_script_utxo(
56			raw_scripts::ScriptId::ReserveValidator as u32,
57			&version_oracle,
58			ctx,
59			client,
60		)
61		.await?
62		.ok_or_else(|| {
63			anyhow!("Reserve Validator Version Utxo not found, is the Reserve Token Management initialized?")
64		})?;
65		let illiquid_circulation_supply_validator_version_utxo = find_script_utxo(
66			raw_scripts::ScriptId::IlliquidCirculationSupplyValidator as u32,
67			&version_oracle,
68			ctx,
69			client,
70		)
71		.await?
72		.ok_or_else(|| {
73			anyhow!("Illiquid Circulation Supply Validator Version Utxo not found, is the Reserve Token Management initialized?")
74		})?;
75		let scripts = scripts_data::reserve_scripts(genesis_utxo, ctx.network)?;
76		Ok(ReserveData {
77			scripts,
78			auth_policy_version_utxo,
79			validator_version_utxo,
80			illiquid_circulation_supply_validator_version_utxo,
81		})
82	}
83
84	pub(crate) async fn get_reserve_utxo<T: QueryLedgerState>(
85		&self,
86		ctx: &TransactionContext,
87		client: &T,
88	) -> Result<ReserveUtxo, anyhow::Error> {
89		let validator_address = self.scripts.validator.address(ctx.network).to_bech32(None)?;
90		let validator_utxos = client.query_utxos(&[validator_address]).await?;
91
92		let auth_token_asset_id = AssetId {
93			policy_id: self.scripts.auth_policy.policy_id(),
94			asset_name: AssetName::empty(),
95		};
96
97		let (reserve_utxo, reserve_settings) = validator_utxos
98			.into_iter()
99			.find_map(|utxo| {
100				if utxo.get_asset_amount(&auth_token_asset_id) != 1u64 {
101					return None;
102				}
103				utxo.get_plutus_data()
104					.and_then(|d| ReserveDatum::try_from(d).ok())
105					.map(|d| (utxo, d))
106			})
107			.ok_or_else(|| {
108				anyhow!("Reserve Utxo not found, is the Reserve Token Management initialized?")
109			})?;
110
111		Ok(ReserveUtxo { utxo: reserve_utxo, datum: reserve_settings })
112	}
113}

120pub(crate) fn reserve_utxo_input_with_validator_script_reference(
121	reserve_utxo: &OgmiosUtxo,
122	reserve: &ReserveData,
123	redeemer: ReserveRedeemer,
124	cost: &ExUnits,
125) -> Result<TxInputsBuilder, JsError> {
126	let mut inputs = TxInputsBuilder::new();
127	let input = reserve_utxo.to_csl_tx_input();
128	let amount = reserve_utxo.value.to_csl()?;
129	let script = &reserve.scripts.validator;
130	let witness = PlutusWitness::new_with_ref_without_datum(
131		&PlutusScriptSource::new_ref_input(
132			&script.csl_script_hash(),
133			&reserve.validator_version_utxo.to_csl_tx_input(),
134			&script.language,
135			script.bytes.len(),
136		),
137		&Redeemer::new(&RedeemerTag::new_spend(), &0u32.into(), &redeemer.into(), cost),
138	);
139	inputs.add_plutus_script_input(&witness, &input, &amount);
140	Ok(inputs)
141}
