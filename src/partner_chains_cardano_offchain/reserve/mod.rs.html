40	pub(crate) async fn get<T: QueryLedgerState>(
41		genesis_utxo: UtxoId,
42		ctx: &TransactionContext,
43		client: &T,
44	) -> Result<Self, anyhow::Error> {
45		let version_oracle = scripts_data::version_oracle(genesis_utxo, ctx.network)?;
46		let auth_policy_version_utxo = find_script_utxo(
47			raw_scripts::ScriptId::ReserveAuthPolicy as u32,
48			&version_oracle,
49			ctx,
50			client,
51		)
52		.await?
53		.ok_or_else(|| {
54			anyhow!(
55				"Reserve Auth Version Utxo not found, is the Reserve Token Management initialized?"
56			)
57		})?;
58		let validator_version_utxo = find_script_utxo(
59			raw_scripts::ScriptId::ReserveValidator as u32,
60			&version_oracle,
61			ctx,
62			client,
63		)
64		.await?
65		.ok_or_else(|| {
66			anyhow!(
67				"Reserve Validator Version Utxo not found, is the Reserve Token Management initialized?"
68			)
69		})?;
70		let illiquid_circulation_supply_validator_version_utxo = find_script_utxo(
71			raw_scripts::ScriptId::IlliquidCirculationSupplyValidator as u32,
72			&version_oracle,
73			ctx,
74			client,
75		)
76		.await?
77		.ok_or_else(|| {
78			anyhow!("Illiquid Circulation Supply Validator Version Utxo not found, is the Reserve Token Management initialized?")
79		})?;
80		let illiquid_circulation_supply_authority_token_policy_version_utxo = find_script_utxo(
81			raw_scripts::ScriptId::IlliquidCirculationSupplyAuthorityTokenPolicy as u32,
82			&version_oracle,
83			ctx,
84			client,
85		)
86		.await?
87		.ok_or_else(|| {
88			anyhow!("Illiquid Circulation Supply Authority Token Policy Version Utxo not found, is the Reserve Token Management initialized?")
89		})?;
90		let scripts = scripts_data::reserve_scripts(genesis_utxo, ctx.network)?;
91		Ok(ReserveData {
92			scripts,
93			auth_policy_version_utxo,
94			validator_version_utxo,
95			illiquid_circulation_supply_validator_version_utxo,
96			illiquid_circulation_supply_authority_token_policy_version_utxo,
97		})
98	}

100	pub(crate) async fn get_reserve_utxo<T: QueryLedgerState>(
101		&self,
102		ctx: &TransactionContext,
103		client: &T,
104	) -> Result<ReserveUtxo, anyhow::Error> {
105		let validator_address = self.scripts.validator.address(ctx.network).to_bech32(None)?;
106		let validator_utxos = client.query_utxos(&[validator_address]).await?;
107
108		let auth_token_asset_id = AssetId {
109			policy_id: self.scripts.auth_policy.policy_id(),
110			asset_name: AssetName::empty(),
111		};
112
113		let (reserve_utxo, reserve_settings) = validator_utxos
114			.into_iter()
115			.find_map(|utxo| {
116				if utxo.get_asset_amount(&auth_token_asset_id) != 1u64 {
117					return None;
118				}
119				utxo.get_plutus_data()
120					.and_then(|d| ReserveDatum::try_from(d).ok())
121					.map(|d| (utxo, d))
122			})
123			.ok_or_else(|| {
124				anyhow!("Reserve Utxo not found, is the Reserve Token Management initialized?")
125			})?;
126
127		Ok(ReserveUtxo { utxo: reserve_utxo, datum: reserve_settings })
128	}

130	pub(crate) async fn get_illiquid_circulation_supply_utxo<T: QueryLedgerState>(
131		&self,
132		ctx: &TransactionContext,
133		client: &T,
134	) -> Result<OgmiosUtxo, anyhow::Error> {
135		self.get_illiquid_circulation_supply_utxos(ctx, client).await?.first().cloned()
136			.ok_or_else(|| {
137				anyhow!("Could not find any UTXO with ICS Auth token at ICS Validator, is the Reserve Token Management initialized?")
138			})
139	}

141	pub(crate) async fn get_illiquid_circulation_supply_utxos<T: QueryLedgerState>(
142		&self,
143		ctx: &TransactionContext,
144		client: &T,
145	) -> Result<Vec<OgmiosUtxo>, anyhow::Error> {
146		let validator_address = self
147			.scripts
148			.illiquid_circulation_supply_validator
149			.address(ctx.network)
150			.to_bech32(None)?;
151		let validator_utxos = client.query_utxos(&[validator_address]).await?;
152
153		let auth_token_asset_id = AssetId {
154			policy_id: self.scripts.illiquid_circulation_supply_auth_token_policy.policy_id(),
155			asset_name: AssetName::empty(),
156		};
157
158		Ok(validator_utxos
159			.into_iter()
160			.filter(|utxo| utxo.get_asset_amount(&auth_token_asset_id) == 1u64)
161			.collect())
162	}

174pub(crate) fn reserve_utxo_input_with_validator_script_reference(
175	reserve_utxo: &OgmiosUtxo,
176	reserve: &ReserveData,
177	redeemer: ReserveRedeemer,
178	cost: &ExUnits,
179) -> Result<TxInputsBuilder, JsError> {
180	let mut inputs = TxInputsBuilder::new();
181	add_reserve_utxo_input_with_validator_script_reference(
182		&mut inputs,
183		reserve_utxo,
184		reserve,
185		redeemer,
186		cost,
187	)?;
188	Ok(inputs)
189}

191pub(crate) fn add_reserve_utxo_input_with_validator_script_reference(
192	inputs: &mut TxInputsBuilder,
193	reserve_utxo: &OgmiosUtxo,
194	reserve: &ReserveData,
195	redeemer: ReserveRedeemer,
196	cost: &ExUnits,
197) -> Result<(), JsError> {
198	let input = reserve_utxo.to_csl_tx_input();
199	let amount = reserve_utxo.value.to_csl()?;
200	let script = &reserve.scripts.validator;
201	let witness = PlutusWitness::new_with_ref_without_datum(
202		&PlutusScriptSource::new_ref_input(
203			&script.csl_script_hash(),
204			&reserve.validator_version_utxo.to_csl_tx_input(),
205			&script.language,
206			script.bytes.len(),
207		),
208		&Redeemer::new(&RedeemerTag::new_spend(), &0u32.into(), &redeemer.into(), cost),
209	);
210	inputs.add_plutus_script_input(&witness, &input, &amount);
211	Ok(())
212}

214pub(crate) fn add_ics_utxo_input_with_validator_script_reference(
215	inputs: &mut TxInputsBuilder,
216	ics_utxo: &OgmiosUtxo,
217	reserve: &ReserveData,
218	cost: &ExUnits,
219) -> Result<(), JsError> {
220	let input = ics_utxo.to_csl_tx_input();
221	let amount = ics_utxo.value.to_csl()?;
222	let script = &reserve.scripts.illiquid_circulation_supply_validator;
223	let witness = PlutusWitness::new_with_ref_without_datum(
224		&PlutusScriptSource::new_ref_input(
225			&script.csl_script_hash(),
226			&reserve.illiquid_circulation_supply_validator_version_utxo.to_csl_tx_input(),
227			&script.language,
228			script.bytes.len(),
229		),
230		&Redeemer::new(
231			&RedeemerTag::new_spend(),
232			&0u32.into(),
233			&PlutusData::new_integer(&BigInt::zero()),
234			cost,
235		),
236	);
237	inputs.add_plutus_script_input(&witness, &input, &amount);
238	Ok(())
