1//! Initialization of the reserve management is execution of three similar transaction to
2//! initialize three scripts: Rerserve Management Validator, Reserve Management Policy and
3//! Illiquid Circulation Supply Validator.
4//!
5//! Transaction for each of these scripts should have:
6//! * an output to Version Oracle Validator address that should:
7//! * * have script reference with the script being initialized attached, script should be applied with Version Oracle Policy Id
8//! * * contain 1 token of Version Oracle Policy with "Version oracle" asset name, minted in this transaction
9//! * * * mint redeemer should be Constr(1, [Int: SCRIPT_ID, Bytes: Applied Script Hash])
10//! * * have Plutus Data that is [Int: SCRIPT_ID, Bytes: Version Oracle Policy Id]
11//! * an output to the current governance (holder of governance token) that should:
12//! * * contain a new Goveranance Policy token, minted in this transaction,
13//! * * * mint redeemer should be empty contructor Plutus Data
14//! * a script reference input of the current Goveranance UTXO
15//! * signature of the current goveranance
16

45pub async fn init_reserve_management<
46	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,
47	A: AwaitTx,
48>(
49	genesis_utxo: UtxoId,
50	payment_key: &CardanoPaymentSigningKey,
51	client: &T,
52	await_tx: &A,
53) -> anyhow::Result<Vec<MultiSigSmartContractResult>> {
54	let reserve_validator = ScriptData::new(
55		"Reserve Management Validator",
56		RESERVE_VALIDATOR.to_vec(),
57		ScriptId::ReserveValidator,
58	);
59	let reserve_policy = ScriptData::new(
60		"Reserve Management Policy",
61		RESERVE_AUTH_POLICY.to_vec(),
62		ScriptId::ReserveAuthPolicy,
63	);
64	let ics_validator = ScriptData::new(
65		"Illiquid Circulation Validator",
66		ILLIQUID_CIRCULATION_SUPPLY_VALIDATOR.to_vec(),
67		ScriptId::IlliquidCirculationSupplyValidator,
68	);
69	Ok(vec![
70		initialize_script(reserve_validator, genesis_utxo, payment_key, client, await_tx).await?,
71		initialize_script(reserve_policy, genesis_utxo, payment_key, client, await_tx).await?,
72		initialize_script(ics_validator, genesis_utxo, payment_key, client, await_tx).await?,
73	]
74	.into_iter()
75	.flatten()
76	.collect())
77}

79struct ScriptData {
80	name: String,
81	plutus_script: PlutusScript,
82	id: u32,
83}
84
85impl ScriptData {
86	fn new(name: &str, raw_bytes: Vec<u8>, id: ScriptId) -> Self {
87		let plutus_script = PlutusScript::from_wrapped_cbor(&raw_bytes, Language::new_plutus_v2())
88			.expect("Plutus script should be valid");
89		Self { name: name.to_string(), plutus_script, id: id as u32 }
90	}
91
92	fn applied_plutus_script(
93		&self,
94		version_oracle: &VersionOracleData,
95	) -> Result<PlutusScript, JsError> {
96		self.plutus_script
97			.clone()
98			.apply_uplc_data(version_oracle.policy_id_as_plutus_data())
99			.map_err(|e| JsError::from_str(&e.to_string()))
100	}
101}
102
103async fn initialize_script<
104	T: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId,
105	A: AwaitTx,
106>(
107	script: ScriptData,
108	genesis_utxo: UtxoId,
109	payment_key: &CardanoPaymentSigningKey,
110	client: &T,
111	await_tx: &A,
112) -> anyhow::Result<Option<MultiSigSmartContractResult>> {
113	let payment_ctx = TransactionContext::for_payment_key(payment_key, client).await?;
114	let governance = GovernanceData::get(genesis_utxo, client).await?;
115	let version_oracle = scripts_data::version_oracle(genesis_utxo, payment_ctx.network)?;
116
117	if script_is_initialized(&script, &version_oracle, &payment_ctx, client).await? {
118		log::info!("Script '{}' is already initialized", script.name);
119		return Ok(None);
120	}
121	Ok(Some(
122		submit_or_create_tx_to_sign(
123			&governance,
124			payment_ctx,
125			|costs, ctx| init_script_tx(&script, &governance, &version_oracle, costs, &ctx),
126			"Init Reserve Management",
127			client,
128			await_tx,
129		)
130		.await?,
131	))
132}
133
134fn init_script_tx(
135	script: &ScriptData,
136	governance: &GovernanceData,
137	version_oracle: &VersionOracleData,
138	costs: Costs,
139	ctx: &TransactionContext,
140) -> anyhow::Result<Transaction> {
141	let mut tx_builder = TransactionBuilder::new(&get_builder_config(ctx)?);
142	let applied_script = script.applied_plutus_script(version_oracle)?;
143	{
144		let witness = PlutusData::new_constr_plutus_data(&ConstrPlutusData::new(
145			&BigNum::one(),
146			&version_oracle_plutus_list(script.id, &applied_script.script_hash()),
147		));
148		tx_builder.add_mint_one_script_token(
149			&version_oracle.policy,
150			&version_oracle_asset_name(),
151			&witness,
152			&costs.get_mint(&version_oracle.policy.clone()),
153		)?;
154	}
155	{
156		let script_ref = ScriptRef::new_plutus_script(&applied_script.to_csl());
157		let amount_builder = TransactionOutputBuilder::new()
158			.with_address(&version_oracle.validator.address(ctx.network))
159			.with_plutus_data(&PlutusData::new_list(&version_oracle_plutus_list(
160				script.id,
161				&version_oracle.policy.script_hash(),
162			)))
163			.with_script_ref(&script_ref)
164			.next()?;
165		let ma = MultiAsset::new()
166			.with_asset_amount(&version_oracle.policy.asset(version_oracle_asset_name())?, 1u64)?;
167		let output = amount_builder.with_minimum_ada_and_asset(&ma, ctx)?.build()?;
168		tx_builder.add_output(&output)?;
169	}
170	// Mint governance token
171	let gov_tx_input = governance.utxo_id_as_tx_input();
172	tx_builder.add_mint_one_script_token_using_reference_script(
173		&governance.policy.script(),
174		&gov_tx_input,
175		&costs,
176	)?;
177	Ok(tx_builder.balance_update_and_build(ctx)?.remove_native_script_witnesses())
178}
179
180fn version_oracle_asset_name() -> AssetName {
181	AssetName::new(b"Version oracle".to_vec()).unwrap()
182}

184fn version_oracle_plutus_list(script_id: u32, script_hash: &[u8]) -> PlutusList {
185	let mut list = PlutusList::new();
186	list.add(&PlutusData::new_integer(&script_id.into()));
187	list.add(&PlutusData::new_bytes(script_hash.to_vec()));
188	list
189}

191// There exist UTXO at Version Oracle Validator with Datum that contains
192// * script id of the script being initialized
193// * Version Oracle Policy Id
194async fn script_is_initialized<T: QueryLedgerState>(
195	script: &ScriptData,
196	version_oracle: &VersionOracleData,
197	ctx: &TransactionContext,
198	client: &T,
199) -> Result<bool, anyhow::Error> {
200	Ok(find_script_utxo(script.id, version_oracle, ctx, client).await?.is_some())
201}
202
203// Finds an UTXO at Version Oracle Validator with Datum that contains
204// * given script id
205// * Version Oracle Policy Id
206pub(crate) async fn find_script_utxo<T: QueryLedgerState>(
207	script_id: u32,
208	version_oracle: &VersionOracleData,
209	ctx: &TransactionContext,
210	client: &T,
211) -> Result<Option<OgmiosUtxo>, anyhow::Error> {
212	let validator_address = version_oracle.validator.address(ctx.network).to_bech32(None)?;
213	let validator_utxos = client.query_utxos(&[validator_address]).await?;
214	// Decode datum from utxos and check if it contains script id
215	Ok(validator_utxos.into_iter().find(|utxo| {
216		utxo.get_plutus_data()
217			.and_then(|data| TryInto::<VersionOracleDatum>::try_into(data).ok())
218			.is_some_and(|datum| {
219				datum.version_oracle == script_id
220					&& datum.currency_symbol == version_oracle.policy.script_hash()
221			})
222	}))
223}

246		let outputs = make_init_script_tx().body().outputs();
247		let voo = outputs
248			.into_iter()
249			.find(|o| o.address() == version_oracle_validator_address())
250			.expect("Init Script Transaction should have output to Version Oracle Validator");
251		let voo_script_ref = voo
252			.script_ref()
253			.expect("Version Oracle Validator output should have script reference");
254		let voo_plutus_script = voo_script_ref

261				.unwrap()

263		);
264		let amount = voo
265			.amount()
266			.multiasset()
267			.and_then(|ma| ma.get(&version_oracle_policy_csl_script_hash()))
268			.and_then(|vo_ma| vo_ma.get(&AssetName::new(b"Version oracle".to_vec()).unwrap()))
269			.expect("Version Oracle Validator output has a token of Version Oracle Policy");
270
271		assert_eq!(amount, 1u64.into());
272
273		let voo_plutus_data = voo
274			.plutus_data()
275			.and_then(|pd| pd.as_list())
276			.expect("Version Oracle Validator output should have Plutus Data of List type");
277		assert_eq!(
278			voo_plutus_data.get(0),
279			PlutusData::new_integer(&test_initialized_script().id.into())
280		);
281		assert_eq!(
