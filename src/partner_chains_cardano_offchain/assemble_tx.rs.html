31pub async fn assemble_tx<
32	C: QueryLedgerState + QueryNetwork + Transactions + QueryUtxoByUtxoId,
33	A: AwaitTx,
34>(
35	transaction: Transaction,
36	witnesses: Vec<Vkeywitness>,
37	ogmios_client: &C,
38	await_tx: &A,
39) -> anyhow::Result<McTxHash> {
40	let mut witness_set = transaction.witness_set();
41
42	let mut vk = witness_set.vkeys().unwrap_or_else(Vkeywitnesses::new);
43
44	for w in witnesses.iter() {
45		vk.add(w);
46	}
47	witness_set.set_vkeys(&vk);
48
49	let new_tx = Transaction::new(&transaction.body(), &witness_set, transaction.auxiliary_data());
50
51	let res = ogmios_client.submit_transaction(&new_tx.to_bytes()).await.map_err(|e| {
52		anyhow!(
53			"Submit assembled transaction request failed: {}, bytes: {}",
54			e,
55			hex::encode(new_tx.to_bytes())
56		)
57	})?;
58	let tx_id = McTxHash(res.transaction.id);
59	log::info!("Transaction submitted: {}", hex::encode(tx_id.0));
60
61	await_tx.await_tx_output(ogmios_client, UtxoId::new(tx_id.0, 0)).await?;
62
63	Ok(tx_id)
64}
