25pub(crate) fn plutus_script_hash(script_bytes: &[u8], language: Language) -> [u8; 28] {
26	// Before hashing the script, we need to prepend with byte denoting the language.
27	let mut buf: Vec<u8> = vec![language_to_u8(language)];
28	buf.extend(script_bytes);
29	sidechain_domain::crypto::blake2b(buf.as_slice())
30}

83/// Creates a CSL [`TransactionBuilderConfig`] for given [`ProtocolParametersResponse`].
84/// This function is not unit-testable because [`TransactionBuilderConfig`] has no public getters.
85pub(crate) fn get_builder_config(
86	context: &TransactionContext,
87) -> Result<TransactionBuilderConfig, JsError> {
88	let protocol_parameters = &context.protocol_parameters;
89	TransactionBuilderConfigBuilder::new()
90		.fee_algo(&linear_fee(protocol_parameters))
91		.pool_deposit(&protocol_parameters.stake_pool_deposit.to_csl()?.coin())
92		.key_deposit(&protocol_parameters.stake_credential_deposit.to_csl()?.coin())
93		.max_value_size(protocol_parameters.max_value_size.bytes)
94		.max_tx_size(protocol_parameters.max_transaction_size.bytes)
95		.ex_unit_prices(&ExUnitPrices::new(
96			&ratio_to_unit_interval(&protocol_parameters.script_execution_prices.memory),
97			&ratio_to_unit_interval(&protocol_parameters.script_execution_prices.cpu),
98		))
99		.coins_per_utxo_byte(&protocol_parameters.min_utxo_deposit_coefficient.into())
100		.ref_script_coins_per_byte(&convert_reference_script_costs(
101			&protocol_parameters.min_fee_reference_scripts.clone(),
102		)?)
103		.deduplicate_explicit_ref_inputs_with_regular_inputs(true)
104		.build()
105}

107fn linear_fee(protocol_parameters: &ProtocolParametersResponse) -> LinearFee {
108	let constant: BigNum = protocol_parameters.min_fee_constant.lovelace.into();
109	LinearFee::new(&protocol_parameters.min_fee_coefficient.into(), &constant)
110}

120impl OgmiosValueExt for OgmiosValue {
121	/// Coverts ogmios value to CSL value.
122	/// It could fail if the input contains negative values, for example ogmios values representing burn.
123	fn to_csl(&self) -> Result<Value, JsError> {
124		if !self.native_tokens.is_empty() {
125			let mut multiasset = MultiAsset::new();
126			for (policy_id, assets) in self.native_tokens.iter() {
127				let mut csl_assets = Assets::new();
128				for asset in assets.iter() {
129					let asset_name = AssetName::new(asset.name.clone()).map_err(|e| {
130						JsError::from_str(&format!(
131							"Could not convert Ogmios UTXO value, asset name is invalid: '{}'",
132							e
133						))
134					})?;
135					csl_assets.insert(&asset_name, &asset.amount.into());
136				}
137				multiasset.insert(&ScriptHash::from(*policy_id), &csl_assets);
138			}
139			Ok(Value::new_with_assets(&self.lovelace.into(), &multiasset))
140		} else {
141			Ok(Value::new(&self.lovelace.into()))
142		}
143	}
144}

155pub(crate) fn convert_reference_script_costs(
156	costs: &ReferenceScriptsCosts,
157) -> Result<UnitInterval, JsError> {
158	let r = Ratio::<u64>::from_f64(costs.base).ok_or_else(|| {
159		JsError::from_str(&format!("Failed to decode cost base {} as a u64 ratio", costs.base))
160	})?;
161	let numerator = BigNum::from(*r.numer());
162	let denominator = BigNum::from(*r.denom());
163	Ok(UnitInterval::new(&numerator, &denominator))
164}

198pub trait CostStore {
199	fn get_mint(&self, script: &PlutusScript) -> ExUnits;
200	fn get_spend(&self, spend_ix: u32) -> ExUnits;
201	fn get_one_spend(&self) -> ExUnits;
202	fn get_spend_indices(&self) -> Vec<u32>;
203}
204
205impl CostStore for Costs {
206	fn get_mint(&self, script: &PlutusScript) -> ExUnits {
207		match self {
208			Costs::ZeroCosts => zero_ex_units(),
209			Costs::Costs(cost_lookup) => cost_lookup
210				.mints
211				.get(&script.csl_script_hash())
212				.expect("get_mint should not be called with an unknown script")
213				.clone(),
214		}
215	}
216	fn get_spend(&self, spend_ix: u32) -> ExUnits {
217		match self {
218			Costs::ZeroCosts => zero_ex_units(),
219			Costs::Costs(cost_lookup) => cost_lookup
220				.spends
221				.get(&spend_ix)
222				.expect("get_spend should not be called with an unknown spend index")
223				.clone(),
224		}
225	}
226	fn get_one_spend(&self) -> ExUnits {
227		match self {
228			Costs::ZeroCosts => zero_ex_units(),
229			Costs::Costs(cost_lookup) => {
230				match cost_lookup.spends.values().collect::<Vec<_>>()[..] {
231					[x] => x.clone(),
232					_ => panic!(
233						"get_one_spend should only be called when exacly one spend is expected to be present"
234					),
235				}
236			},
237		}
238	}
239	fn get_spend_indices(&self) -> Vec<u32> {
240		match self {
241			Costs::ZeroCosts => vec![],
242			Costs::Costs(cost_lookup) => cost_lookup.spends.keys().cloned().collect(),
243		}
244	}
245}

254
255	pub async fn calculate_costs<T: Transactions, F>(
256		make_tx: F,
257		client: &T,
258	) -> anyhow::Result<Transaction>
259	where
260		F: Fn(Costs) -> anyhow::Result<Transaction>,
261	{
262		let tx = make_tx(Costs::ZeroCosts)?;
263		// stage 1
264		let costs = Self::from_ogmios(&tx, client).await?;
265
266		let tx = make_tx(costs)?;
267		// stage 2
268		let costs = Self::from_ogmios(&tx, client).await?;
269
270		make_tx(costs)
271	}
272
273	async fn from_ogmios<T: Transactions>(tx: &Transaction, client: &T) -> anyhow::Result<Costs> {
274		let evaluate_response = client
275			.evaluate_transaction(&tx.to_bytes())
276			.await
277			.context("calculate_costs received Ogmios error from evaluate_transaction")?;
278
279		let mut mints = HashMap::new();
280		let mut spends = HashMap::new();
281		for er in evaluate_response {
282			match er.validator.purpose.as_str() {
283				"mint" => {
284					mints.insert(
285						tx.body()
286							.mint()
287							.expect(
288								"tx.body.mint() should not be empty if we received a 'mint' response from Ogmios",
289							)
290							.keys()
291							.get(er.validator.index as usize),
292						ex_units_from_response(er),
293					);
294				},
295				"spend" => {
296					spends.insert(er.validator.index, ex_units_from_response(er));
297				},
298				_ => {},
299			}
300		}
301
302		Ok(Costs::Costs(CostLookup { mints, spends }))
303	}
304}

324impl OgmiosUtxoExt for OgmiosUtxo {
325	fn to_csl_tx_input(&self) -> TransactionInput {
326		TransactionInput::new(&TransactionHash::from(self.transaction.id), self.index.into())
327	}
328
329	fn to_csl_tx_output(&self) -> Result<TransactionOutput, JsError> {
330		let mut tx_out = TransactionOutput::new(
331			&Address::from_bech32(&self.address).map_err(|e| {
332				JsError::from_str(&format!("Couldn't convert address from ogmios: '{}'", e))
333			})?,
334			&self.value.to_csl()?,
335		);
336		if let Some(script) = self.script.clone() {
337			let plutus_script_ref_opt = PlutusScript::from_ogmios(script.clone())
338				.ok()
339				.map(|plutus_script| ScriptRef::new_plutus_script(&plutus_script.to_csl()));
340			let script_ref_opt = plutus_script_ref_opt.or_else(|| {
341				NativeScript::from_bytes(script.cbor)
342					.ok()
343					.map(|native_script| ScriptRef::new_native_script(&native_script))
344			});
345			if let Some(script_ref) = script_ref_opt {
346				tx_out.set_script_ref(&script_ref);
347			}
348		}
349		if let Some(data) = self.get_plutus_data() {
350			tx_out.set_plutus_data(&data);
351		}
352		Ok(tx_out)
353	}
354
355	fn to_csl(&self) -> Result<TransactionUnspentOutput, JsError> {
356		Ok(TransactionUnspentOutput::new(&self.to_csl_tx_input(), &self.to_csl_tx_output()?))
357	}
358
359	fn get_asset_amount(&self, asset_id: &AssetId) -> u64 {
360		self.value
361			.native_tokens
362			.get(&asset_id.policy_id.0)
363			.cloned()
364			.unwrap_or_default()
365			.iter()
366			.find(|asset| asset.name == asset_id.asset_name.0.to_vec())
367			.map_or_else(|| 0, |asset| asset.amount)
368	}
369
370	fn get_plutus_data(&self) -> Option<PlutusData> {
371		(self.datum.as_ref())
372			.map(|datum| datum.bytes.clone())
373			.and_then(|bytes| PlutusData::from_bytes(bytes).ok())
374	}
375}

399impl TransactionContext {
400	/// Gets `TransactionContext`, having UTXOs for the given payment key and the network configuration,
401	/// required to perform most of the partner-chains smart contract operations.
402	pub(crate) async fn for_payment_key<C: QueryLedgerState + QueryNetwork>(
403		payment_key: &CardanoPaymentSigningKey,
404		client: &C,
405	) -> Result<TransactionContext, anyhow::Error> {
406		let payment_key = payment_key.clone();
407		let network = client.shelley_genesis_configuration().await?.network.to_csl();
408		let protocol_parameters = client.query_protocol_parameters().await?;
409		let payment_address = key_hash_address(&payment_key.0.to_public().hash(), network);
410		let payment_key_utxos = client.query_utxos(&[payment_address.to_bech32(None)?]).await?;
411		Ok(TransactionContext {
412			payment_key,
413			payment_key_utxos,
414			network,
415			protocol_parameters,
416			change_address: payment_address,
417		})
418	}

423			payment_key_utxos: self.payment_key_utxos.clone(),
424			network: self.network,
425			protocol_parameters: self.protocol_parameters.clone(),
426			change_address: change_address.clone(),
427		}

430	pub(crate) fn payment_key_hash(&self) -> Ed25519KeyHash {
431		self.payment_key.0.to_public().hash()
432	}
433
434	pub(crate) fn sign(&self, tx: &Transaction) -> Transaction {
435		let tx_hash: [u8; 32] = sidechain_domain::crypto::blake2b(tx.body().to_bytes().as_ref());
436		let signature = self.payment_key.0.sign(&tx_hash);
437		let mut witness_set = tx.witness_set();
438		let mut vkeywitnesses = witness_set.vkeys().unwrap_or_else(Vkeywitnesses::new);
439		vkeywitnesses
440			.add(&Vkeywitness::new(&Vkey::new(&self.payment_key.0.to_public()), &signature));
441		witness_set.set_vkeys(&vkeywitnesses);
442		Transaction::new(&tx.body(), &witness_set, tx.auxiliary_data())
443	}

534impl TransactionBuilderExt for TransactionBuilder {

536		&mut self,

538		policy: &PlutusScript,

540		ctx: &TransactionContext,

542		let amount_builder = TransactionOutputBuilder::new()

547		let output = amount_builder.with_minimum_ada_and_asset(&ma, ctx)?.build()?;

552		&mut self,
553		ctx: &TransactionContext,

555	) -> Result<(), JsError> {
556		let mut collateral_builder = TxInputsBuilder::new();
557		for utxo in inputs.iter() {
558			collateral_builder.add_regular_input(
559				&key_hash_address(&ctx.payment_key_hash(), ctx.network),
560				&utxo.to_csl_tx_input(),
561				&utxo.value.to_csl()?,
562			)?;
563		}
564		self.set_collateral(&collateral_builder);
565		Ok(())
566	}

568	fn add_mint_one_script_token(
569		&mut self,
570		script: &PlutusScript,
571		asset_name: &AssetName,
572		redeemer_data: &PlutusData,
573		ex_units: &ExUnits,
574	) -> Result<(), JsError> {
575		let mut mint_builder = self.get_mint_builder().unwrap_or(MintBuilder::new());
576
577		let validator_source = PlutusScriptSource::new(&script.to_csl());
578		let mint_witness = MintWitness::new_plutus_script(
579			&validator_source,
580			&Redeemer::new(&RedeemerTag::new_mint(), &0u32.into(), redeemer_data, ex_units),
581		);
582		mint_builder.add_asset(&mint_witness, asset_name, &Int::new_i32(1))?;
583		self.set_mint_builder(&mint_builder);
584		Ok(())
585	}

587	fn add_mint_script_tokens(
588		&mut self,
589		script: &PlutusScript,
590		asset_name: &AssetName,
591		redeemer_data: &PlutusData,
592		ex_units: &ExUnits,
593		amount: &Int,
594	) -> Result<(), JsError> {
595		let mut mint_builder = self.get_mint_builder().unwrap_or(MintBuilder::new());
596
597		let validator_source = PlutusScriptSource::new(&script.to_csl());
598		let mint_witness = MintWitness::new_plutus_script(
599			&validator_source,
600			&Redeemer::new(&RedeemerTag::new_mint(), &0u32.into(), redeemer_data, ex_units),
601		);
602		mint_builder.add_asset(&mint_witness, asset_name, amount)?;
603		self.set_mint_builder(&mint_builder);
604		Ok(())
605	}

607	fn add_mint_script_token_using_reference_script(
608		&mut self,
609		script: &Script,
610		ref_input: &TransactionInput,
611		amount: &Int,
612		costs: &Costs,
613	) -> Result<(), JsError> {
614		let mut mint_builder = self.get_mint_builder().unwrap_or(MintBuilder::new());
615
616		match script {
617			Script::Plutus(script) => {
618				let source = PlutusScriptSource::new_ref_input(
619					&script.csl_script_hash(),
620					ref_input,
621					&script.language,
622					script.bytes.len(),
623				);
624				let mint_witness = MintWitness::new_plutus_script(
625					&source,
626					&Redeemer::new(
627						&RedeemerTag::new_mint(),
628						&0u32.into(),
629						&unit_plutus_data(),
630						&costs.get_mint(script),
631					),
632				);
633				mint_builder.add_asset(&mint_witness, &empty_asset_name(), amount)?;
634				self.set_mint_builder(&mint_builder);
635			},
636			Script::Native(script) => {
637				// new_ref_input causes invalid fee
638				let source = NativeScriptSource::new(script);
639				let mint_witness = MintWitness::new_native_script(&source);
640				mint_builder.add_asset(&mint_witness, &empty_asset_name(), amount)?;
641				self.set_mint_builder(&mint_builder);
642				self.add_reference_input(ref_input);
643			},
644		}
645		Ok(())
646	}
647
648	fn balance_update_and_build(
649		&mut self,
650		ctx: &TransactionContext,
651	) -> Result<Transaction, JsError> {
652		fn max_possible_collaterals(ctx: &TransactionContext) -> Vec<OgmiosUtxo> {
653			let mut utxos = ctx.payment_key_utxos.clone();
654			utxos.sort_by(|a, b| b.value.lovelace.cmp(&a.value.lovelace));
655			let max_inputs = ctx.protocol_parameters.max_collateral_inputs;
656			utxos
657				.into_iter()
658				.take(max_inputs.try_into().expect("max_collateral_input fit in usize"))
659				.collect()
660		}
661		// Tries to balance tx with given collateral inputs
662		fn try_balance(
663			builder: &mut TransactionBuilder,
664			collateral_inputs: &[OgmiosUtxo],
665			ctx: &TransactionContext,
666		) -> Result<Transaction, JsError> {
667			builder.add_required_signer(&ctx.payment_key_hash());
668			if collateral_inputs.is_empty() {
669				builder.add_inputs_from_and_change(
670					&ctx.payment_key_utxos.to_csl()?,
671					CoinSelectionStrategyCIP2::LargestFirstMultiAsset,
672					&ChangeConfig::new(&ctx.change_address),
673				)?;
674			} else {
675				builder.add_collateral_inputs(ctx, collateral_inputs)?;
676				builder.set_script_data_hash(&[0u8; 32].into());
677				// Fake script script data hash is required for proper fee computation
678				builder.add_inputs_from_and_change_with_collateral_return(
679					&ctx.payment_key_utxos.to_csl()?,
680					CoinSelectionStrategyCIP2::LargestFirstMultiAsset,
681					&ChangeConfig::new(&ctx.change_address),
682					&ctx.protocol_parameters.collateral_percentage.into(),
683				)?;
684				builder.calc_script_data_hash(&convert_cost_models(
685					&ctx.protocol_parameters.plutus_cost_models,
686				))?;
687			}
688			builder.build_tx()
689		}
690		// Tries if the largest UTXO is enough to cover collateral, if not, adds more UTXOs
691		// starting from the largest remaining.
692		let mut selected = vec![];
693		for input in max_possible_collaterals(ctx) {
694			let mut builder = self.clone();
695			// Check if the used inputs are enough
696			let result = try_balance(&mut builder, &selected, ctx);
697			if result.is_ok() {
698				return result;
699			}
700			selected.push(input);
701		}
702
703		let balanced_transaction =
704		try_balance(self, &selected, ctx)
705			.map_err(|e| JsError::from_str(&format!("Could not balance transaction. Usually it means that the payment key does not own UTXO set required to cover transaction outputs and fees or to provide collateral. Cause: {}", e)))?;
706
707		debug_assert!(
708			balanced_transaction.body().collateral().is_some(),
709			"BUG: Balanced transaction should have collateral set."
710		);
711		debug_assert!(
712			balanced_transaction.body().collateral_return().is_some(),
713			"BUG: Balanced transaction should have collateral returned."
714		);
715
716		Ok(balanced_transaction)
717	}
718}

857pub(crate) trait MultiAssetExt: Sized {
858	fn from_ogmios_utxo(utxo: &OgmiosUtxo) -> Result<Self, JsError>;
859	fn with_asset_amount(self, asset: &AssetId, amount: impl Into<BigNum>)
860	-> Result<Self, JsError>;
861}
862
863impl MultiAssetExt for MultiAsset {
864	fn from_ogmios_utxo(utxo: &OgmiosUtxo) -> Result<Self, JsError> {
865		let mut ma = MultiAsset::new();
866		for (policy, policy_assets) in utxo.value.native_tokens.iter() {
867			let mut assets = Assets::new();
868			for asset in policy_assets {
869				assets.insert(
870					&cardano_serialization_lib::AssetName::new(asset.name.clone())?,
871					&asset.amount.into(),
872				);
873			}
874			ma.insert(&PolicyID::from(*policy), &assets);
875		}
876		Ok(ma)
877	}
878	fn with_asset_amount(
879		mut self,
880		asset: &AssetId,
881		amount: impl Into<BigNum>,
882	) -> Result<Self, JsError> {
883		let policy_id = asset.policy_id.0.into();
884		let asset_name = asset.asset_name.to_csl()?;
885		let amount: BigNum = amount.into();
886		if amount > BigNum::zero() {
887			self.set_asset(&policy_id, &asset_name, &amount);
888			Ok(self)
889		} else {
890			// CSL doesn't have a public API to remove asset from MultiAsset, setting it to 0 isn't really helpful.
891			let current_value = self.get_asset(&policy_id, &asset_name);
892			if current_value > BigNum::zero() {
893				let ma_to_sub = MultiAsset::new().with_asset_amount(asset, current_value)?;
894				Ok(self.sub(&ma_to_sub))
895			} else {
896				Ok(self)
897			}
898		}
899	}

904	fn remove_native_script_witnesses(self) -> Self;
905}

907impl TransactionExt for Transaction {
908	fn remove_native_script_witnesses(self) -> Self {
909		let ws = self.witness_set();
910		let mut new_ws = TransactionWitnessSet::new();
911		if let Some(bootstraps) = ws.bootstraps() {
912			new_ws.set_bootstraps(&bootstraps)
913		}
914		if let Some(plutus_data) = ws.plutus_data() {
915			new_ws.set_plutus_data(&plutus_data);
916		}
917		if let Some(plutus_scripts) = ws.plutus_scripts() {
918			new_ws.set_plutus_scripts(&plutus_scripts);
919		}
920		if let Some(redeemers) = ws.redeemers() {
921			new_ws.set_redeemers(&redeemers);
922		}
923		if let Some(vkeys) = ws.vkeys() {
924			new_ws.set_vkeys(&vkeys);
925		}
926		Transaction::new(&self.body(), &new_ws, self.auxiliary_data())
927	}

1204	fn used_inputs_lovelace_equals_outputs_and_fee(tx: &Transaction, payment_utxos: &[OgmiosUtxo]) {
1205		let used_inputs: Vec<OgmiosUtxo> = match_inputs(&tx.body().inputs(), payment_utxos);
1206		let used_inputs_value: u64 = sum_lovelace(&used_inputs);
1207		let outputs_lovelace_sum: u64 = tx
1208			.body()
1209			.outputs()
1210			.into_iter()
1211			.map(|output| {
1212				let value: u64 = output.amount().coin().into();
1213				value
1214			})
1215			.sum();
1216		let fee: u64 = tx.body().fee().into();
1217		// Used inputs are qual to the sum of the outputs plus the fee
1218		assert_eq!(used_inputs_value, outputs_lovelace_sum + fee);
1219	}
