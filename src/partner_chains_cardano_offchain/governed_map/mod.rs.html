29pub async fn run_insert<
30	C: QueryLedgerState + QueryNetwork + Transactions + QueryUtxoByUtxoId,
31	A: AwaitTx,
32>(
33	genesis_utxo: UtxoId,
34	key: String,
35	value: ByteString,
36	payment_signing_key: &CardanoPaymentSigningKey,
37	ogmios_client: &C,
38	await_tx: &A,
39) -> anyhow::Result<Option<MultiSigSmartContractResult>> {
40	let ctx = TransactionContext::for_payment_key(payment_signing_key, ogmios_client).await?;
41	let (validator, policy) = crate::scripts_data::governed_map_scripts(genesis_utxo, ctx.network)?;
42	let validator_address = validator.address_bech32(ctx.network)?;
43	let validator_utxos = ogmios_client.query_utxos(&[validator_address]).await?;
44
45	let tx_hash_opt = match get_current_value(validator_utxos, key.clone(), policy.policy_id()) {
46		Some(current_value) if current_value != value => {
47			return Err(anyhow!(
48				"There is already a value stored for key '{key}': {current_value:?}"
49			));
50		},
51		Some(current_value) => {
52			log::info!(
53				"Value for key '{}' is already set to {:?}. Skipping insert.",
54				key,
55				current_value
56			);
57			None
58		},
59		None => {
60			log::info!("There is no value stored for key '{}'. Inserting new one.", key);
61			Some(insert(&validator, &policy, key, value, ctx, genesis_utxo, ogmios_client).await?)
62		},
63	};
64	if let Some(TransactionSubmitted(tx_hash)) = tx_hash_opt {
65		await_tx.await_tx_output(ogmios_client, UtxoId::new(tx_hash.0, 0)).await?;
66	}
67	Ok(tx_hash_opt)
68}

70fn ogmios_utxos_to_governed_map_utxos(
71	utxos: impl Iterator<Item = OgmiosUtxo>,
72	token: PolicyId,
73) -> impl Iterator<Item = GovernedMapDatum> {
74	utxos.flat_map(move |utxo| {
75		let _ = utxo.value.native_tokens.get(&token.0)?;
76		let datum = utxo.datum?;
77		let datum_plutus_data = PlutusData::from_bytes(datum.bytes).ok()?;
78
79		GovernedMapDatum::try_from(datum_plutus_data).ok()
80	})
81}

83fn get_current_value(
84	validator_utxos: Vec<OgmiosUtxo>,
85	key: String,
86	token: PolicyId,
87) -> Option<ByteString> {
88	ogmios_utxos_to_governed_map_utxos(validator_utxos.into_iter(), token)
89		.find(|datum| datum.key == key)
90		.map(|datum| datum.value)
91}

93async fn insert<C: QueryLedgerState + Transactions + QueryNetwork + QueryUtxoByUtxoId>(
94	validator: &PlutusScript,
95	policy: &PlutusScript,
96	key: String,
97	value: ByteString,
98	ctx: TransactionContext,
99	genesis_utxo: UtxoId,
100	client: &C,
101) -> anyhow::Result<MultiSigSmartContractResult> {
102	let governance_data = GovernanceData::get(genesis_utxo, client).await?;
103
104	submit_or_create_tx_to_sign(
105		&governance_data,
106		ctx,
107		|costs, ctx| {
108			insert_key_value_tx(
109				validator,
110				policy,
111				key.clone(),
112				value.clone(),
113				&governance_data,
114				costs,
115				&ctx,
116			)
117		},
118		"Insert Key-Value pair",
119		client,
120		&FixedDelayRetries::five_minutes(),
121	)
122	.await
123}

125fn insert_key_value_tx(
126	validator: &PlutusScript,
127	policy: &PlutusScript,
128	key: String,
129	value: ByteString,
130	governance_data: &GovernanceData,
131	costs: Costs,
132	ctx: &TransactionContext,
133) -> anyhow::Result<Transaction> {
134	let mut tx_builder = TransactionBuilder::new(&get_builder_config(ctx)?);
135	tx_builder.add_mint_one_script_token(
136		policy,
137		&empty_asset_name(),
138		&unit_plutus_data(),
139		&costs.get_mint(&policy.clone()),
140	)?;
141	tx_builder.add_output_with_one_script_token(
142		validator,
143		policy,
144		&governed_map_datum_to_plutus_data(&GovernedMapDatum::new(key, value)),
145		ctx,
146	)?;
147
148	let gov_tx_input = governance_data.utxo_id_as_tx_input();
149	tx_builder.add_mint_one_script_token_using_reference_script(
150		&governance_data.policy.script(),
151		&gov_tx_input,
152		&costs,
153	)?;
154
155	Ok(tx_builder.balance_update_and_build(ctx)?.remove_native_script_witnesses())
156}
