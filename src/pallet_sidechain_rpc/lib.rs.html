58#[async_trait]
59impl<C, Block> SidechainRpcApiServer for SidechainRpc<C, Block>
60where
61	Block: BlockT,
62	C: Send + Sync + 'static,
63	C: ProvideRuntimeApi<Block>,
64	C: GetBestHash<Block>,
65	C::Api: SlotApi<Block> + GetGenesisUtxo<Block> + GetSidechainStatus<Block>,
66{
67	fn get_params(&self) -> RpcResult<sp_sidechain::query::Output> {
68		let api = self.client.runtime_api();
69		let best_block = self.client.best_hash();
70
71		let genesis_utxo = api.genesis_utxo(best_block).map_err(error_object_from)?;
72
73		Ok(sp_sidechain::query::Output { genesis_utxo })
74	}
75
76	async fn get_status(&self) -> RpcResult<GetStatusResponse> {
77		let api = self.client.runtime_api();
78		let best_block = self.client.best_hash();
79
80		let slot_config = api.slot_config(best_block).map_err(error_object_from)?;
81
82		let current_timestamp = self.get_current_timestamp();
83		let current_sidechain_slot =
84			slot_config.slot_from_timestamp(current_timestamp.unix_millis());
85		let current_sidechain_epoch = slot_config.epoch_number(current_sidechain_slot);
86		let next_sidechain_epoch_timestamp = slot_config
87			.epoch_start_time(current_sidechain_epoch.next())
88			.ok_or(GetStatusRpcError::CannotConvertSidechainSlotToTimestamp)?;
89
90		let latest_mainchain_block =
91			self.data_source.get_latest_block_info().await.map_err(|err| {
92				ErrorObject::owned(
93					ErrorCode::InternalError.code(),
94					format!("Internal error: GetLatestBlockResponse error '{:?}", err),
95					None::<u8>,
96				)
97			})?;
98		let next_mainchain_epoch_timestamp = self
99			.mc_epoch_config
100			.mainchain_epoch_to_timestamp(latest_mainchain_block.epoch.next());
101
102		Ok(GetStatusResponse {
103			sidechain: SidechainData {
104				epoch: current_sidechain_epoch.0,
105				slot: current_sidechain_slot.into(),
106				next_epoch_timestamp: next_sidechain_epoch_timestamp,
107			},
108			mainchain: MainchainData {
109				epoch: latest_mainchain_block.epoch.0,
110				slot: latest_mainchain_block.slot.0,
111				next_epoch_timestamp: next_mainchain_epoch_timestamp,
112			},
113		})
114	}
115}
