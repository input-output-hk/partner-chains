85/// Handler trait for runtime components which need to react to changes in the Governed Map.
86pub trait OnGovernedMappingChange<MaxKeyLength, MaxValueLength>
87where
88	MaxKeyLength: Get<u32>,
89	MaxValueLength: Get<u32>,
90{
91	/// Processes a change to a single governed mapping.
92	fn on_governed_mapping_change(
93		key: BoundedString<MaxKeyLength>,
94		new_value: Option<BoundedVec<u8, MaxValueLength>>,
95		old_value: Option<BoundedVec<u8, MaxValueLength>>,
96	);
97}
98
99impl<MaxKeyLength, MaxValueLength> OnGovernedMappingChange<MaxKeyLength, MaxValueLength> for ()
100where
101	MaxKeyLength: Get<u32>,
102	MaxValueLength: Get<u32>,
103{
104	fn on_governed_mapping_change(
105		_key: BoundedString<MaxKeyLength>,
106		_new_value: Option<BoundedVec<u8, MaxValueLength>>,
107		_old_value: Option<BoundedVec<u8, MaxValueLength>>,
108	) {
109	}
110}
111
112macro_rules! impl_tuple_on_governed_mapping_change {
113    ($first_type:ident, $($type:ident),+) => {
114		impl<MaxKeyLength, MaxValueLength, $first_type, $($type),+>
115			OnGovernedMappingChange<MaxKeyLength, MaxValueLength>
116		for ($first_type, $($type),+) where
117			MaxKeyLength: Get<u32>,
118			MaxValueLength: Get<u32>,
119			$first_type: OnGovernedMappingChange<MaxKeyLength, MaxValueLength>,
120			$($type: OnGovernedMappingChange<MaxKeyLength, MaxValueLength>),+
121		{
122			fn on_governed_mapping_change(
123				key: BoundedString<MaxKeyLength>,
124				new_value: Option<BoundedVec<u8, MaxValueLength>>,
125				old_value: Option<BoundedVec<u8, MaxValueLength>>,
126			) {
127				<$first_type as OnGovernedMappingChange<MaxKeyLength, MaxValueLength>>::on_governed_mapping_change(key.clone(), new_value.clone(), old_value.clone());
128				$(<$type as OnGovernedMappingChange<MaxKeyLength, MaxValueLength>>::on_governed_mapping_change(key.clone(), new_value.clone(), old_value.clone());)+
129			}
130		}
131    };
132}

141/// Inherent data provider providing the list of Governed Map changes that occured since previous observation.
142#[cfg(feature = "std")]
143#[derive(Debug, PartialEq)]
144pub enum GovernedMapInherentDataProvider {
145	/// Inactive variant that will not provide any data and will not raise any errors.
146	Inert,
147	/// Active variant that will provide data.
148	ActiveV1 { changes: ChangesV1 },
149}
150
151#[cfg(feature = "std")]
152#[async_trait::async_trait]
153impl sp_inherents::InherentDataProvider for GovernedMapInherentDataProvider {
154	async fn provide_inherent_data(
155		&self,
156		inherent_data: &mut sp_inherents::InherentData,
157	) -> Result<(), sp_inherents::Error> {
158		match self {
159			Self::ActiveV1 { changes } if !changes.is_empty() => {
160				inherent_data.put_data(INHERENT_IDENTIFIER, &changes)?;
161			},
162			_ => {},
163		}
164		Ok(())
165	}
166
167	async fn try_handle_error(
168		&self,
169		identifier: &InherentIdentifier,
170		mut error: &[u8],
171	) -> Option<Result<(), sp_inherents::Error>> {
172		if identifier == &INHERENT_IDENTIFIER {
173			let error = InherentError::decode(&mut error).ok()?;
174			Some(Err(sp_inherents::Error::Application(Box::from(error))))
175		} else {
176			None
177		}
178	}
179}

193/// Error type returned when creation of [GovernedMapInherentDataProvider] fails.
194#[cfg(feature = "std")]
195#[derive(Debug, thiserror::Error)]
196pub enum InherentProviderCreationError {
197	#[error("Runtime API call failed: {0}")]
198	ApiError(#[from] sp_api::ApiError),
199	#[error("Data source call failed: {0}")]
200	DataSourceError(Box<dyn std::error::Error + Send + Sync>),
201	#[error("Unsupported pallet version {0} (highest supported version: {1})")]
202	UnsupportedPalletVersion(u32, u32),
203}

205#[cfg(feature = "std")]
206impl GovernedMapInherentDataProvider {
207	/// Creates a new [GovernedMapInherentDataProvider] with reference to the passed Cardano block hash.
208	///
209	/// This function is version aware and will return:
210	/// - an inert IDP if the pallet is not present in the runtime
211	/// - an inert IDP if the pallet does not have main chain scripts configured yet
212	/// - an IDP compatible with the pallet version signalled through the runtime API
213	///
214	/// Parameters:
215	/// - `client`: runtime client capable of providing [GovernedMapIDPApi] runtime API
216	/// - `parent_hash`: parent hash of the current block
217	/// - `mc_hash`: Cardano block hash referenced by the current block
218	/// - `data_source`: data source implementing [GovernedMapDataSource]
219	pub async fn new<T, Block>(
220		client: &T,
221		parent_hash: Block::Hash,
222		mc_hash: McBlockHash,
223		data_source: &(dyn GovernedMapDataSource + Send + Sync),
224	) -> Result<Self, InherentProviderCreationError>
225	where
226		Block: sp_runtime::traits::Block,
227		T: ProvideRuntimeApi<Block> + Send + Sync,
228		T::Api: GovernedMapIDPApi<Block>,
229	{
230		let api = client.runtime_api();
231		if !api.has_api::<dyn GovernedMapIDPApi<Block>>(parent_hash)? {
232			log::info!("ðŸ’¤ Skipping Governed Map observation. Pallet not detected in the runtime.");
233			return Ok(Self::Inert);
234		}
235
236		match api.get_pallet_version(parent_hash)? {
237			1 => Self::new_v1(client, parent_hash, mc_hash, data_source).await,
238			unsupported_version => {
239				Err(InherentProviderCreationError::UnsupportedPalletVersion(unsupported_version, 1))
240			},
241		}
242	}
243
244	async fn new_v1<T, Block>(
245		client: &T,
246		parent_hash: Block::Hash,
247		mc_hash: McBlockHash,
248		data_source: &(dyn GovernedMapDataSource + Send + Sync),
249	) -> Result<Self, InherentProviderCreationError>
250	where
251		Block: sp_runtime::traits::Block,
252		T: ProvideRuntimeApi<Block> + Send + Sync,
253		T::Api: GovernedMapIDPApi<Block>,
254	{
255		let api = client.runtime_api();
256
257		let Some(main_chain_script) = api.get_main_chain_scripts(parent_hash)? else {
258			log::info!("ðŸ’¤ Skipping Governed Map observation. Main chain scripts not set yet.");
259			return Ok(Self::Inert);
260		};
261
262		let entries_in_storage = api.get_stored_mappings(parent_hash)?;
263		let current_entries = data_source
264			.get_current_mappings(mc_hash, main_chain_script)
265			.await
266			.map_err(InherentProviderCreationError::DataSourceError)?;
267
268		let mut changes: ChangesV1 = ChangesV1::new();
269
270		for (key, value) in current_entries.iter() {
271			if entries_in_storage.get(key) != Some(value) {
272				changes.push(GovernedMapChangeV1::upsert(&key, &value));
273			}
274		}
275
276		for key in entries_in_storage.keys() {
277			if !current_entries.contains_key(key) {
278				changes.push(GovernedMapChangeV1::delete(&key));
279			}
280		}
281
282		changes.sort();
283
284		Ok(Self::ActiveV1 { changes })
285	}
286}
