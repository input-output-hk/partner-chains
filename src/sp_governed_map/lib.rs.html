1//! # Governed Map primitives
2//!
3//! This crate provides shared types and logic of the Governed Map feature, together with its inherent
4//! data provider logic.
5//!
6//! ## Usage
7//!
8//!    This crate supports operation of `pallet_governed_map`. Consult the pallet's documentation on how to
9//!    include it in the runtime
10//!
11//! ### Adding to the node
12//!
13//! #### Implementing the runtime API
14//!
15//! [GovernedMapInherentDataProvider] requires access to the pallet's configuration via a runtime API.
16//! Assuming the pallet has been added to the runtime and is named `GovernedMap`, the API should be
17//! implemented like this:
18//!
19//! ```rust,ignore
20//! impl sp_governed_map::GovernedMapIDPApi<Block> for Runtime {
21//!     fn get_main_chain_scripts() -> Option<MainChainScriptsV1> {
22//!         GovernedMap::get_main_chain_scripts()
23//!     }
24//!     fn get_pallet_version() -> u32 {
25//!         GovernedMap::get_version()
26//!     }
27//! }
28//! ```
29//!
30//! #### Adding the data source
31//!
32//! [GovernedMapInherentDataProvider] needs a data source implementing the [GovernedMapDataSource] trait.
33//! For nodes using Db-Sync, one is provided by the `partner_chains_db_sync_data_sources` crate. Consult
34//! its documentation for more information.
35//!
36//! #### Adding the inherent data provider
37//!
38//! The [GovernedMapInherentDataProvider] should be added to your IDP stack using [GovernedMapInherentDataProvider::new]
39//! for both block proposal and validation, like so:
40//! ```rust
41//! # use sp_governed_map::*;
42//! # use sidechain_domain::*;
43//! type InherentDataProviders = (/* other IDPs */ GovernedMapInherentDataProvider);
44//! async fn create_idps<T, Block>(
45//!     client: &T,
46//!     governed_map_data_source: &(impl GovernedMapDataSource + Send + Sync),
47//!     parent_hash: Block::Hash,
48//! #   // Arguments below should be provided by the MC Hash IDP from `sidechain_mc_hash` crate
49//! #   mc_hash: McBlockHash,
50//! #   previous_mc_hash: Option<McBlockHash>,
51//! ) -> Result<InherentDataProviders, Box<dyn std::error::Error + Send + Sync>>
52//! where
53//!     Block: sp_runtime::traits::Block,
54//!     T: sp_api::ProvideRuntimeApi<Block> + Send + Sync,
55//!     T::Api: GovernedMapIDPApi<Block> + Send + Sync
56//! {
57//!     /*
58//!      Create other IDPs
59//!      */
60//!     let governed_map = GovernedMapInherentDataProvider::new(
61//!         client,
62//!         parent_hash,
63//!         mc_hash,
64//!         previous_mc_hash,
65//!         governed_map_data_source
66//!     )
67//!     .await?;
68//!     Ok((/* other IDPs */ governed_map))
69//! # }
70//! ```
71//!
72//! Note that it requires access to the current and previous referenced Cardano block hash (`mc_hash` and `previous_mc_hash`).
73//! These are provided by the Partner Chains Toolkit's MC Hash inherent data provider from the `sidechain_mc_hash` crate.
74//!
75//! ## Adding to a running chain
76//!
77//! As with any other feature, if the Governed Map feature is to be added to an already running chain, a strict order
78//! of operations is required:
79//! 1. The node should be updated according to the steps described above, so that the inherent data provider is present
80//!    in the nodes IDP stack for both proposal and verfication.
81//! 2. The updated node binary should be distributed to the block producers who should update their nodes.
82//! 3. A new runtime version should be released with the pallet added according to its documentation and the runtime
83//!    API implemented as described above.
84//! 4. The Governed Map main chain scripts should be set through the `set_main_chain_scripts` extrinsic in the pallet.
85//!
86//! [GovernedMapInherentDataProvider] is version-aware and will stay inactive until the pallet is added and fully configured.
87#![cfg_attr(not(feature = "std"), no_std)]
88#![deny(missing_docs)]

143/// Error type returned when creating or validating the Governed Map inherent
144#[derive(Decode, Encode, Debug, PartialEq)]
145#[cfg_attr(feature = "std", derive(thiserror::Error))]
146pub enum InherentError {
147	/// Signals that the inherent was expected but not produced
148	#[cfg_attr(feature = "std", error("Inherent missing for Governed Map pallet"))]
149	InherentMissing,
150	/// Signals that the inherent was produced when not expected
151	#[cfg_attr(feature = "std", error("Unexpected inherent for Governed Map pallet"))]
152	InherentNotExpected,
153	#[cfg_attr(
154		feature = "std",
155		error("Data in Governed Map pallet inherent differs from inherent data")
156	)]
157	/// Signals that the inherent produced contains incorrect data
158	IncorrectInherent,
159	#[cfg_attr(feature = "std", error("Governed Map key {0} exceeds size bounds"))]
160	/// Signals that a key in the mapping change list is longer than the configured bound
161	KeyExceedsBounds(String),
162	#[cfg_attr(feature = "std", error("Governed Map value {1:?} for key {0} exceeds size bounds"))]
163	/// Signals that a value in the mapping change list is longer than the configured bound
164	ValueExceedsBounds(String, ByteString),
165	/// Signals that the number of unregistered changes to the mapping exceeds the configured upper limit
166	///
167	/// This should not normally occur if the pallet is configured to accept at least as many changes
168	/// as the planned number of keys in use, or if the number of keys exceeds this limit but the number
169	/// of changes is low enough not to overwhelm a non-stalled chain.
170	///
171	/// As this error prevents the production of a block, if this error occurs on a live chain, then the
172	/// only way of fixing it is to change the mappings on Cardano close enough to the last state
173	/// registered in the pallet to bring the change count below the limit.
174	#[cfg_attr(feature = "std", error("Number of changes to the Governed Map exceeds the limit"))]
175	TooManyChanges,
176}

184/// Handler trait for runtime components which need to react to changes in the Governed Map.
185pub trait OnGovernedMappingChange<MaxKeyLength, MaxValueLength>
186where
187	MaxKeyLength: Get<u32>,
188	MaxValueLength: Get<u32>,
189{
190	/// Processes a change to a single governed mapping.
191	fn on_governed_mapping_change(
192		key: BoundedString<MaxKeyLength>,
193		new_value: Option<BoundedVec<u8, MaxValueLength>>,
194		old_value: Option<BoundedVec<u8, MaxValueLength>>,
195	);
196}
197
198impl<MaxKeyLength, MaxValueLength> OnGovernedMappingChange<MaxKeyLength, MaxValueLength> for ()
199where
200	MaxKeyLength: Get<u32>,
201	MaxValueLength: Get<u32>,
202{
203	fn on_governed_mapping_change(
204		_key: BoundedString<MaxKeyLength>,
205		_new_value: Option<BoundedVec<u8, MaxValueLength>>,
206		_old_value: Option<BoundedVec<u8, MaxValueLength>>,
207	) {
208	}
209}
210
211macro_rules! impl_tuple_on_governed_mapping_change {
212    ($first_type:ident, $($type:ident),+) => {
213		impl<MaxKeyLength, MaxValueLength, $first_type, $($type),+>
214			OnGovernedMappingChange<MaxKeyLength, MaxValueLength>
215		for ($first_type, $($type),+) where
216			MaxKeyLength: Get<u32>,
217			MaxValueLength: Get<u32>,
218			$first_type: OnGovernedMappingChange<MaxKeyLength, MaxValueLength>,
219			$($type: OnGovernedMappingChange<MaxKeyLength, MaxValueLength>),+
220		{
221			fn on_governed_mapping_change(
222				key: BoundedString<MaxKeyLength>,
223				new_value: Option<BoundedVec<u8, MaxValueLength>>,
224				old_value: Option<BoundedVec<u8, MaxValueLength>>,
225			) {
226				<$first_type as OnGovernedMappingChange<MaxKeyLength, MaxValueLength>>::on_governed_mapping_change(key.clone(), new_value.clone(), old_value.clone());
227				$(<$type as OnGovernedMappingChange<MaxKeyLength, MaxValueLength>>::on_governed_mapping_change(key.clone(), new_value.clone(), old_value.clone());)+
228			}
229		}
230    };
231}

233impl_tuple_on_governed_mapping_change!(A, B);
234impl_tuple_on_governed_mapping_change!(A, B, C);
235impl_tuple_on_governed_mapping_change!(A, B, C, D);
236impl_tuple_on_governed_mapping_change!(A, B, C, D, E);

240/// Inherent data provider providing the list of Governed Map changes that occurred since previous observation.
241#[cfg(feature = "std")]
242#[derive(Debug, PartialEq)]
243pub enum GovernedMapInherentDataProvider {
244	/// Inactive variant that will not provide any data and will not raise any errors.
245	Inert,
246	/// Active variant that will provide data.
247	ActiveV1 {
248		/// List of changes to the Governed Map that occurred since previous observation
249		changes: ChangesV1,
250	},
251}
252
253#[cfg(feature = "std")]
254#[async_trait::async_trait]
255impl sp_inherents::InherentDataProvider for GovernedMapInherentDataProvider {
256	async fn provide_inherent_data(
257		&self,
258		inherent_data: &mut sp_inherents::InherentData,
259	) -> Result<(), sp_inherents::Error> {
260		match self {
261			Self::ActiveV1 { changes } if !changes.is_empty() => {
262				inherent_data.put_data(INHERENT_IDENTIFIER, &changes)?;
263			},
264			_ => {},
265		}
266		Ok(())
267	}
268
269	async fn try_handle_error(
270		&self,
271		identifier: &InherentIdentifier,
272		mut error: &[u8],
273	) -> Option<Result<(), sp_inherents::Error>> {
274		if identifier == &INHERENT_IDENTIFIER {
275			let error = InherentError::decode(&mut error).ok()?;
276			Some(Err(sp_inherents::Error::Application(Box::from(error))))
277		} else {
278			None
279		}
280	}
281}

283/// Cardano observability data source API used by [GovernedMapInherentDataProvider].
284#[cfg(feature = "std")]
285#[async_trait::async_trait]
286pub trait GovernedMapDataSource {
287	/// Queries all changes that occurred in the mappings of the Governed Map on Cardano in the given range of blocks.
288	///
289	/// # Arguments:
290	/// - `since_mc_block`: lower bound (exclusive). If [None], the data source should return all changes since the genesis block.
291	/// - `up_to_block`: upper bound (inclusive).
292	///
293	/// # Return value:
294	/// Items in the returned vector should be key-value pairs representing changes to the Governed Map. Inserts and updates should
295	/// be represented as a [Some] containing the new value, while deletions should be indicated by a [None]. The vector should be
296	/// ordered from the oldest change to the newest.
297	async fn get_mapping_changes(
298		&self,
299		since_mc_block: Option<McBlockHash>,
300		up_to_mc_block: McBlockHash,
301		main_chain_scripts: MainChainScriptsV1,
302	) -> Result<Vec<(String, Option<ByteString>)>, Box<dyn std::error::Error + Send + Sync>>;
303}

305/// Error type returned when creation of [GovernedMapInherentDataProvider] fails.
306#[cfg(feature = "std")]
307#[derive(Debug, thiserror::Error)]
308pub enum InherentProviderCreationError {
309	/// Signals that a runtime API call failed
310	#[error("Runtime API call failed: {0}")]
311	ApiError(#[from] sp_api::ApiError),
312	/// Signals that the data source returned an error
313	#[error("Data source call failed: {0}")]
314	DataSourceError(Box<dyn std::error::Error + Send + Sync>),
315	/// Signals that the current pallet version on the chain is higher than supported by the node's IDP
316	#[error("Unsupported pallet version {0} (highest supported version: {1})")]
317	UnsupportedPalletVersion(u32, u32),
318}

320#[cfg(feature = "std")]
321impl GovernedMapInherentDataProvider {
322	/// Creates a new [GovernedMapInherentDataProvider] with reference to the passed Cardano block hash.
323	///
324	/// This function is version aware and will return:
325	/// - an inert IDP if the pallet is not present in the runtime
326	/// - an inert IDP if the pallet does not have main chain scripts configured yet
327	/// - an IDP compatible with the pallet version signalled through the runtime API
328	///
329	/// Parameters:
330	/// - `client`: runtime client capable of providing [GovernedMapIDPApi] runtime API
331	/// - `parent_hash`: parent hash of the current block
332	/// - `mc_hash`: Cardano block hash referenced by the current block
333	/// - `data_source`: data source implementing [GovernedMapDataSource]
334	pub async fn new<T, Block>(
335		client: &T,
336		parent_hash: Block::Hash,
337		mc_hash: McBlockHash,
338		parent_mc_hash: Option<McBlockHash>,
339		data_source: &(dyn GovernedMapDataSource + Send + Sync),
340	) -> Result<Self, InherentProviderCreationError>
341	where
342		Block: sp_runtime::traits::Block,
343		T: ProvideRuntimeApi<Block> + Send + Sync,
344		T::Api: GovernedMapIDPApi<Block>,
345	{
346		let api = client.runtime_api();
347		if !api.has_api::<dyn GovernedMapIDPApi<Block>>(parent_hash)? {
348			log::info!("ðŸ’¤ Skipping Governed Map observation. Pallet not detected in the runtime.");
349			return Ok(Self::Inert);
350		}
351
352		match api.get_pallet_version(parent_hash)? {
353			1 => Self::new_v1(client, parent_hash, mc_hash, parent_mc_hash, data_source).await,
354			unsupported_version => {
355				Err(InherentProviderCreationError::UnsupportedPalletVersion(unsupported_version, 1))
356			},
357		}
358	}
359
360	async fn new_v1<T, Block>(
361		client: &T,
362		parent_hash: Block::Hash,
363		mc_hash: McBlockHash,
364		parent_mc_hash: Option<McBlockHash>,
365		data_source: &(dyn GovernedMapDataSource + Send + Sync),
366	) -> Result<Self, InherentProviderCreationError>
367	where
368		Block: sp_runtime::traits::Block,
369		T: ProvideRuntimeApi<Block> + Send + Sync,
370		T::Api: GovernedMapIDPApi<Block>,
371	{
372		let api = client.runtime_api();
373
374		let Some(main_chain_script) = api.get_main_chain_scripts(parent_hash)? else {
375			log::info!("ðŸ’¤ Skipping Governed Map observation. Main chain scripts not set yet.");
376			return Ok(Self::Inert);
377		};
378
379		let current_entries = data_source
380			.get_mapping_changes(parent_mc_hash, mc_hash, main_chain_script)
381			.await
382			.map_err(InherentProviderCreationError::DataSourceError)?;
383
384		let mut changes: ChangesV1 = ChangesV1::new();
385
386		for (key, value) in current_entries.into_iter() {
387			let key = key.into();
388			let change = match value {
389				None => GovernedMapChangeV1 { key, new_value: None },
390				Some(value) => GovernedMapChangeV1 { key, new_value: Some(value.into()) },
391			};
392			changes.push(change);
393		}
394
395		changes.sort();
396
397		Ok(Self::ActiveV1 { changes })
398	}
399}
