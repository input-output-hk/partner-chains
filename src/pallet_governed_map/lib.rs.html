1//! Pallet exposing the key-value pairs sourced from the Cardano ledger as part of the Governed Map feature.
2//!
3//! # Purpose of this pallet
4//!
5//! This pallet stores the most recent state of the key-value pairs in the Governed Map on Cardano for use
6//! by other runtime components. It also exposes hooks for other components to be notified when particular
7//! key-value pair is inserted, updated or deleted.

31	#[pallet::config]
32	pub trait Config: frame_system::Config {
33		/// Maximum number of changes that can be registered in a single inherent.
34		///
35		/// This value should be high enough for all changes to be registered in one block.
36		/// Setting this to a value higher than the total number of parameters in the Governed Map guarantees that.
37		type MaxChanges: Get<u32>;
38
39		/// Maximum length of the key in the Governed Map in bytes.
40		type MaxKeyLength: Get<u32>;
41
42		/// Maximum length of data stored under a single key in the Governed Map
43		type MaxValueLength: Get<u32>;
44
45		// TODO benchmarks
46		type WeightInfo: WeightInfo;
47	}
48
49	pub(crate) type MapKey<T> = BoundedString<<T as Config>::MaxKeyLength>;
50	pub(crate) type MapValue<T> = BoundedVec<u8, <T as Config>::MaxValueLength>;

79	#[pallet::inherent]
80	impl<T: Config> ProvideInherent for Pallet<T> {
81		type Call = Call<T>;
82		type Error = InherentError;
83		const INHERENT_IDENTIFIER: InherentIdentifier = INHERENT_IDENTIFIER;
84
85		fn create_inherent(data: &InherentData) -> Option<Self::Call> {
86			Self::create_inherent_or_err(data).expect("Creating Governed Map inherent failed")
87		}
88
89		fn check_inherent(call: &Self::Call, data: &InherentData) -> Result<(), Self::Error> {
90			let Some(expected_call) = Self::create_inherent(data) else {
91				return Err(Self::Error::InherentNotExpected);
92			};
93
94			if *call != expected_call {
95				return Err(Self::Error::IncorrectInherent);
96			}
97
98			Ok(())
99		}
100
101		fn is_inherent(call: &Self::Call) -> bool {
102			matches!(call, Call::register_changes { .. })
103		}
104
105		fn is_inherent_required(data: &InherentData) -> Result<Option<Self::Error>, Self::Error> {
106			match Self::decode_inherent_data(data) {
107				None => Ok(None),
108				Some(_) => Ok(Some(Self::Error::InherentMissing)),
109			}
110		}
111	}
112
113	impl<T: Config> Pallet<T> {
114		fn decode_inherent_data(data: &InherentData) -> Option<Vec<GovernedMapChangeV1>> {
115			data.get_data::<Vec<GovernedMapChangeV1>>(&INHERENT_IDENTIFIER)
116				.expect("Governed Map inherent data is not encoded correctly")
117		}
118
119		fn create_inherent_or_err(data: &InherentData) -> Result<Option<Call<T>>, InherentError> {
120			use InherentError::*;
121
122			let Some(raw_changes) = Self::decode_inherent_data(data) else { return Ok(None) };
123
124			if raw_changes.len() > T::MaxChanges::get() as usize {
125				return Err(TooManyChanges);
126			}
127
128			let mut changes: BoundedVec<(MapKey<T>, Option<MapValue<T>>), T::MaxChanges> =
129				BoundedVec::new();
130			for GovernedMapChangeV1 { key, new_value } in raw_changes {
131				let new_value = match new_value {
132					Some(new_value) => Some(Self::bound_value(&key, new_value)?),
133					None => None,
134				};
135				let key = Self::bound_key(&key)?;
136				changes.try_push((key, new_value)).expect("Number of changes is below maximum");
137			}
138			Ok(Some(Call::register_changes { changes }))
139		}
140
141		fn bound_key(key: &str) -> Result<MapKey<T>, InherentError> {
142			key.try_into().map_err(|_| InherentError::KeyExceedsBounds(key.into()))
143		}
144
145		fn bound_value(key: &str, value: ByteString) -> Result<MapValue<T>, InherentError> {
146			(value.0.clone().try_into())
147				.map_err(|_| InherentError::ValueExceedsBounds(key.into(), value))
148		}
149	}

151	#[pallet::call]
152	impl<T: Config> Pallet<T> {
153		/// Inherent to register any changes in the state of the Governed Map on Cardano compared to the state currently stored in the pallet.
154		#[pallet::call_index(0)]
155		#[pallet::weight((0, DispatchClass::Mandatory))]
156		pub fn register_changes(
157			origin: OriginFor<T>,
158			changes: BoundedVec<(MapKey<T>, Option<MapValue<T>>), T::MaxChanges>,
159		) -> DispatchResult {
160			ensure_none(origin)?;
161
162			log::info!("💾 Registering {} Governed Map changes", changes.len(),);
163
164			for (key, value) in changes {
165				Mapping::<T>::set(key, value);
166			}
167
168			Ok(())
169		}
170
171		/// Changes the address of the Governed Map validator used for observation.
172		///
173		/// This extrinsic must be run either using `sudo` or some other chain governance mechanism.
174		#[pallet::call_index(1)]
175		#[pallet::weight((0, DispatchClass::Normal))]
176		pub fn set_main_chain_scripts(
177			origin: OriginFor<T>,
178			new_main_chain_script: MainChainScriptsV1,
179		) -> DispatchResult {
180			ensure_root(origin)?;
181			MainChainScripts::<T>::put(new_main_chain_script.clone());
182			log::info!("🗂️ Governed Map main chain scripts updated to {new_main_chain_script:?}");
183			Ok(())
184		}
185	}

187	impl<T: Config> Pallet<T> {
188		/// Returns the value under `key` or [None] otherwise.
189		pub fn get_key_value(key: MapKey<T>) -> Option<BoundedVec<u8, T::MaxValueLength>> {

193		/// Returns an iterator over all key-value pairs in the pallet storage.
194		pub fn get_all_key_value_pairs() -> impl Iterator<Item = (MapKey<T>, MapValue<T>)> {
195			Mapping::<T>::iter()
196		}
197
198		/// Returns current pallet version.

201		}
