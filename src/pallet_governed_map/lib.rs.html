1//! Pallet exposing the key-value pairs sourced from the Cardano ledger as part of the Governed Map feature.
2//!
3//! # Purpose of this pallet
4//!
5//! This pallet stores the most recent state of the key-value pairs in the Governed Map on Cardano for use
6//! by other runtime components. It also exposes hooks for other components to be notified when particular
7//! key-value pair is inserted, updated or deleted.

41	#[pallet::config]
42	pub trait Config: frame_system::Config {
43		/// Maximum number of changes that can be registered in a single inherent.
44		///
45		/// This value should be high enough for all changes to be registered in one block.
46		/// Setting this to a value higher than the total number of parameters in the Governed Map guarantees that.
47		type MaxChanges: Get<u32>;
48
49		/// Maximum length of the key in the Governed Map in bytes.
50		type MaxKeyLength: Get<u32>;
51
52		/// Maximum length of data stored under a single key in the Governed Map
53		type MaxValueLength: Get<u32>;
54
55		/// Handler called for each change in the governed mappings.
56		///
57		/// If your runtime does not need to react to any changes, a no-op implementation for [()] can be used.
58		/// Otherwise, it is advised to benchmark the runtime and use your own weights to include weight consumed
59		/// by the handler.
60		type OnGovernedMappingChange: OnGovernedMappingChange<
61			Self::MaxKeyLength,
62			Self::MaxValueLength,
63		>;
64
65		/// Weight functions for the pallet's extrinsics
66		type WeightInfo: weights::WeightInfo;
67
68		/// Helper functions required by the pallet's benchmarks to construct realistic input data.
69		///
70		/// An implementation for [()] is provided for simplicity. Chains that require more precise weights or
71		/// expect an unusual number of parameter changes should implement this trait themselves in their runtime.
72		#[cfg(feature = "runtime-benchmarks")]
73		type BenchmarkHelper: crate::benchmarking::BenchmarkHelper<Self>;
74	}

107	#[pallet::inherent]
108	impl<T: Config> ProvideInherent for Pallet<T> {
109		type Call = Call<T>;
110		type Error = InherentError;
111		const INHERENT_IDENTIFIER: InherentIdentifier = INHERENT_IDENTIFIER;
112
113		fn create_inherent(data: &InherentData) -> Option<Self::Call> {
114			Self::create_inherent_or_err(data).expect("Creating Governed Map inherent failed")
115		}
116
117		fn check_inherent(call: &Self::Call, data: &InherentData) -> Result<(), Self::Error> {
118			let Some(expected_call) = Self::create_inherent(data) else {
119				return Err(Self::Error::InherentNotExpected);
120			};
121
122			if *call != expected_call {
123				return Err(Self::Error::IncorrectInherent);
124			}
125
126			Ok(())
127		}
128
129		fn is_inherent(call: &Self::Call) -> bool {
130			matches!(call, Call::register_changes { .. })
131		}
132
133		fn is_inherent_required(data: &InherentData) -> Result<Option<Self::Error>, Self::Error> {
134			match Self::decode_inherent_data(data) {
135				None => Ok(None),
136				Some(_) => Ok(Some(Self::Error::InherentMissing)),
137			}
138		}
139	}

141	impl<T: Config> Pallet<T> {
142		fn decode_inherent_data(
143			data: &InherentData,
144		) -> Option<alloc::vec::Vec<GovernedMapChangeV1>> {
145			data.get_data::<alloc::vec::Vec<GovernedMapChangeV1>>(&INHERENT_IDENTIFIER)
146				.expect("Governed Map inherent data is not encoded correctly")
147		}
148
149		fn create_inherent_or_err(data: &InherentData) -> Result<Option<Call<T>>, InherentError> {
150			use InherentError::*;
151
152			let Some(raw_changes) = Self::decode_inherent_data(data) else { return Ok(None) };
153
154			if raw_changes.len() > T::MaxChanges::get() as usize {
155				return Err(TooManyChanges);
156			}
157
158			let mut changes = Changes::<T>::new();
159			for GovernedMapChangeV1 { key, new_value } in raw_changes {
160				let new_value = match new_value {
161					Some(new_value) => Some(Self::bound_value(&key, new_value)?),
162					None => None,
163				};
164				let key = Self::bound_key(&key)?;
165				changes.try_push((key, new_value)).expect("Number of changes is below maximum");
166			}
167			Ok(Some(Call::register_changes { changes }))
168		}
169
170		fn bound_key(key: &str) -> Result<MapKey<T>, InherentError> {
171			key.try_into().map_err(|_| InherentError::KeyExceedsBounds(key.into()))
172		}
173
174		fn bound_value(key: &str, value: ByteString) -> Result<MapValue<T>, InherentError> {
175			(value.0.clone().try_into())
176				.map_err(|_| InherentError::ValueExceedsBounds(key.into(), value))
177		}
178	}

180	#[pallet::call]
181	impl<T: Config> Pallet<T> {
182		/// Inherent to register any changes in the state of the Governed Map on Cardano compared to the state currently stored in the pallet.
183		#[pallet::call_index(0)]
184		#[pallet::weight((T::WeightInfo::register_changes(changes.len() as u32), DispatchClass::Mandatory))]
185		pub fn register_changes(origin: OriginFor<T>, changes: Changes<T>) -> DispatchResult {
186			ensure_none(origin)?;
187
188			log::info!("üíæ Registering {} Governed Map changes", changes.len(),);
189
190			for (key, value) in changes {
191				let old_value = Mapping::<T>::get(&key);
192				Mapping::<T>::set(&key, value.clone());
193				T::OnGovernedMappingChange::on_governed_mapping_change(key, value, old_value);
194			}
195
196			Ok(())
197		}
198
199		/// Changes the address of the Governed Map validator used for observation.
200		///
201		/// This extrinsic must be run either using `sudo` or some other chain governance mechanism.
202		#[pallet::call_index(1)]
203		#[pallet::weight((T::WeightInfo::set_main_chain_scripts(), DispatchClass::Normal))]
204		pub fn set_main_chain_scripts(
205			origin: OriginFor<T>,
206			new_main_chain_script: MainChainScriptsV1,
207		) -> DispatchResult {
208			ensure_root(origin)?;
209			MainChainScripts::<T>::put(new_main_chain_script.clone());
210			log::info!("üóÇÔ∏è Governed Map main chain scripts updated to {new_main_chain_script:?}");
211			Ok(())
212		}
213	}

215	impl<T: Config> Pallet<T> {
216		/// Returns the value under `key` or [None] otherwise.
217		pub fn get_key_value(key: MapKey<T>) -> Option<BoundedVec<u8, T::MaxValueLength>> {
218			Mapping::<T>::get(key)
219		}
220
221		/// Returns an iterator over all key-value pairs in the pallet storage.
222		pub fn get_all_key_value_pairs() -> impl Iterator<Item = (MapKey<T>, MapValue<T>)> {
223			Mapping::<T>::iter()
224		}
225
226		/// Returns an iterator over all key-value pairs in the pallet storage, using unbound types.
227		pub fn get_all_key_value_pairs_unbounded() -> impl Iterator<Item = (String, ByteString)> {
228			Self::get_all_key_value_pairs()
229				.map(|(key, value)| (key.to_string(), value.to_vec().into()))
230		}
231
232		/// Returns current pallet version.
233		pub fn get_version() -> u32 {
234			PALLET_VERSION
235		}
236
237		/// Returns the current main chain scripts
238		pub fn get_main_chain_scripts() -> Option<MainChainScriptsV1> {
239			MainChainScripts::<T>::get()
240		}
