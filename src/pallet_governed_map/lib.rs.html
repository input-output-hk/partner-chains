1//! Pallet exposing key-value pairs sourced from the Cardano ledger as part of the Governed Map feature.
2//!
3//! # Purpose of this pallet
4//!
5//! This pallet stores the most recent state of the key-value pairs in the Governed Map on Cardano for use
6//! by other runtime components. It also exposes hooks for other components to be notified when particular
7//! key-value pair is inserted, updated or deleted.
8//!
9//! # Usage
10//!
11//! ## Adding to runtime
12//!
13//! ### Defining size limits
14//!
15//! Before adding the pallet to your runtime, first decide on the limits for the data that it will process:
16//! - `MaxChanges`: the maximum number of changes that will be processed in one inherent invocation.
17//!                 This number should be high enough to guarantee the inherent will always have the
18//!                 capacity required to process incoming changes. Setting this limit above the expected
19//!                 number of keys in use is a safe option.
20//! - `MaxKeyLength`: maximum length of keys used. Be warned that if a key is set in the Governed Map
21//!                   that exceeds this lenght limit, this pallet's inherent will fail and stall block
22//!                   production, with the only recovery path being removal of this key so it is no longer
23//!                   in the change set.
24//! - `MaxValueLength`: maximum length of the value under a key. Same considerations as for `MaxKeyLength`
25//!                     apply.
26//! Once the limit values are decided, define them in your runtime, like so:
27//! ```rust
28//! frame_support::parameter_types! {
29//!        pub const MaxChanges: u32 = 16;
30//!        pub const MaxKeyLength: u32 = 64;
31//!        pub const MaxValueLength: u32 = 512;
32//! }
33//! ```
34//!
35//! ### Implementing on-change handler
36//!
37//! The pallet allows the runtime implementer to define a handler that will be called for all key-value
38//! changes registered by the pallet. If your runtime needs to react to changes, crate a type implementing
39//! the [OnGovernedMappingChange] trait, eg:
40//! ```rust
41//! # use frame_support::BoundedVec;
42//! # use sidechain_domain::byte_string::BoundedString;
43//! # use sp_core::Get;
44//! struct ChangeHandler;
45//!
46//! impl<MaxKeyLength, MaxValueLength> sp_governed_map::OnGovernedMappingChange<MaxKeyLength, MaxValueLength> for ChangeHandler
47//! where
48//!    MaxKeyLength: Get<u32>,
49//!    MaxValueLength: Get<u32>,
50//! {
51//!    fn on_governed_mapping_change(
52//!        key: BoundedString<MaxKeyLength>,
53//!        new_value: Option<BoundedVec<u8, MaxValueLength>>,
54//!        old_value: Option<BoundedVec<u8, MaxValueLength>>,
55//!    ) {
56//!        log::info!("Governed Map change for key {key}: old value: {old_value:?}, new value: {new_value:?}");
57//!    }
58//! }
59//! ```
60//! If any handling is not needed, a no-op implementation for [()] can be used instead.
61//!
62//! ### Weights and Benchmarking
63//!
64//! The pallet comes with pre-defined weights for its extrinsics that can be used during initial development
65//! through the [pallet_governed_map::weights::SubstrateWeight] type.
66//!
67//! However, since data size limits and the on-change logic both can affect the weights, it is advisable to run
68//! your own benchmark to account for their impact. See the documentation on `[crate::benchmarking]` for details.
69//!
70//! ### Configuring the pallet
71//!
72//! Once the above are defined, the pallet can finally be added to the runtime and configured like this:
73//!
74//! ```rust,ignore
75//! impl pallet_governed_map::Config for Runtime {
76//!     type MaxChanges = MaxChanges;
77//!     type MaxKeyLength = MaxKeyLength;
78//!     type MaxValueLength = MaxValueLength;
79//!     type WeightInfo = pallet_governed_map::weights::SubstrateWeight<Runtime>;
80//!
81//!     type OnGovernedMappingChange = ChangeHandler;
82//!
83//!     #[cfg(feature = "runtime-benchmarks")]
84//!     type BenchmarkHelper = ();
85//! }
86//! ```

122	#[pallet::config]
123	pub trait Config: frame_system::Config {
124		/// Maximum number of changes that can be registered in a single inherent.
125		///
126		/// This value should be high enough for all changes to be registered in one block.
127		/// Setting this to a value higher than the total number of parameters in the Governed Map guarantees that.
128		type MaxChanges: Get<u32>;
129
130		/// Maximum length of the key in the Governed Map in bytes.
131		type MaxKeyLength: Get<u32>;
132
133		/// Maximum length of data stored under a single key in the Governed Map
134		type MaxValueLength: Get<u32>;
135
136		/// Handler called for each change in the governed mappings.
137		///
138		/// If your runtime does not need to react to any changes, a no-op implementation for [()] can be used.
139		/// Otherwise, it is advised to benchmark the runtime and use your own weights to include weight consumed
140		/// by the handler.
141		type OnGovernedMappingChange: OnGovernedMappingChange<Self::MaxKeyLength, Self::MaxValueLength>;
142
143		/// Origin for governance calls
144		type MainChainScriptsOrigin: EnsureOrigin<Self::RuntimeOrigin>;
145
146		/// Weight functions for the pallet's extrinsics
147		type WeightInfo: weights::WeightInfo;
148
149		/// Helper functions required by the pallet's benchmarks to construct realistic input data.
150		///
151		/// An implementation for [()] is provided for simplicity. Chains that require more precise weights or
152		/// expect an unusual number of parameter changes should implement this trait themselves in their runtime.
153		#[cfg(feature = "runtime-benchmarks")]
154		type BenchmarkHelper: crate::benchmarking::BenchmarkHelper<Self>;
155	}

192	#[pallet::inherent]
193	impl<T: Config> ProvideInherent for Pallet<T> {
194		type Call = Call<T>;
195		type Error = InherentError;
196		const INHERENT_IDENTIFIER: InherentIdentifier = INHERENT_IDENTIFIER;
197
198		fn create_inherent(data: &InherentData) -> Option<Self::Call> {
199			Self::create_inherent_or_err(data).expect("Creating Governed Map inherent failed")
200		}
201
202		fn check_inherent(call: &Self::Call, data: &InherentData) -> Result<(), Self::Error> {
203			let Some(expected_call) = Self::create_inherent(data) else {
204				return Err(Self::Error::InherentNotExpected);
205			};
206
207			if *call != expected_call {
208				return Err(Self::Error::IncorrectInherent);
209			}
210
211			Ok(())
212		}
213
214		fn is_inherent(call: &Self::Call) -> bool {
215			matches!(call, Call::register_changes { .. })
216		}
217
218		fn is_inherent_required(data: &InherentData) -> Result<Option<Self::Error>, Self::Error> {
219			match Self::decode_inherent_data(data) {
220				None => Ok(None),
221				Some(_) => Ok(Some(Self::Error::InherentMissing)),
222			}
223		}
224	}

226	impl<T: Config> Pallet<T> {
227		fn decode_inherent_data(
228			data: &InherentData,
229		) -> Option<alloc::vec::Vec<GovernedMapChangeV1>> {
230			data.get_data::<alloc::vec::Vec<GovernedMapChangeV1>>(&INHERENT_IDENTIFIER)
231				.expect("Governed Map inherent data is not encoded correctly")
232		}
233
234		fn create_inherent_or_err(data: &InherentData) -> Result<Option<Call<T>>, InherentError> {
235			use InherentError::*;
236
237			let Some(raw_changes) = Self::decode_inherent_data(data) else { return Ok(None) };
238
239			if raw_changes.len() > T::MaxChanges::get() as usize {
240				return Err(TooManyChanges);
241			}
242
243			let mut changes = Changes::<T>::new();
244			for GovernedMapChangeV1 { key, new_value } in raw_changes {
245				let new_value = match new_value {
246					Some(new_value) => Some(Self::bound_value(&key, new_value)?),
247					None => None,
248				};
249				let key = Self::bound_key(&key)?;
250				changes.try_push((key, new_value)).expect("Number of changes is below maximum");
251			}
252			Ok(Some(Call::register_changes { changes }))
253		}
254
255		fn bound_key(key: &str) -> Result<MapKey<T>, InherentError> {
256			key.try_into().map_err(|_| InherentError::KeyExceedsBounds(key.into()))
257		}
258
259		fn bound_value(key: &str, value: ByteString) -> Result<MapValue<T>, InherentError> {
260			(value.0.clone().try_into())
261				.map_err(|_| InherentError::ValueExceedsBounds(key.into(), value))
262		}
263	}

265	#[pallet::call]
266	impl<T: Config> Pallet<T> {
267		/// Inherent to register any changes in the state of the Governed Map on Cardano compared to the state currently stored in the pallet.
268		#[pallet::call_index(0)]
269		#[pallet::weight((T::WeightInfo::register_changes(changes.len() as u32), DispatchClass::Mandatory))]
270		pub fn register_changes(origin: OriginFor<T>, changes: Changes<T>) -> DispatchResult {
271			ensure_none(origin)?;
272
273			log::info!("üíæ Registering {} Governed Map changes", changes.len(),);
274
275			for (key, value) in changes {
276				let old_value = Mapping::<T>::get(&key);
277				Mapping::<T>::set(&key, value.clone());
278				T::OnGovernedMappingChange::on_governed_mapping_change(key, value, old_value);
279			}
280
281			Ok(())
282		}
283
284		/// Changes the address of the Governed Map validator used for observation.
285		///
286		/// This extrinsic must be run either using `sudo` or some other chain governance mechanism.
287		#[pallet::call_index(1)]
288		#[pallet::weight((T::WeightInfo::set_main_chain_scripts(), DispatchClass::Normal))]
289		pub fn set_main_chain_scripts(
290			origin: OriginFor<T>,
291			new_main_chain_script: MainChainScriptsV1,
292		) -> DispatchResult {
293			T::MainChainScriptsOrigin::ensure_origin(origin)?;
294			MainChainScripts::<T>::put(new_main_chain_script.clone());
295			log::info!("üóÇÔ∏è Governed Map main chain scripts updated to {new_main_chain_script:?}");
296			Ok(())
297		}
298	}

300	impl<T: Config> Pallet<T> {
301		/// Returns the value under `key` or [None] otherwise.
302		pub fn get_key_value(key: MapKey<T>) -> Option<BoundedVec<u8, T::MaxValueLength>> {
303			Mapping::<T>::get(key)
304		}
305
306		/// Returns an iterator over all key-value pairs in the pallet storage.
307		pub fn get_all_key_value_pairs() -> impl Iterator<Item = (MapKey<T>, MapValue<T>)> {
308			Mapping::<T>::iter()
309		}
310
311		/// Returns an iterator over all key-value pairs in the pallet storage, using unbound types.
312		pub fn get_all_key_value_pairs_unbounded() -> impl Iterator<Item = (String, ByteString)> {
313			Self::get_all_key_value_pairs()
314				.map(|(key, value)| (key.to_string(), value.to_vec().into()))
315		}
316
317		/// Returns current pallet version.
318		pub fn get_version() -> u32 {
319			PALLET_VERSION
320		}
321
322		/// Returns the current main chain scripts
323		pub fn get_main_chain_scripts() -> Option<MainChainScriptsV1> {
